019package org.jdrupes.httpcodec.plugin;
020
021import org.jdrupes.httpcodec.Decoder;
022import org.jdrupes.httpcodec.Encoder;
023import org.jdrupes.httpcodec.MessageHeader;
024import org.jdrupes.httpcodec.ResponseDecoder;
025import org.jdrupes.httpcodec.protocols.http.HttpResponse;
026
027/**
028 * The base class for a protocol that the HTTP connection
029 * can be upgraded to.
030 */
031public abstract class ProtocolProvider {
032
033        /**
034         * Checks if the plugin supports the given protocol.
035         * 
036         * @param protocol the protocol in question
037         * @return the result
038         */
039        public abstract boolean supportsProtocol(String protocol);
040
041        /**
042         * Add any required information to the "switching protocols" response
043         * that is sent as the last package of the HTTP and starts the usage
044         * of the new protocol.
045         * 
046         * @param response the response
047         */
048        public abstract void augmentInitialResponse(HttpResponse response);
049        
050        /**
051         * Creates a new request encoder for the protocol.
052         * 
053         * @param protocol the protocol, which must be supported by this plugin
054         * @return the request encoder
055         */
056        public abstract Encoder<?>      createRequestEncoder(String protocol);
057        
058        /**
059         * Creates a new request decoder for the protocol.
060         * 
061         * @param protocol the protocol, which must be supported by this plugin
062         * @return the request decoder
063         */
064        public abstract Decoder<?, ?> createRequestDecoder(String protocol);
065        
066        /**
067         * Creates a new response encoder for the protocol.
068         * 
069         * @param protocol the protocol, which must be supported by this plugin
070         * @return the response encoder
071         */
072        public abstract Encoder<? extends MessageHeader>
073                createResponseEncoder(String protocol);
074        
075        /**
076         * Creates a new response decoder for the protocol.
077         * 
078         * @param protocol the protocol, which must be supported by this plugin
079         * @return the response decoder
080         */
081        public abstract ResponseDecoder<?, ?>
082                createResponseDecoder(String protocol);
083        
084        
085}
