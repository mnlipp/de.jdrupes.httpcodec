019package org.jdrupes.httpcodec.protocols.http.fields;
020
021import java.text.ParseException;
022import java.util.ArrayList;
023import java.util.List;
024
025import org.jdrupes.httpcodec.types.CommentedValue;
026import org.jdrupes.httpcodec.types.CommentedValue.CommentedValueConverter;
027import org.jdrupes.httpcodec.types.Converters;
028import org.jdrupes.httpcodec.types.ListConverter;
029
030/**
031 * Represents products descriptions as used in the `User-Agent`
032 * or `Server` field.
033 */
034public class HttpProductsDescriptionField 
035        extends HttpListField<CommentedValue<String>> {
036
037        private static final ProductsDescriptionConverter DESCRIPTIONS_CONVERTER 
038                = new ProductsDescriptionConverter();
039        
040        public HttpProductsDescriptionField(String name, 
041                        List<CommentedValue<String>> items) {
042                super(name, items, DESCRIPTIONS_CONVERTER);
043        }
044
045        /**
046         * Creates a new list field with the elements obtained by parsing the given
047         * String.
048         * 
049         * @param text the string to parse
050         * @return the result
051         * @throws ParseException if the input violates the field format
052         */
053        public static HttpProductsDescriptionField fromString(
054                        String name, String text) throws ParseException {
055                return new HttpProductsDescriptionField(name, 
056                                DESCRIPTIONS_CONVERTER.fromFieldValue(text));
057        }
058        
059        public static class ProductsDescriptionConverter 
060                extends ListConverter<CommentedValue<String>> {
061
062                private static final CommentedValueConverter<String> ITEM_CONVERTER
063                        = new CommentedValueConverter<>(Converters.STRING_CONVERTER);
064                
065                public ProductsDescriptionConverter() {
066                        super(ITEM_CONVERTER, " ");
067                }
068
069                /* (non-Javadoc)
070                 * @see org.jdrupes.httpcodec.types.Converter#fromFieldValue(java.lang.String)
071                 */
072                @Override
073                public List<CommentedValue<String>> fromFieldValue(String text)
074                        throws ParseException {
075                        List<CommentedValue<String>> result = new ArrayList<>();
076                        int pos = 0;
077                        while (pos < text.length()) {
078                                int length = HttpField.tokenLength(text, pos);
079                                if (length == 0) {
080                                        throw new ParseException(
081                                                        "Must start with token: " + text, pos);
082                                }
083                                String product = text.substring(pos, pos + length);
084                                pos += length;
085                                if (pos < text.length() && text.charAt(pos) == '/') {
086                                        pos += 1;
087                                        length = HttpField.tokenLength(text, pos);
088                                        if (length == 0) {
089                                                throw new ParseException(
090                                                                "Token expected: " + text, pos);
091                                        }
092                                        product = product + text.substring(pos - 1, pos + length);
093                                        pos += length;
094                                }
095                                List<String> comments = new ArrayList<>();
096                                while (pos < text.length()) {
097                                        length = HttpField.whiteSpaceLength(text, pos);
098                                        if (length == 0) {
099                                                throw new ParseException(
100                                                                "Whitespace expected: " + text, pos);
101                                        }
102                                        pos += length;
103                                        if (text.charAt(pos) != '(') {
104                                                break;
105                                        }
106                                        length = HttpField.commentLength(text, pos);
107                                        if (text.charAt(pos + length - 1) != ')') {
108                                                throw new ParseException(
109                                                                "Comment end expected: " + text, pos + length - 1);
110                                        }
111                                        comments.add(text.substring(pos + 1, pos + length - 1));
112                                        pos += length;
113                                }
114                                result.add(new CommentedValue<String>(product, 
115                                                comments.size() == 0 ? null
116                                                : comments.toArray(new String[comments.size()])));
117                        }
118                        return result;
119                }
120                
121        }
122}
