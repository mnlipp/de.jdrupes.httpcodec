018package org.jdrupes.httpcodec.protocols.http.fields;
019
020import java.text.ParseException;
021import java.util.Arrays;
022
023import org.jdrupes.httpcodec.protocols.http.HttpMessageHeader;
024
025/**
026 * An HTTP field value that consists of a comma separated list of 
027 * strings. The class provides a "list of strings" view
028 * of the values.
029 * 
030 * @author Michael N. Lipp
031 */
032public class HttpStringListField extends HttpListField<String>
033        implements Cloneable {
034
035        /**
036         * Creates a new object with the given field name and no elements. Note 
037         * that in this
038         * initial state, the field is invalid and no string representation
039         * can be generated. This constructor must be followed by method invocations
040         * that add values.
041         * 
042         * @param name the field name
043         */
044        public HttpStringListField(String name) {
045                super(name);
046                reset();
047        }
048
049        /**
050         * Creates the new object from the given values.
051         * 
052         * @param name the field name
053         * @param value the first value
054         * @param values more values
055         */
056        public HttpStringListField(String name, String value, String... values) {
057                super(name);
058                add(value);
059                addAll(Arrays.asList(values));
060        }
061
062        /**
063         * Creates a new object with the given field name and unparsed value.
064         * Derived classes must implement this constructor to support automatic
065         * conversion (see {@link HttpMessageHeader#getField(Class, String)}).
066         * 
067         * @param name the field name
068         * @param unparsedValue the unparsed value
069         * @param unparsed used to distinguish constructors
070         * @throws ParseException if the input violates the field format
071         */
072        public HttpStringListField(String name, String unparsedValue, 
073                        boolean unparsed) throws ParseException {
074                super(name, unparsedValue);
075                while (true) {
076                        String element = nextElement();
077                        if (element == null) {
078                                break;
079                        }
080                        add(unquote(element));
081                }
082        }
083
084        /**
085         * Creates a new object with the elements obtained by parsing the given
086         * String.
087         * 
088         * @param name the field name
089         * @param s the string to parse
090         * @return the result
091         * @throws ParseException if the input violates the field format
092         */
093        public static HttpStringListField fromString(String name, String s) 
094                        throws ParseException {
095                return new HttpStringListField(name, s, true);
096        }
097
098        /* (non-Javadoc)
099         * @see java.lang.Object#clone()
100         */
101        @Override
102        public HttpStringListField clone() {
103                return (HttpStringListField)super.clone();
104        }
105
106        /* (non-Javadoc)
107         * @see org.jdrupes.httpcodec.fields.HttpListField#elementAsString(java.lang.Object)
108         */
109        @Override
110        protected String elementToString(String element) {
111                return quoteIfNecessary(element);
112        }
113
114        /**
115         * Returns whether the list contains the given value, ignoring
116         * differences in the cases of the letters.
117         * 
118         * @param value the value to compare with
119         * @return the result
120         */
121        public boolean containsIgnoreCase(String value) {
122                for (String s: getValue()) {
123                        if (s.equalsIgnoreCase(value)) {
124                                return true;
125                        }
126                }
127                return false;
128        }
129        
130        /**
131         * Removes all strings equal to the given value, ignoring
132         * differences in the cases of the letters.
133         * 
134         * @param value the value to compare with
135         */
136        public void removeIgnoreCase(String value) {
137                removeIf(s -> s.equalsIgnoreCase(value));
138        }
139}
