018package org.jdrupes.httpcodec.protocols.http.fields;
019
020import java.net.HttpCookie;
021import java.text.ParseException;
022import java.util.Optional;
023
024/**
025 * @author Michael N. Lipp
026 *
027 */
028public class HttpCookieListField extends HttpListField<HttpCookie> {
029
030        /**
031         * Creates a new object with the field name "Cookie" and the given unparsed
032         * value.
033         * 
034         * @param s the unparsed value
035         */
036        protected HttpCookieListField(String s) {
037                super(HttpField.COOKIE, s);
038        }
039
040        /**
041         * Creates a new object with the elements obtained by parsing the given
042         * String.
043         * 
044         * @param s the string to parse
045         * @return the result
046         * @throws ParseException if the input violates the field format
047         */
048        public static HttpCookieListField fromString(String s) 
049                        throws ParseException {
050                HttpCookieListField result = new HttpCookieListField(s);
051                while (true) {
052                        String element = result.nextElement();
053                        if (element == null) {
054                                break;
055                        }
056                        try {
057                                result.addAll(HttpCookie.parse(element));
058                        } catch (IllegalArgumentException e) {
059                                throw new ParseException(element, 0);
060                        }
061                }
062                return result;
063        }
064
065        /* (non-Javadoc)
066         * @see org.jdrupes.httpcodec.fields.HttpListField#getSeparator()
067         */
068        @Override
069        protected char getDelimiter() {
070                return ';';
071        }
072
073        /* (non-Javadoc)
074         * @see org.jdrupes.httpcodec.fields.HttpListField#elementToString(java.lang.Object)
075         */
076        @Override
077        protected String elementToString(HttpCookie element) {
078                return element.toString();
079        }
080
081        /**
082         * Returns the value for the cookie with the given name.
083         * 
084         * @param name the name
085         * @return the value if a cookie with the given name exists
086         */
087        public Optional<String> valueForName(String name) {
088                return stream().filter(cookie -> cookie.getName().equals(name))
089                                .findFirst().map(HttpCookie::getValue);
090        }
091}
