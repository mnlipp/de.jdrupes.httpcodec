030/**
031 * A base class for all kinds of header field values.
032 * 
033 * @param <T> the type of the header field's value 
034 * 
035 * @see "[MessageHeaders](https://www.iana.org/assignments/message-headers/message-headers.xhtml)"
036 */

135        /**
136         * Creates a new representation of a header field. For fields with
137         * a constant definition in this class, the name is normalized.
138         * 
139         * @param name the field name
140         * @param converter the converter
141         */
142        protected HttpField(String name, Converter<T> converter) {
143                this.name = fieldNameMap.getOrDefault(name, name);
144                this.converter = converter;
145        }
146
147        /**
148         * Creates a new representation of a header field with the 
149         * given value and converter. For fields with a 
150         * constant definition in this class, the name is normalized.
151         * 
152         * @param name the field name
153         * @param value the value
154         * @param converter the converter
155         */
156        protected HttpField(String name, T value, Converter<T> converter) {
157                this(name, converter);
158                this.value = value;
159        }

161        /**
162         * Returns an HttpField that represents the given header field, using the
163         * best matching derived class in this package. Works for all well known
164         * field names, i.e. the field names defined as constants in this class. If
165         * the field name is unknown, the result will be of type
166         * {@link HttpStringField}.
167         * 
168         * @param fieldName
169         *            the field name
170         * @param fieldValue
171         *            the field value
172         * @return a typed representation
173         * @throws ParseException if the input violates the field format
174         */
175        public static HttpField<?> fromString(String fieldName,
176                String fieldValue) throws ParseException {
177                String normalizedFieldName = fieldNameMap
178                                .getOrDefault(fieldName, fieldName);
179                switch (normalizedFieldName) {
180                case ACCEPT:
181                        return HttpWeightedListField.fromString(
182                                        fieldName, fieldValue, Converters.MEDIA_RANGE_CONVERTER);
183                case ACCEPT_CHARSET:
184                        return HttpWeightedListField.fromString(
185                                        fieldName, fieldValue, new ParameterizedValueConverter<>(
186                                                        Converters.STRING_CONVERTER));
187                case ACCEPT_ENCODING:
188                        return HttpWeightedListField.fromString(
189                                        fieldName, fieldValue, new ParameterizedValueConverter<>(
190                                                        Converters.STRING_CONVERTER));
191                case ACCEPT_LANGUAGE:
192                        return HttpWeightedListField.fromString(
193                                        fieldName, fieldValue, new ParameterizedValueConverter<>(
194                                                        Converters.LANGUAGE_CONVERTER));
195                case ALLOW:
196                        return HttpStringListField.fromString(fieldName, fieldValue);
197                case COOKIE:
198                        return HttpCookieListField.fromString(fieldValue);
199                case CONNECTION:
200                        return HttpStringListField.fromString(fieldName, fieldValue);
201                case CONTENT_LENGTH:
202                        return HttpContentLengthField.fromString(fieldValue);
203                case CONTENT_LOCATION:
204                        return HttpUriField.fromString(fieldName, fieldValue); 
205                case CONTENT_TYPE:
206                        return HttpMediaTypeField.fromString(fieldName, fieldValue);
207                case DATE:
208                        return HttpDateTimeField.fromString(fieldName, fieldValue);
209                case IF_MATCH:
210                        return HttpStringListField.fromString(fieldName, fieldValue);
211                case IF_MODIFIED_SINCE:
212                        return HttpDateTimeField.fromString(fieldName, fieldValue);
213                case IF_NONE_MATCH:
214                        return HttpStringListField.fromString(fieldName, fieldValue);
215                case IF_UNMODIFIED_SINCE:
216                        return HttpDateTimeField.fromString(fieldName, fieldValue);
217                case LAST_MODIFIED:
218                        return HttpDateTimeField.fromString(fieldName, fieldValue);
219                case LOCATION:
220                        return HttpUriField.fromString(fieldName, fieldValue); 
221                case MAX_FORWARDS:
222                        return HttpIntField.fromString(fieldName, fieldValue); 
223                case RETRY_AFTER:
224                        return dateOrSpanField(fieldName, fieldValue);
225                case SERVER:
226                        return HttpProductsDescriptionField
227                                        .fromString(fieldName, fieldValue);
228                case SET_COOKIE:
229                        return HttpSetCookieListField.fromString(fieldValue);
230                case TRAILER:
231                        return HttpStringListField.fromString(fieldName, fieldValue);
232                case TRANSFER_ENCODING:
233                        return HttpStringListField.fromString(fieldName, fieldValue);
234                case UPGRADE:
235                        return HttpStringListField.fromString(fieldName, fieldValue);
236                case USER_AGENT:
237                        return HttpProductsDescriptionField
238                                        .fromString(fieldName, fieldValue);
239                case VIA:
240                        return HttpStringListField.fromString(fieldName, fieldValue);
241                default:
242                        return HttpStringField.fromString(fieldName, fieldValue);
243                }
244        }
245        
246        private static HttpField<?> dateOrSpanField(String name, String value) 
247                        throws ParseException {
248                if (Character.isDigit(value.charAt(0))) {
249                        return HttpIntField.fromString(name, value);
250                }
251                return HttpDateTimeField.fromString(name, value);
252        }

315        @Override
316        public String toString() {
317                StringBuilder result = new StringBuilder();
318                result.append(getClass().getSimpleName());
319                result.append(" [");
320                result.append(asHeaderField().replace("\r\n", " CRLF "));
321                result.append("]");
322                return result.toString();

325        /**
326         * Determines the length of a token.
327         * 
328         * @param text the text to parse
329         * @param startPos the start position
330         * @return the length of the token
331         */
332        public static int tokenLength(String text, int startPos) {
333                int pos = startPos;
334                while (pos < text.length()
335                                && HttpConstants.TOKEN_CHARS.indexOf(text.charAt(pos)) >= 0) {
336                        pos += 1;
337                }
338                return pos - startPos;
339        }
340
341        /**
342         * Determines the length of a white space sequence. 
343         * 
344         * @param text the test to parse 
345         * @param startPos the start position
346         * @return the length of the white space sequence
347         */
348        public static int whiteSpaceLength(String text, int startPos) {
349                int pos = startPos;
350                while (pos < text.length()) {
351                        switch (text.charAt(pos)) {
352                        case ' ':
353                                // fall through
354                        case '\t':
355                                pos += 1;
356                                continue;
357                                
358                        default:
359                                break;
360                        }
361                        break;
362                }
363                return pos - startPos;
364        }
365        
366        /**
367         * Determines the length of a comment.
368         * 
369         * @param text the text to parse
370         * @param startPos the staring position (must be the position of the
371         * opening brace)
372         * @return the length of the comment
373         */
374        public static int commentLength(String text, int startPos) {
375                int pos = startPos + 1;
376                while (pos < text.length()) {
377                        switch(text.charAt(pos)) {
378                        case ')':
379                                return pos - startPos + 1;
380                                
381                        case '(':
382                                pos += commentLength(text, pos);
383                                break;
384                                
385                        case '\\':
386                                pos = Math.min(pos + 2, text.length());
387                                break;
388                                
389                        default:
390                                pos += 1;
391                                break;
392                        }
393                }
394                return pos - startPos;
395        }
