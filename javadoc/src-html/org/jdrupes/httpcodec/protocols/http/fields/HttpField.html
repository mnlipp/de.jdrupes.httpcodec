018package org.jdrupes.httpcodec.protocols.http.fields;
019
020import java.text.ParseException;
021import java.util.Map;
022import java.util.TreeMap;
023
024import org.jdrupes.httpcodec.protocols.http.HttpConstants;
025
026/**
027 * A base class for all kinds of header field values.
028 * 
029 * @author Michael N. Lipp
030 */
031public abstract class HttpField<T> implements Cloneable {
032
033        final public static String COOKIE = "Cookie";
034        final public static String CONNECTION = "Connection";
035        final public static String CONTENT_LENGTH = "Content-Length";
036        final public static String CONTENT_TYPE = "Content-Type";
037        final public static String DATE = "Date";
038        final public static String HOST = "Host";
039        final public static String SET_COOKIE = "Set-Cookie";
040        final public static String TE = "TE";
041        final public static String TRAILER = "Trailer";
042        final public static String TRANSFER_ENCODING = "Transfer-Encoding";
043        final public static String UPGRADE = "Upgrade";
044        final public static String VIA = "Via";
045        
046        private static Map<String, String> fieldNameMap = new TreeMap<>(
047                String.CASE_INSENSITIVE_ORDER);
048        static {
049                fieldNameMap.put(COOKIE, COOKIE);
050                fieldNameMap.put(CONNECTION, CONNECTION);
051                fieldNameMap.put(CONTENT_LENGTH, CONTENT_LENGTH);
052                fieldNameMap.put(CONTENT_TYPE, CONTENT_TYPE);
053                fieldNameMap.put(DATE, DATE);
054                fieldNameMap.put(HOST, HOST);
055                fieldNameMap.put(SET_COOKIE, SET_COOKIE);
056                fieldNameMap.put(TE, TE);
057                fieldNameMap.put(TRAILER, TRAILER);
058                fieldNameMap.put(TRANSFER_ENCODING, TRANSFER_ENCODING);
059                fieldNameMap.put(UPGRADE, UPGRADE);
060                fieldNameMap.put(VIA, VIA);
061        }
062        
063        /**
064         * Returns an HttpField that represents the given header field, using the
065         * best matching derived class in this package. Works for all well known
066         * field names, i.e. the field names defined as constants in this class. If
067         * the field name is unknown, the result will be of type
068         * {@link HttpStringField}.
069         * 
070         * @param fieldName
071         *            the field name
072         * @param fieldValue
073         *            the field value
074         * @return a typed representation
075         * @throws ParseException if the input violates the field format
076         */
077        public static HttpField<?> fromString(String fieldName,
078                String fieldValue) throws ParseException {
079                String normalizedFieldName = fieldNameMap
080                                .getOrDefault(fieldName, fieldName);
081                switch (normalizedFieldName) {
082                case HttpField.COOKIE:
083                        return HttpCookieListField.fromString(fieldValue);
084                case HttpField.CONNECTION:
085                        return HttpStringListField.fromString(fieldName, fieldValue);
086                case HttpField.CONTENT_LENGTH:
087                        return HttpContentLengthField.fromString(fieldValue);
088                case HttpField.CONTENT_TYPE:
089                        return HttpMediaTypeField.fromString(fieldName, fieldValue);
090                case HttpField.DATE:
091                        return HttpDateField.fromString(fieldName, fieldValue);
092                case HttpField.SET_COOKIE:
093                        return HttpSetCookieListField.fromString(fieldValue);
094                case HttpField.TRAILER:
095                        return HttpStringListField.fromString(fieldName, fieldValue);
096                case HttpField.TRANSFER_ENCODING:
097                        return HttpStringListField.fromString(fieldName, fieldValue);
098                case HttpField.UPGRADE:
099                        return HttpStringListField.fromString(fieldName, fieldValue);
100                case HttpField.VIA:
101                        return HttpStringListField.fromString(fieldName, fieldValue);
102                default:
103                        return HttpStringField.fromString(fieldName, fieldValue);
104                }
105        }
106        
107        final private String name;
108        
109        /**
110         * Creates a new representation of a header field value. For fields with
111         * a constant definition in this class, the name is normalized.
112         * 
113         * @param name the field name
114         */
115        protected HttpField(String name) {
116                this.name = fieldNameMap.getOrDefault(name, name);
117        }
118
119        /* (non-Javadoc)
120         * @see java.lang.Object#clone()
121         */
122        @Override
123        @SuppressWarnings("unchecked")
124        public HttpField<T> clone() {
125                try {
126                        return (HttpField<T>)super.clone();
127                } catch (CloneNotSupportedException e) {
128                        return null;
129                }
130        }
131
132        /**
133         * Returns the header field name.
134         * 
135         * @return the name
136         */
137        public String getName() {
138                return name;
139        }
140
141        /**
142         * Returns the header field's parsed value.
143         * 
144         * @return the field's value
145         */
146        public abstract T getValue();
147        
148        /**
149         * Returns the string representation of this field's value.
150         * 
151         * @return the field value as string
152         */
153        public abstract String asFieldValue();
154        
155        /**
156         * Returns the string representation of this header field as it appears in
157         * an HTTP message. Note that the returned string may span several
158         * lines (may contain CRLF).
159         * 
160         * @return the field as it occurs in a header
161         */
162        public String asHeaderField() {
163                return getName() + ": " + asFieldValue();
164        }
165        
166        /* (non-Javadoc)
167         * @see java.lang.Object#toString()
168         */
169        @Override
170        public String toString() {
171                StringBuilder s = new StringBuilder();
172                s.append(getClass().getSimpleName());
173                s.append(" [");
174                s.append(asHeaderField().replace("\r\n", " CRLF "));
175                s.append("]");
176                return s.toString();
177        }
178
179        /**
180         * If the value is double quoted, remove the quotes and escape
181         * characters.
182         * 
183         * @param value the value to unquote
184         * @return the unquoted value
185         * @throws ParseException if the input violates the field format
186         */
187        public static String unquote(String value) throws ParseException {
188                // RFC 7230 3.2.6
189                if (value.length() == 0 || value.charAt(0) != '\"') {
190                        return value;
191                }
192                int startPosition = 1;
193                int position = 1;
194                try {
195                        StringBuilder result = new StringBuilder();
196                        while (true) {
197                                char ch = value.charAt(position);
198                                switch (ch) {
199                                case '\\':
200                                        result.append(value.substring(startPosition, position));
201                                        position += 1;
202                                        result.append(value.charAt(position));
203                                        position += 1;
204                                        startPosition = position;
205                                        continue;
206                                case '\"':
207                                        if (position != value.length() - 1) {
208                                                throw new ParseException(value, position);
209                                        }
210                                        result.append(value.substring(startPosition, position));
211                                        return result.toString();
212                                default:
213                                        position += 1;
214                                        continue;
215                                }
216                        }
217                } catch (IndexOutOfBoundsException e) {
218                        throw new ParseException(value, position);
219                }
220        }
221
222        /**
223         * Returns the given string as double quoted string if necessary.
224         * 
225         * @param value the value to quote if necessary
226         * @return the result
227         */
228        public static String quoteIfNecessary(String value) {
229                // RFC 7230 3.2.6
230                StringBuilder result = new StringBuilder();
231                int position = 0;
232                boolean needsQuoting = false;
233                result.append('"');
234                while (position < value.length()) {
235                        char ch = value.charAt(position++);
236                        if (!needsQuoting && HttpConstants.TOKEN_CHARS.indexOf(ch) < 0) {
237                                needsQuoting = true;
238                        }
239                        switch(ch) {
240                        case '"':
241                        case '\\':
242                                result.append('\\');
243                        default:
244                                result.append(ch);
245                        }
246                }
247                result.append('\"');
248                if (needsQuoting) {
249                        return result.toString();
250                }
251                return value;
252        }
