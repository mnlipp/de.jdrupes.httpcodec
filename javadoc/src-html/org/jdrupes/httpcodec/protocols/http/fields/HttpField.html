019package org.jdrupes.httpcodec.protocols.http.fields;
020
021import java.text.ParseException;
022import java.util.Map;
023import java.util.TreeMap;
024
025import org.jdrupes.httpcodec.types.Converter;
026
027/**
028 * A base class for all kinds of header field values.
029 * 
030 * @see "[MessageHeaders](https://www.iana.org/assignments/message-headers/message-headers.xhtml)"
031 */
032public abstract class HttpField<T> {
033
034        // https://tools.ietf.org/html/rfc7231#section-5.3.2
035        public static final String ACCEPT = "Accept";
036        public static final String COOKIE = "Cookie";
037        // https://tools.ietf.org/html/rfc7230#section-6.1
038        public static final String CONNECTION = "Connection";
039        // https://tools.ietf.org/html/rfc7230#section-3.3.2
040        public static final String CONTENT_LENGTH = "Content-Length";
041        public static final String CONTENT_TYPE = "Content-Type";
042        public static final String DATE = "Date";
043        public static final String ETAG = "ETag";
044        // https://tools.ietf.org/html/rfc7230#section-5.4
045        public static final String HOST = "Host";
046        public static final String IF_MATCH = "If-Match";
047        public static final String IF_NONE_MATCH = "If-None-Match";
048        public static final String IF_MODIFIED_SINCE = "If-Modified-Since";
049        public static final String IF_UNMODIFIED_SINCE = "If-Unmodified-Since";
050        public static final String LAST_MODIFIED = "Last-Modified";
051        public static final String SET_COOKIE = "Set-Cookie";
052        // https://tools.ietf.org/html/rfc7230#section-4.3
053        public static final String TE = "TE";
054        // https://tools.ietf.org/html/rfc7230#section-4.4
055        public static final String TRAILER = "Trailer";
056        // https://tools.ietf.org/html/rfc7230#section-3.3.1
057        public static final String TRANSFER_ENCODING = "Transfer-Encoding";
058        // https://tools.ietf.org/html/rfc7230#section-6.7
059        public static final String UPGRADE = "Upgrade";
060        // https://tools.ietf.org/html/rfc7230#section-5.7.1
061        public static final String VIA = "Via";
062        
063        private static Map<String, String> fieldNameMap = new TreeMap<>(
064                String.CASE_INSENSITIVE_ORDER);
065        
066        static {
067                fieldNameMap.put(ACCEPT, ACCEPT);
068                fieldNameMap.put(COOKIE, COOKIE);
069                fieldNameMap.put(CONNECTION, CONNECTION);
070                fieldNameMap.put(CONTENT_LENGTH, CONTENT_LENGTH);
071                fieldNameMap.put(CONTENT_TYPE, CONTENT_TYPE);
072                fieldNameMap.put(DATE, DATE);
073                fieldNameMap.put(ETAG, ETAG);
074                fieldNameMap.put(HOST, HOST);
075                fieldNameMap.put(IF_MATCH, IF_MATCH);
076                fieldNameMap.put(IF_NONE_MATCH, IF_NONE_MATCH);
077                fieldNameMap.put(IF_MODIFIED_SINCE, IF_MODIFIED_SINCE);
078                fieldNameMap.put(IF_UNMODIFIED_SINCE, IF_UNMODIFIED_SINCE);
079                fieldNameMap.put(LAST_MODIFIED, LAST_MODIFIED);
080                fieldNameMap.put(SET_COOKIE, SET_COOKIE);
081                fieldNameMap.put(TE, TE);
082                fieldNameMap.put(TRAILER, TRAILER);
083                fieldNameMap.put(TRANSFER_ENCODING, TRANSFER_ENCODING);
084                fieldNameMap.put(UPGRADE, UPGRADE);
085                fieldNameMap.put(VIA, VIA);
086        }
087        
088        private final String name;
089        private T value;
090        private Converter<T> converter;
091        
092        /**
093         * Creates a new representation of a header field. For fields with
094         * a constant definition in this class, the name is normalized.
095         * 
096         * @param name the field name
097         * @param converter the converter
098         */
099        protected HttpField(String name, Converter<T> converter) {
100                this.name = fieldNameMap.getOrDefault(name, name);
101                this.converter = converter;
102        }
103
104        /**
105         * Creates a new representation of a header field with the 
106         * given value and converter. For fields with a 
107         * constant definition in this class, the name is normalized.
108         * 
109         * @param name the field name
110         * @param value the value
111         * @param converter the converter
112         */
113        protected HttpField(String name, T value, Converter<T> converter) {
114                this(name, converter);
115                this.value = value;
116        }
117
118        /**
119         * Returns an HttpField that represents the given header field, using the
120         * best matching derived class in this package. Works for all well known
121         * field names, i.e. the field names defined as constants in this class. If
122         * the field name is unknown, the result will be of type
123         * {@link HttpStringField}.
124         * 
125         * @param fieldName
126         *            the field name
127         * @param fieldValue
128         *            the field value
129         * @return a typed representation
130         * @throws ParseException if the input violates the field format
131         */
132        public static HttpField<?> fromString(String fieldName,
133                String fieldValue) throws ParseException {
134                String normalizedFieldName = fieldNameMap
135                                .getOrDefault(fieldName, fieldName);
136                switch (normalizedFieldName) {
137                case HttpField.ACCEPT:
138                        return HttpMediaTypeListField.fromString(fieldName, fieldValue);
139                case HttpField.COOKIE:
140                        return HttpCookieListField.fromString(fieldValue);
141                case HttpField.CONNECTION:
142                        return HttpStringListField.fromString(fieldName, fieldValue);
143                case HttpField.CONTENT_LENGTH:
144                        return HttpContentLengthField.fromString(fieldValue);
145                case HttpField.CONTENT_TYPE:
146                        return HttpMediaTypeField.fromString(fieldName, fieldValue);
147                case HttpField.DATE:
148                        return HttpDateTimeField.fromString(fieldName, fieldValue);
149                case HttpField.IF_MATCH:
150                        return HttpStringListField.fromString(fieldName, fieldValue);
151                case HttpField.IF_MODIFIED_SINCE:
152                        return HttpDateTimeField.fromString(fieldName, fieldValue);
153                case HttpField.IF_NONE_MATCH:
154                        return HttpStringListField.fromString(fieldName, fieldValue);
155                case HttpField.IF_UNMODIFIED_SINCE:
156                        return HttpDateTimeField.fromString(fieldName, fieldValue);
157                case HttpField.LAST_MODIFIED:
158                        return HttpDateTimeField.fromString(fieldName, fieldValue);
159                case HttpField.SET_COOKIE:
160                        return HttpSetCookieListField.fromString(fieldValue);
161                case HttpField.TRAILER:
162                        return HttpStringListField.fromString(fieldName, fieldValue);
163                case HttpField.TRANSFER_ENCODING:
164                        return HttpStringListField.fromString(fieldName, fieldValue);
165                case HttpField.UPGRADE:
166                        return HttpStringListField.fromString(fieldName, fieldValue);
167                case HttpField.VIA:
168                        return HttpStringListField.fromString(fieldName, fieldValue);
169                default:
170                        return HttpStringField.fromString(fieldName, fieldValue);
171                }
172        }
173        
174        /**
175         * Returns the header field name.
176         * 
177         * @return the name
178         */
179        public String getName() {
180                return name;
181        }
182
183        /**
184         * Returns the header field's parsed value.
185         * 
186         * @return the field's value
187         */
188        public T getValue() {
189                return value;
190        }
191        
192        /**
193         * Sets the header field's value.
194         * 
195         * @param value the new value
196         * @return the field for easy chaining
197         */
198        public HttpField<T> setValue(T value) {
199                this.value = value;
200                return this;
201        }
202        
203        /**
204         * Returns the string representation of this field's value.
205         * 
206         * @return the field value as string
207         */
208        public String asFieldValue() {
209                return converter.asFieldValue(value);
210        }
211        
212        /**
213         * Returns the string representation of this header field as it appears in
214         * an HTTP message. Note that the returned string may span several
215         * lines (may contain CR/LF), but is has no trailing CR/LF.
216         * 
217         * @return the field as it occurs in a header
218         */
219        public String asHeaderField() {
220                return getName() + ": " + asFieldValue();
221        }
222        
223        /* (non-Javadoc)
224         * @see java.lang.Object#toString()
225         */
226        @Override
227        public String toString() {
228                StringBuilder result = new StringBuilder();
229                result.append(getClass().getSimpleName());
230                result.append(" [");
231                result.append(asHeaderField().replace("\r\n", " CRLF "));
232                result.append("]");
233                return result.toString();
234        }
235}
