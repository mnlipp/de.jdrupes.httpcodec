019package org.jdrupes.httpcodec.protocols.http.fields;
020
021import java.text.ParseException;
022import java.util.Map;
023import java.util.TreeMap;
024
025import org.jdrupes.httpcodec.types.Converter;
026import org.jdrupes.httpcodec.types.Converters;
027import org.jdrupes.httpcodec.types.ParameterizedValue.ParameterizedValueConverter;
028
029/**
030 * A base class for all kinds of header field values.
031 * 
032 * @param <T> the type of the header field's value 
033 * 
034 * @see "[MessageHeaders](https://www.iana.org/assignments/message-headers/message-headers.xhtml)"
035 */
036public abstract class HttpField<T> {
037
038        // https://tools.ietf.org/html/rfc7231#section-5.3.2
039        public static final String ACCEPT = "Accept";
040        // https://tools.ietf.org/html/rfc7231#section-5.3.3
041        public static final String ACCEPT_CHARSET = "Accept-Charset";
042        // https://tools.ietf.org/html/rfc7231#section-5.3.4
043        public static final String ACCEPT_ENCODING = "Accept-Encoding";
044        // https://tools.ietf.org/html/rfc7231#section-5.3.5
045        public static final String ACCEPT_LANGUAGE = "Accept-Language";
046        // https://tools.ietf.org/html/rfc7231#section-7.4.1
047        public static final String ALLOW = "Allow";
048        public static final String COOKIE = "Cookie";
049        // https://tools.ietf.org/html/rfc7230#section-6.1
050        public static final String CONNECTION = "Connection";
051        // https://tools.ietf.org/html/rfc7230#section-3.3.2
052        public static final String CONTENT_LENGTH = "Content-Length";
053        public static final String CONTENT_TYPE = "Content-Type";
054        public static final String DATE = "Date";
055        public static final String ETAG = "ETag";
056        // https://tools.ietf.org/html/rfc7230#section-5.4
057        public static final String HOST = "Host";
058        public static final String IF_MATCH = "If-Match";
059        public static final String IF_NONE_MATCH = "If-None-Match";
060        public static final String IF_MODIFIED_SINCE = "If-Modified-Since";
061        public static final String IF_UNMODIFIED_SINCE = "If-Unmodified-Since";
062        public static final String LAST_MODIFIED = "Last-Modified";
063        public static final String SET_COOKIE = "Set-Cookie";
064        // https://tools.ietf.org/html/rfc7230#section-4.3
065        public static final String TE = "TE";
066        // https://tools.ietf.org/html/rfc7230#section-4.4
067        public static final String TRAILER = "Trailer";
068        // https://tools.ietf.org/html/rfc7230#section-3.3.1
069        public static final String TRANSFER_ENCODING = "Transfer-Encoding";
070        // https://tools.ietf.org/html/rfc7230#section-6.7
071        public static final String UPGRADE = "Upgrade";
072        // https://tools.ietf.org/html/rfc7230#section-5.7.1
073        public static final String VIA = "Via";
074        
075        private static Map<String, String> fieldNameMap = new TreeMap<>(
076                String.CASE_INSENSITIVE_ORDER);
077        
078        static {
079                fieldNameMap.put(ACCEPT, ACCEPT);
080                fieldNameMap.put(ACCEPT_CHARSET, ACCEPT_CHARSET);
081                fieldNameMap.put(ACCEPT_ENCODING, ACCEPT_ENCODING);
082                fieldNameMap.put(ACCEPT_LANGUAGE, ACCEPT_LANGUAGE);
083                fieldNameMap.put(ALLOW, ALLOW);
084                fieldNameMap.put(COOKIE, COOKIE);
085                fieldNameMap.put(CONNECTION, CONNECTION);
086                fieldNameMap.put(CONTENT_LENGTH, CONTENT_LENGTH);
087                fieldNameMap.put(CONTENT_TYPE, CONTENT_TYPE);
088                fieldNameMap.put(DATE, DATE);
089                fieldNameMap.put(ETAG, ETAG);
090                fieldNameMap.put(HOST, HOST);
091                fieldNameMap.put(IF_MATCH, IF_MATCH);
092                fieldNameMap.put(IF_NONE_MATCH, IF_NONE_MATCH);
093                fieldNameMap.put(IF_MODIFIED_SINCE, IF_MODIFIED_SINCE);
094                fieldNameMap.put(IF_UNMODIFIED_SINCE, IF_UNMODIFIED_SINCE);
095                fieldNameMap.put(LAST_MODIFIED, LAST_MODIFIED);
096                fieldNameMap.put(SET_COOKIE, SET_COOKIE);
097                fieldNameMap.put(TE, TE);
098                fieldNameMap.put(TRAILER, TRAILER);
099                fieldNameMap.put(TRANSFER_ENCODING, TRANSFER_ENCODING);
100                fieldNameMap.put(UPGRADE, UPGRADE);
101                fieldNameMap.put(VIA, VIA);
102        }
103        
104        private final String name;
105        private T value;
106        private Converter<T> converter;
107        
108        /**
109         * Creates a new representation of a header field. For fields with
110         * a constant definition in this class, the name is normalized.
111         * 
112         * @param name the field name
113         * @param converter the converter
114         */
115        protected HttpField(String name, Converter<T> converter) {
116                this.name = fieldNameMap.getOrDefault(name, name);
117                this.converter = converter;
118        }
119
120        /**
121         * Creates a new representation of a header field with the 
122         * given value and converter. For fields with a 
123         * constant definition in this class, the name is normalized.
124         * 
125         * @param name the field name
126         * @param value the value
127         * @param converter the converter
128         */
129        protected HttpField(String name, T value, Converter<T> converter) {
130                this(name, converter);
131                this.value = value;
132        }
133
134        /**
135         * Returns an HttpField that represents the given header field, using the
136         * best matching derived class in this package. Works for all well known
137         * field names, i.e. the field names defined as constants in this class. If
138         * the field name is unknown, the result will be of type
139         * {@link HttpStringField}.
140         * 
141         * @param fieldName
142         *            the field name
143         * @param fieldValue
144         *            the field value
145         * @return a typed representation
146         * @throws ParseException if the input violates the field format
147         */
148        public static HttpField<?> fromString(String fieldName,
149                String fieldValue) throws ParseException {
150                String normalizedFieldName = fieldNameMap
151                                .getOrDefault(fieldName, fieldName);
152                switch (normalizedFieldName) {
153                case HttpField.ACCEPT:
154                        return HttpWeightedListField.fromString(
155                                        fieldName, fieldValue, Converters.MEDIA_RANGE_CONVERTER);
156                case HttpField.ACCEPT_CHARSET:
157                        return HttpWeightedListField.fromString(
158                                        fieldName, fieldValue, new ParameterizedValueConverter<>(
159                                                        Converters.STRING_CONVERTER));
160                case HttpField.ACCEPT_ENCODING:
161                        return HttpWeightedListField.fromString(
162                                        fieldName, fieldValue, new ParameterizedValueConverter<>(
163                                                        Converters.STRING_CONVERTER));
164                case HttpField.ACCEPT_LANGUAGE:
165                        return HttpWeightedListField.fromString(
166                                        fieldName, fieldValue, new ParameterizedValueConverter<>(
167                                                        Converters.LANGUAGE_CONVERTER));
168                case HttpField.ALLOW:
169                        return HttpStringListField.fromString(fieldName, fieldValue);
170                case HttpField.COOKIE:
171                        return HttpCookieListField.fromString(fieldValue);
172                case HttpField.CONNECTION:
173                        return HttpStringListField.fromString(fieldName, fieldValue);
174                case HttpField.CONTENT_LENGTH:
175                        return HttpContentLengthField.fromString(fieldValue);
176                case HttpField.CONTENT_TYPE:
177                        return HttpMediaTypeField.fromString(fieldName, fieldValue);
178                case HttpField.DATE:
179                        return HttpDateTimeField.fromString(fieldName, fieldValue);
180                case HttpField.IF_MATCH:
181                        return HttpStringListField.fromString(fieldName, fieldValue);
182                case HttpField.IF_MODIFIED_SINCE:
183                        return HttpDateTimeField.fromString(fieldName, fieldValue);
184                case HttpField.IF_NONE_MATCH:
185                        return HttpStringListField.fromString(fieldName, fieldValue);
186                case HttpField.IF_UNMODIFIED_SINCE:
187                        return HttpDateTimeField.fromString(fieldName, fieldValue);
188                case HttpField.LAST_MODIFIED:
189                        return HttpDateTimeField.fromString(fieldName, fieldValue);
190                case HttpField.SET_COOKIE:
191                        return HttpSetCookieListField.fromString(fieldValue);
192                case HttpField.TRAILER:
193                        return HttpStringListField.fromString(fieldName, fieldValue);
194                case HttpField.TRANSFER_ENCODING:
195                        return HttpStringListField.fromString(fieldName, fieldValue);
196                case HttpField.UPGRADE:
197                        return HttpStringListField.fromString(fieldName, fieldValue);
198                case HttpField.VIA:
199                        return HttpStringListField.fromString(fieldName, fieldValue);
200                default:
201                        return HttpStringField.fromString(fieldName, fieldValue);
202                }
203        }
204        
205        /**
206         * Returns the header field name.
207         * 
208         * @return the name
209         */
210        public String getName() {
211                return name;
212        }
213
214        /**
215         * Returns the header field's parsed value.
216         * 
217         * @return the field's value
218         */
219        public T getValue() {
220                return value;
221        }
222        
223        /**
224         * Returns the cconverter used by this field.
225         * 
226         * @return the converter
227         */
228        public Converter<T> getConverter() {
229                return converter;
230        }
231
232        /**
233         * Sets the header field's value.
234         * 
235         * @param value the new value
236         * @return the field for easy chaining
237         */
238        public HttpField<T> setValue(T value) {
239                this.value = value;
240                return this;
241        }
242        
243        /**
244         * Returns the string representation of this field's value.
245         * 
246         * @return the field value as string
247         */
248        public String asFieldValue() {
249                return converter.asFieldValue(value);
250        }
251        
252        /**
253         * Returns the string representation of this header field as it appears in
254         * an HTTP message. Note that the returned string may span several
255         * lines (may contain CR/LF), but is has no trailing CR/LF.
256         * 
257         * @return the field as it occurs in a header
258         */
259        public String asHeaderField() {
260                return getName() + ": " + asFieldValue();
261        }
262        
263        /* (non-Javadoc)
264         * @see java.lang.Object#toString()
265         */
266        @Override
267        public String toString() {
268                StringBuilder result = new StringBuilder();
269                result.append(getClass().getSimpleName());
270                result.append(" [");
271                result.append(asHeaderField().replace("\r\n", " CRLF "));
272                result.append("]");
273                return result.toString();
274        }
275}
