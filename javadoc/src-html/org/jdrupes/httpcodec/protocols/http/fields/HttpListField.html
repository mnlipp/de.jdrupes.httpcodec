019package org.jdrupes.httpcodec.protocols.http.fields;
020
021import java.util.ArrayList;
022import java.util.Collection;
023import java.util.Iterator;
024import java.util.List;
025import java.util.ListIterator;
026
027import org.jdrupes.httpcodec.types.ListConverter;
028
029/**
030 * An HTTP field value that consists of a list of values separated by 
031 * a delimiter. The class provides a "list of field values" view 
032 * of the values.
033 * 
034 * @param <I> the type of the items in the list
035 */
036public abstract class HttpListField<I> extends HttpField<List<I>>
037        implements List<I> {
038
039        /**
040         * Creates a new object with the given field name and no elements. Note 
041         * that in this
042         * initial state, the field is invalid and no string representation
043         * can be generated. This constructor must be followed by method invocations
044         * that add values.
045         * 
046         * @param name the field name
047         * @param converter the converter for the items
048         */
049        protected HttpListField(String name, ListConverter<I> converter) {
050                super(name, new ArrayList<>(), converter);
051        }
052
053        /**
054         * Creates a new header field object with the given field name and values.
055         * 
056         * @param name
057         *            the field name
058         * @param items
059         *                        the items
060         * @param converter 
061         *                        the converter for the items
062         */
063        protected HttpListField(String name, List<I> items, 
064                        ListConverter<I> converter) {
065                super(name, items, converter);
066        }
067
068        /**
069         * Returns the cconverter used by this field.
070         * 
071         * @return the converter
072         */
073        public ListConverter<I> getConverter() {
074                return (ListConverter<I>)super.getConverter();
075        }
076
077        /**
078         * Appends the value to the list of values.
079         * 
080         * @param value the value
081         * @return the field
082         */
083        public HttpListField<I> append(I value) {
084                getValue().add(value);
085                return this;
086        }
087        
088        /**
089         * Appends the value to the list of values if it is not already in the list.
090         * 
091         * @param value the value
092         * @return the field
093         */
094        public HttpListField<I> appendIfNotContained(I value) {
095                if (!getValue().contains(value)) {
096                        getValue().add(value);
097                }
098                return this;
099        }
100        
101        /**
102         * Combine this list with another list of the same type.
103         * 
104         * @param other the other list
105         */
106        @SuppressWarnings("unchecked")
107        public void combine(@SuppressWarnings("rawtypes") HttpListField other) {
108                if (!(getClass().equals(other.getClass()))
109                                || !getName().equals(other.getName())) {
110                        throw new IllegalArgumentException("Types and name must be equal.");
111                }
112                addAll(other);
113        }
114
115        /**
116         * @see java.util.List#add(int, java.lang.Object)
117         */
118        public void add(int index, I element) {
119                getValue().add(index, element);
120        }
121
122        /**
123         * @see java.util.List#add(java.lang.Object)
124         */
125        public boolean add(I element) {
126                return getValue().add(element);
127        }
128
129        /**
130         * @see java.util.List#addAll(java.util.Collection)
131         */
132        public boolean addAll(Collection<? extends I> collection) {
133                return getValue().addAll(collection);
134        }
135
136        /**
137         * @see java.util.List#addAll(int, java.util.Collection)
138         */
139        public boolean addAll(int index, Collection<? extends I> collection) {
140                return getValue().addAll(index, collection);
141        }
142
143        /**
144         * @see java.util.List#clear()
145         */
146        public void clear() {
147                getValue().clear();
148        }
149
150        /**
151         * @see java.util.List#contains(java.lang.Object)
152         */
153        public boolean contains(Object object) {
154                return getValue().contains(object);
155        }
156
157        /**
158         * @see java.util.List#containsAll(java.util.Collection)
159         */
160        public boolean containsAll(Collection<?> collection) {
161                return getValue().containsAll(collection);
162        }
163
164        /**
165         * @see java.util.List#get(int)
166         */
167        public I get(int index) {
168                return getValue().get(index);
169        }
170
171        /**
172         * @see java.util.List#indexOf(java.lang.Object)
173         */
174        public int indexOf(Object object) {
175                return getValue().indexOf(object);
176        }
177
178        /**
179         * @see java.util.List#isEmpty()
180         */
181        public boolean isEmpty() {
182                return getValue().isEmpty();
183        }
184
185        /**
186         * @see java.util.List#iterator()
187         */
188        public Iterator<I> iterator() {
189                return getValue().iterator();
190        }
191
192        /**
193         * @see java.util.List#lastIndexOf(java.lang.Object)
194         */
195        public int lastIndexOf(Object object) {
196                return getValue().lastIndexOf(object);
197        }
198
199        /**
200         * @see java.util.List#listIterator()
201         */
202        public ListIterator<I> listIterator() {
203                return getValue().listIterator();
204        }
205
206        /**
207         * @see java.util.List#listIterator(int)
208         */
209        public ListIterator<I> listIterator(int index) {
210                return getValue().listIterator(index);
211        }
212
213        /**
214         * @see java.util.List#remove(int)
215         */
216        public I remove(int index) {
217                return getValue().remove(index);
218        }
219
220        /**
221         * @see java.util.List#remove(java.lang.Object)
222         */
223        public boolean remove(Object object) {
224                return getValue().remove(object);
225        }
226
227        /**
228         * @see java.util.List#removeAll(java.util.Collection)
229         */
230        public boolean removeAll(Collection<?> collection) {
231                return getValue().removeAll(collection);
232        }
233
234        /**
235         * @see java.util.List#retainAll(java.util.Collection)
236         */
237        public boolean retainAll(Collection<?> collection) {
238                return getValue().retainAll(collection);
239        }
240
241        /**
242         * @see java.util.List#set(int, java.lang.Object)
243         */
244        public I set(int index, I element) {
245                return getValue().set(index, element);
246        }
247
248        /**
249         * @see java.util.List#size()
250         */
251        public int size() {
252                return getValue().size();
253        }
254
255        /**
256         * @see java.util.List#subList(int, int)
257         */
258        public List<I> subList(int fromIndex, int toIndex) {
259                return getValue().subList(fromIndex, toIndex);
260        }
261
262        /**
263         * @see java.util.List#toArray()
264         */
265        public Object[] toArray() {
266                return getValue().toArray();
267        }
268
269        /**
270         * @see java.util.List#toArray(java.lang.Object[])
271         */
272        public <U> U[] toArray(U[] array) {
273                return getValue().toArray(array);
274        }
275}
