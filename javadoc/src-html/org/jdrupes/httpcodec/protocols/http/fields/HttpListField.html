018package org.jdrupes.httpcodec.protocols.http.fields;

020import java.text.ParseException;

027/**
028 * An HTTP field value that consists of a list of separated by a delimiter
029 * strings. The class provides a "list of field values" view of the values.
030 * 
031 * @author Michael N. Lipp
032 */
033public abstract class HttpListField<T> extends HttpField<List<T>>
034        implements List<T>, Cloneable {
035
036        private String unparsedValue;
037        private int position;
038        private List<T> elements = new ArrayList<>();
039        
040        /**
041         * Creates a new object with the given field name and no elements. Note 
042         * that in this
043         * initial state, the field is invalid and no string representation
044         * can be generated. This constructor must be followed by method invocations
045         * that add values.
046         * 
047         * @param name the field name
048         */
049        protected HttpListField(String name) {
050                super(name);
051                reset();
052        }
053
054        /**
055         * Creates a new header field object with the given field name and unparsed
056         * value.
057         * 
058         * @param name
059         *            the field name
060         * @param unparsedValue
061         *                        the field value as it appears in the HTTP header
062         */
063        protected HttpListField(String name, String unparsedValue) {
064                this(name);
065                this.unparsedValue = unparsedValue;
066        }
067
068        /* (non-Javadoc)
069         * @see java.lang.Object#clone()
070         */
071        @Override
072        public HttpListField<T> clone() {
073                HttpListField<T> result = (HttpListField<T>)super.clone();
074                result.elements = new ArrayList<>(elements);
075                return result;
076        }
077
078        /**
079         * Returns the char that separates the items in the list.
080         * 
081         * @return the default implementation returns a comma
082         */
083        protected char getDelimiter() {
084                // Used by default in RFC 7230, see section 7.
085                return ',';
086        }
087        
088        /**
089         * Reset the parsing state.
090         */
091        protected void reset() {
092                position = 0;
093        }
094        
095        /**
096         * Returns the next element from the unparsed value.
097         * 
098         * @return the next element or {@code null} if no elements remain
099         * @throws ParseException if the input violates the field format
100         */
101        protected String nextElement() throws ParseException {
102                // RFC 7230 3.2.6
103                boolean inDquote = false;
104                int startPosition = position;
105                char separator = getDelimiter();
106                try {
107                        while (true) {
108                                if (inDquote) {
109                                        char ch = unparsedValue.charAt(position);
110                                        switch (ch) {
111                                         case '\\':
112                                                 position += 2;
113                                                 continue;
114                                         case '\"':
115                                                 inDquote = false;
116                                         default:
117                                                 position += 1;
118                                                 continue;
119                                        }
120                                }
121                                if (position == unparsedValue.length()) {
122                                        if (position == startPosition) {
123                                                return null;
124                                        }
125                                        return unparsedValue.substring(startPosition, position);
126                                }
127                                char ch = unparsedValue.charAt(position);
128                                if (ch == separator) {
129                                        String result = unparsedValue
130                                                .substring(startPosition, position);
131                                        position += 1; // Skip comma
132                                        while (true) { // Skip optional white space
133                                                ch = unparsedValue.charAt(position);
134                                                if (ch != ' ' && ch != '\t') {
135                                                        break;
136                                                }
137                                                position += 1;
138                                        }
139                                        return result;
140                                }
141                                switch (ch) {
142                                case '\"':
143                                        inDquote = true;
144                                default:
145                                        position += 1;
146                                        continue;
147                                }
148                        }
149                } catch (IndexOutOfBoundsException e) {
150                        throw new ParseException(unparsedValue, position);
151                }
152        }
153        
154        /* (non-Javadoc)
155         * @see org.jdrupes.httpcodec.fields.HttpField#getValue()
156         */
157        @Override
158        public List<T> getValue() {
159                return this;
160        }
161
162        protected abstract String elementToString(T element);
163
164        /* (non-Javadoc)
165         * @see org.jdrupes.httpcodec.util.HttpFieldValue#asString()
166         */
167        @Override
168        public String asFieldValue() {
169                if (size() == 0) {
170                        throw new IllegalStateException(
171                                "Field with list value may not be empty.");
172                }
173                char separator = getDelimiter();
174                boolean first = true;
175                StringBuilder result = new StringBuilder();
176                for (T e: this) {
177                        if (first) {
178                                first = false;
179                        } else {
180                                result.append(separator);
181                                result.append(" ");
182                        }
183                        result.append(elementToString(e));
184                }
185                return result.toString();
186        }
187
188        /**
189         * Appends the value to the list of values.
190         * 
191         * @param value the value
192         * @return the field
193         */
194        public HttpListField<T> append(T value) {
195                elements.add(value);
196                return this;
197        }
198        
199        /**
200         * Appends the value to the list of values if it is not already in the list.
201         * 
202         * @param value the value
203         * @return the field
204         */
205        public HttpListField<T> appendIfNotContained(T value) {
206                if (!elements.contains(value)) {
207                        elements.add(value);
208                }
209                return this;
210        }
211        
212        /**
213         * Combine this list with another list of the same type.
214         * 
215         * @param other the other list
216         */
217        @SuppressWarnings("unchecked")
218        public void combine(@SuppressWarnings("rawtypes") HttpListField other) {
219                if (!(getClass().equals(other.getClass()))
220                                || !getName().equals(other.getName())) {
221                        throw new IllegalArgumentException("Types and name must be equal.");
222                }
223                addAll(other);
224        }
