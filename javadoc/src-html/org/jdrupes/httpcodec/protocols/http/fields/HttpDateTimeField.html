019package org.jdrupes.httpcodec.protocols.http.fields;
020
021import java.lang.reflect.InvocationTargetException;
022import java.text.ParseException;
023import java.time.Instant;
024import java.time.Year;
025import java.time.ZoneId;
026import java.time.format.DateTimeFormatter;
027import java.time.format.DateTimeFormatterBuilder;
028import java.time.format.DateTimeParseException;
029import java.time.temporal.ChronoField;
030import java.util.Locale;
031
032/**
033 * An HTTP Date/Time field as specified by 
034 * [RFC 7231, Section 7.1.1.1](https://tools.ietf.org/html/rfc7231#section-7.1.1.1)
035 */
036public class HttpDateTimeField extends HttpField<Instant> {
037
038        // "Sunday, 06-Nov-94 08:49:37 GMT"
039        private static final DateTimeFormatter RFC_850_DATE_TIME
040                = new DateTimeFormatterBuilder().appendPattern("EEEE, dd-MMM-")
041                .appendValueReduced(ChronoField.YEAR, 2, 2, Year.now().getValue() - 50)
042                .appendPattern(" HH:mm:ss zz")
043                .toFormatter().withZone(ZoneId.of("GMT"))
044                .withLocale(Locale.US);
045        // "Sun Nov  6 08:49:37 1994"
046        private static final DateTimeFormatter ANSI_DATE_TIME
047                = DateTimeFormatter.ofPattern("EEE MMM ppd HH:mm:ss yyyy", Locale.US)
048                .withZone(ZoneId.of("GMT"));
049        
050        private Instant value;
051
052        /**
053         * Creates a header field object with the given value.
054         * 
055         * @param name the field name
056         * @param value
057         *            the field value
058         */
059        public HttpDateTimeField(String name, Instant value) {
060                super(name);
061                this.value = value;
062        }
063
064        protected static Instant instantFromString(String text) 
065                        throws ParseException {
066                try {
067                        return Instant.from(
068                                        DateTimeFormatter.RFC_1123_DATE_TIME.parse(text));
069                } catch (DateTimeParseException e) {
070                        try {
071                                return Instant.from(RFC_850_DATE_TIME.parse(text));
072                        } catch (DateTimeParseException e1) {
073                                try {
074                                        return Instant.from(ANSI_DATE_TIME.parse(text));
075                                } catch (DateTimeParseException e2) {
076                                        throw new ParseException(text, 0);
077                                }
078                        }
079                }
080        }
081        
082        protected static <T extends HttpDateTimeField> T fromString(
083                        Class<T> type, String name, String text) throws ParseException {
084                try {
085                        Instant value = instantFromString(text);
086                        T result = type.getConstructor(String.class, Instant.class)
087                                .newInstance(name, value);
088                        return result;
089                } catch (InstantiationException | IllegalAccessException
090                        | IllegalArgumentException | InvocationTargetException
091                        | NoSuchMethodException | SecurityException e) {
092                        throw new IllegalArgumentException();
093                }
094        }
095
096        /**
097         * Creates a new object with a value obtained by parsing the given String.
098         * 
099         * @param name
100         *            the field name
101         * @param text
102         *            the string to parse
103         * @return the result
104         * @throws ParseException if the input violates the field format
105         */
106        public static HttpDateTimeField fromString(String name, String text)
107                throws ParseException {
108                return fromString(HttpDateField.class, name, text);
109        }
110
111        /**
112         * Returns the value.
113         * 
114         * @return the value
115         */
116        @Override
117        public Instant getValue() {
118                return value;
119        }
120
121        /*
122         * (non-Javadoc)
123         * 
124         * @see org.jdrupes.httpcodec.util.HttpFieldValue#asString()
125         */
126        @Override
127        public String asFieldValue() {
128                return DateTimeFormatter.RFC_1123_DATE_TIME.format(
129                                value.atZone(ZoneId.of("GMT")));        
130        }
131
132}
