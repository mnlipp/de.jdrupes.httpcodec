019package org.jdrupes.httpcodec.protocols.http;
020
021import java.lang.reflect.InvocationTargetException;
022import java.lang.reflect.Method;
023import java.lang.reflect.Modifier;
024import java.util.Collections;
025import java.util.HashMap;
026import java.util.Map;
027import java.util.Optional;
028import java.util.TreeMap;
029import java.util.function.Function;
030
031import org.jdrupes.httpcodec.MessageHeader;
032
033import static org.jdrupes.httpcodec.protocols.http.HttpConstants.*;
034
035import org.jdrupes.httpcodec.protocols.http.fields.HttpField;
036import org.jdrupes.httpcodec.protocols.http.fields.HttpStringField;
037import org.jdrupes.httpcodec.protocols.http.fields.HttpStringListField;
038
039/**
040 * Represents an HTTP message header (either request or response).
041 */
042public abstract class HttpMessageHeader implements MessageHeader {
043
044        private HttpProtocol httpProtocol;
045        private Map<String,HttpField<?>> headers 
046                = new TreeMap<>(String.CASE_INSENSITIVE_ORDER);
047        private boolean messageHasBody;
048        private Map<Class<?>, Method> fromStringMethods = new HashMap<>();
049
050        /**
051         * Creates a new message header.
052         * 
053         * @param httpProtocol the HTTP protocol
054         * @param messageHasBody indicates that a body is expected after the header
055         */
056        public HttpMessageHeader(HttpProtocol httpProtocol, boolean messageHasBody) {
057                this.httpProtocol = httpProtocol;
058                this.messageHasBody = messageHasBody;
059        }
060
061        /**
062         * Return the protocol.
063         * 
064         * @return the HTTP protocol
065         */
066        public HttpProtocol getProtocol() {
067                return httpProtocol;
068        }
069
070        /**
071         * Returns all header fields as unmodifiable map.
072         * 
073         * @return the headers
074         */
075        public Map<String, HttpField<?>> fields() {
076                return Collections.unmodifiableMap(headers);
077        }
078        
079        /**
080         * Set a header field for the message.
081         * 
082         * @param value the header field's value
083         * @return the message header for easy chaining
084         */
085        public HttpMessageHeader setField(HttpField<?> value) {
086                headers.put(value.getName(), value);
087                // Check some consistency rules
088                if (value.getName().equalsIgnoreCase(HttpField.UPGRADE)) {
089                        computeIfAbsent(HttpStringListField.class, HttpField.CONNECTION,
090                                        n -> new HttpStringListField(n))
091                                .appendIfNotContained(HttpField.UPGRADE);
092                }
093                return this;
094        }
095
096        /**
097         * Clear all headers.
098         * 
099         * @return the message header for easy chaining
100         */
101        public HttpMessageHeader clearHeaders() {
102                headers.clear();
103                return this;
104        }
105        
106        /**
107         * Removes a header field from the message.
108         * 
109         * @param name the header field's name
110         */
111        public void removeField(String name) {
112                headers.remove(name);
113        }
114
115        /**
116         * Returns the header field with the given type if it exists.
117         * 
118         * The well known header fields are always parsed with their
119         * proper type. Unknown header fields are provisionally 
120         * parsed as {@link HttpStringField}s. When an attempt is made to
121         * retrieve such a provisional string field with its real type,
122         * it is automatically converted to the real type.
123         * 
124         * In order for the automatic conversion to take place,
125         * the requested type must declare a static method `fromString`
126         * with one or two parameters of type `String`. A method with
127         * one parameter is assumed to return a concrete field type,
128         * i.e. a type that has a predefined field name. It is invoked
129         * with the text to be parsed. A `fromString` method with two 
130         * parameters is invoked with the requested field name as first,
131         * and the text to be parsed as second parameter.
132         * 
133         * If the conversion fails, the field is considered ill-formatted 
134         * and handled as if it didn't exist.
135         * 
136         * @param <T> the header field class
137         * @param type the header field type
138         * @param name the field name
139         * @return the header field if it exists
140         */
141        public <T extends HttpField<?>> Optional<T> 
142                getField(Class<T> type, String name) {
143                HttpField<?> field = headers.get(name);
144                // Not found or type matches
145                if (field == null || type.isAssignableFrom(field.getClass()) ) {
146                        return Optional.ofNullable(type.cast(field));
147                }
148                // Not a string field
149                if (!(field instanceof HttpStringField)) {
150                        return Optional.empty();
151                }
152                // Try conversion...
153                Method fromString = fromStringMethods.get(type);
154                if (fromString == null) {
155                        for (Method m: type.getDeclaredMethods()) {
156                                if ("fromString".equals(m.getName())) {
157                                        if (!Modifier.isStatic(m.getModifiers())) {
158                                                continue;
159                                        }
160                                        Class<?>[] paramTypes = m.getParameterTypes();
161                                        if (paramTypes.length == 1 
162                                                        && paramTypes[0].isAssignableFrom(String.class)) {
163                                                fromString = m;
164                                                break;
165                                        }
166                                        if (paramTypes.length == 2
167                                                        && paramTypes[0].isAssignableFrom(String.class)
168                                                        && paramTypes[1].isAssignableFrom(String.class)) {
169                                                fromString = m;
170                                        }
171                                }
172                        }
173                        fromStringMethods.put(type, fromString);
174                }
175                if (fromString == null) {
176                        return Optional.empty();
177                }
178                try {
179                        T newField = null;
180                        if (fromString.getParameterTypes().length == 1) {
181                                @SuppressWarnings("unchecked")
182                                T result = (T)fromString.invoke(
183                                                null, ((HttpStringField)field).getValue());
184                                newField = result;
185                        } else {
186                                @SuppressWarnings("unchecked")
187                                T result = (T)fromString.invoke(
188                                                null, name, ((HttpStringField)field).getValue());
189                                newField = result;
190                        }
191                        removeField(name);
192                        setField(newField);
193                        return Optional.of(newField);
194                } catch (IllegalAccessException | IllegalArgumentException
195                        | InvocationTargetException e) {
196                        return Optional.empty();
197                }
198        }
199
200        /**
201         * Convenience method for getting a field with a string value.
202         * 
203         * @param name the field name
204         * @return the header field if it exists
205         * @see #getField(Class, String)
206         */
207        public Optional<HttpStringField> getStringField(String name) {
208                return getField(HttpStringField.class, name);
209        }
210        
211        /**
212         * Returns the header field with the given type, computing 
213         * and adding it if it doesn't exist.
214         * 
215         * @param <T> the header field class
216         * @param type the header field type
217         * @param name the field name
218         * @param computeFunction the function that computes the filed if
219         * it doesn't exist
220         * @return the header field if it exists
221         */
222        public <T extends HttpField<?>> T computeIfAbsent(
223                        Class<T> type, String name, Function<String, T> computeFunction) {
224                Optional<T> result = getField(type, name);
225                if (result.isPresent()) {
226                        return result.get();
227                }
228                T value = computeFunction.apply(name);
229                setField(value);
230                return value;
231        }
232        
233        /**
234         * Set the flag that indicates whether this header is followed by a body.
235         * 
236         * @param messageHasBody new value
237         * @return the message for easy chaining
238         */
239        public MessageHeader setMessageHasBody(boolean messageHasBody) {
240                this.messageHasBody = messageHasBody;
241                return this;
242        }
243        
244        /**
245         * Returns {@code true} if the header is followed by a body.
246         * 
247         * @return {@code true} if body data follows
248         */
249        public boolean messageHasBody() {
250                return messageHasBody;
251        }
252
253        /**
254         * Returns true if this is a final message. A message is
255         * final if the value of the `Connection` header field
256         * includes the value "`close`".
257         * 
258         * @return the result
259         */
260        public boolean isFinal() {
261                return getField(HttpStringListField.class, HttpField.CONNECTION)
262                                .map(f -> f.contains("close")).orElse(false);
263        }
264}
