019package org.jdrupes.httpcodec.protocols.http;
020
021import java.text.ParseException;
022import java.util.Collections;
023import java.util.Map;
024import java.util.Optional;
025import java.util.TreeMap;
026import java.util.function.Supplier;
027
028import org.jdrupes.httpcodec.MessageHeader;
029
030import static org.jdrupes.httpcodec.protocols.http.HttpConstants.*;
031
032import org.jdrupes.httpcodec.types.Converter;
033import org.jdrupes.httpcodec.types.Converters;
034import org.jdrupes.httpcodec.types.StringList;
035
036/**
037 * Represents an HTTP message header (either request or response).
038 */
039public abstract class HttpMessageHeader implements MessageHeader {
040
041        private HttpProtocol httpProtocol;
042        private Map<String,HttpField<?>> headers 
043                = new TreeMap<>(String.CASE_INSENSITIVE_ORDER);
044        private boolean messageHasBody;
045
046        /**
047         * Creates a new message header.
048         * 
049         * @param httpProtocol the HTTP protocol
050         * @param messageHasBody indicates that a body is expected after the header
051         */
052        public HttpMessageHeader(HttpProtocol httpProtocol, boolean messageHasBody) {
053                this.httpProtocol = httpProtocol;
054                this.messageHasBody = messageHasBody;
055        }
056
057        /**
058         * Return the protocol.
059         * 
060         * @return the HTTP protocol
061         */
062        public HttpProtocol getProtocol() {
063                return httpProtocol;
064        }
065
066        /**
067         * Returns all header fields as unmodifiable map.
068         * 
069         * @return the headers
070         */
071        public Map<String, HttpField<?>> fields() {
072                return Collections.unmodifiableMap(headers);
073        }
074        
075        /**
076         * Sets a header field for the message.
077         * 
078         * @param value the header field's value
079         * @return the message header for easy chaining
080         */
081        public HttpMessageHeader setField(HttpField<?> value) {
082                headers.put(value.name(), value);
083                // Check some consistency rules
084                if (value.name().equalsIgnoreCase(HttpField.UPGRADE)) {
085                        computeIfAbsent(HttpField.CONNECTION,
086                                        Converters.STRING_LIST, StringList::new)
087                        .value().appendIfNotContained(HttpField.UPGRADE);
088                }
089                return this;
090        }
091
092        /**
093         * Sets a header field for the message. The converter for the
094         * field is lookup using {@link HttpField#lookupConverter(String)}.
095         * 
096         * @param <T> the type of the value
097         * @param name the field name
098         * @param value the header field's value
099         * @return the message header for easy chaining
100         */
101        @SuppressWarnings("unchecked")
102        public <T> HttpMessageHeader setField(String name, T value) {
103                setField(new HttpField<T>(
104                                name, value, (Converter<T>)HttpField.lookupConverter(name)));
105                return this;
106        }
107        
108        /**
109         * Clear all headers.
110         * 
111         * @return the message header for easy chaining
112         */
113        public HttpMessageHeader clearHeaders() {
114                headers.clear();
115                return this;
116        }
117        
118        /**
119         * Removes a header field from the message.
120         * 
121         * @param name the header field's name
122         */
123        public void removeField(String name) {
124                headers.remove(name);
125        }
126
127        /**
128         * Returns the header field with the given type if it exists.
129         * 
130         * Header fields are provisionally parsed as 
131         * {@link HttpField}s with value type `String`. When an attempt is 
132         * made to retrieve such a string field with its real type,
133         * it is automatically converted to the real type.
134         * 
135         * In order for the automatic conversion to take place,
136         * the requested type must declare a static method `fromString`
137         * with one or two parameters of type `String`. A method with
138         * one parameter is assumed to return a concrete field type,
139         * i.e. a type that has a predefined field name. It is invoked
140         * with the text to be parsed. A `fromString` method with two 
141         * parameters is invoked with the requested field name as first,
142         * and the text to be parsed as second parameter.
143         * 
144         * If the conversion fails, the field is considered ill-formatted 
145         * and handled as if it didn't exist.
146         * 
147         * Note that field type conversion may already occur while doing internal
148         * checks. This implies that not all fields can initially be
149         * accessed as {@link HttpField}s with a `String` value.   
150         * 
151         * @param <T> the type of the value in the header field
152         * @param name the field name
153         * @param converter the converter for the value type
154         * @return the header field if it exists
155         */
156        public <T> Optional<HttpField<T>> 
157                getField(String name, Converter<T> converter) {
158                HttpField<?> field = headers.get(name);
159                // Not found
160                if (field == null) {
161                        return Optional.ofNullable(null);
162                }
163                Object value = field.value();
164                Class<?> valueType = value.getClass();
165                Class<?> expectedType = null;
166                try {
167                        expectedType = converter.getClass().getMethod(
168                                "fromFieldValue", String.class).getReturnType();
169                } catch (NoSuchMethodException e) {
170                        // Known to exists
171                }
172                // Match already?
173                if (expectedType.isAssignableFrom(valueType)) {
174                        @SuppressWarnings("unchecked")
175                        HttpField<T> result = (HttpField<T>)field;
176                        return Optional.of(result);
177                }
178                // String field?
179                if (!(value instanceof String)) {
180                        return Optional.empty();
181                }
182                // Try conversion...
183                try {
184                        return Optional.ofNullable(new HttpField<>(
185                                        name, converter.fromFieldValue((String)value), converter));
186                } catch (ParseException e) {
187                        return Optional.empty();
188                }
189        }
190
191        /**
192         * Convenience method for getting the value of a header field.
193         * 
194         * @param <T> the type of the value in the header field
195         * @param name the field name
196         * @param converter the converter for the value type
197         * @return the value if the header field exists
198         */
199        public <T> Optional<T> getValue(String name, Converter<T> converter) {
200                return getField(name, converter).map(HttpField<T>::value);
201        }
202        
203        /**
204         * Convenience method for getting the value of a string field.
205         * 
206         * @param name the field name
207         * @return the value if the header field exists
208         * @see #getField(String, Converter)
209         */
210        public Optional<String> getStringValue(String name) {
211                return getValue(name, Converters.STRING);
212        }
213        
214        /**
215         * Returns the header field with the given name, computing 
216         * and adding it if it doesn't exist.
217         * 
218         * @param <T> the type of the header field's value
219         * @param name the field name
220         * @param converter the converter for the value type
221         * @param supplier the function that computes a value for
222         * a new field.
223         * @return the header field
224         */
225        public <T> HttpField<T> computeIfAbsent(String name, 
226                        Converter<T> converter, Supplier<T> supplier) {
227                Optional<HttpField<T>> result = getField(name, converter);
228                if (result.isPresent()) {
229                        return result.get();
230                }
231                HttpField<T> value = new HttpField<>(name, supplier.get(), converter);
232                
233                setField(value);
234                return value;
235        }
236        
237        /**
238         * Set the flag that indicates whether this header is followed by a body.
239         * 
240         * @param messageHasBody new value
241         * @return the message for easy chaining
242         */
243        public MessageHeader setMessageHasBody(boolean messageHasBody) {
244                this.messageHasBody = messageHasBody;
245                return this;
246        }
247        
248        /**
249         * Returns {@code true} if the header is followed by a body.
250         * 
251         * @return {@code true} if body data follows
252         */
253        public boolean messageHasBody() {
254                return messageHasBody;
255        }
256
257        /**
258         * Returns true if this is a final message. A message is
259         * final if the value of the `Connection` header field
260         * includes the value "`close`".
261         * 
262         * @return the result
263         */
264        public boolean isFinal() {
265                return getField(HttpField.CONNECTION, Converters.STRING_LIST)
266                                .map(h -> h.value()).map(f -> f.contains("close")).orElse(false);
267        }
268}
