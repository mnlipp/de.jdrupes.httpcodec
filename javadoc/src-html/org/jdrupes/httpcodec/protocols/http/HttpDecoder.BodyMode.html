046/**
047 * Implements a decoder for HTTP. The class can be used as base class for both
048 * a request and a response decoder.
049 */
050public abstract class   HttpDecoder<T extends HttpMessageHeader,
051        R extends HttpMessageHeader> 
052        extends HttpCodec<T> implements Decoder<T, R> {
053
054        protected static final String TOKEN = "[" + Pattern.quote(TOKEN_CHARS)
055                + "]+";
056        protected static final String SP = "[ \\t]+";
057        protected static final String HTTP_VERSION = "HTTP/\\d+\\.\\d";
058
059        // RFC 7230 3.2, 3.2.4
060        protected static final Pattern headerLinePatter = Pattern
061                .compile("^(" + TOKEN + "):(.*)$");
062
063        private enum State {
064            // Main states
065                AWAIT_MESSAGE_START, HEADER_LINE_RECEIVED, COPY_UNTIL_CLOSED, 
066                LENGTH_RECEIVED, CHUNK_START_RECEIVED, CHUNK_END_RECEIVED, 
067                CHUNK_TRAILER_LINE_RECEIVED, CLOSED,
068                // Sub states
069                RECEIVE_LINE, AWAIT_LINE_END, COPY_SPECIFIED, FINISH_CHARDECODER, 
070                FLUSH_CHARDECODER
071        }
072
073        protected enum BodyMode {
074                NO_BODY, CHUNKED, LENGTH, UNTIL_CLOSE
075        }
076
077        private long maxHeaderLength = 4194304;
078        private Stack<State> states = new Stack<>();
079        private DynamicByteArray lineBuilder = new DynamicByteArray(8192);
080        private String receivedLine;
081        private String headerLine = null;
082        protected HttpProtocol protocolVersion = HttpProtocol.HTTP_1_0;
083        private long headerLength = 0;
084        private T building;
085        private long leftToRead = 0;
086        private OptimizedCharsetDecoder charDecoder = null;

133        /**
134         * Returns {@code true} if the decoder does not accept further input because
135         * the processed data indicated that the connection has been or is to be
136         * closed.
137         * 
138         * @return the result
139         */
140        public boolean isClosed() {
141                return states.peek() == State.CLOSED;

144        /**
145         * Informs the derived class about the start of a new message.
146         * 
147         * @param startLine
148         *            the start line (first line) of the message
149         * @return the new HttpMessage object that is to hold the decoded data
150         * @throws HttpProtocolException if the input violates the HTTP
151         */
152        protected abstract T newMessage(String startLine)
153                throws HttpProtocolException;
154
155        /**
156         * Informs the derived class that the header has been received completely.
157         * 
158         * @param message the message
159         * @return indication how the body will be transferred
160         * @throws HttpProtocolException if the input violates the HTTP
161         */
162        protected abstract BodyMode headerReceived(T message) 
163                        throws HttpProtocolException;
164
165        /**
166         * Decodes the next chunk of data.
167         * 
168         * @param in
169         *            holds the data to be decoded
170         * @param out
171         *            gets the body data (if any) written to it
172         * @param endOfInput
173         *            {@code true} if there is no input left beyond the data
174         *            currently in the {@code in} buffer (indicates end of body or
175         *            no body at all)
176         * @return the result
177         * @throws HttpProtocolException
178         *             if the message violates the HTTP
179         */
180        public Decoder.Result<R> decode( 
181                ByteBuffer in, Buffer out, boolean endOfInput)
182                throws HttpProtocolException {
183                try {
184                        try {
185                                return uncheckedDecode(in, out, endOfInput);
186                        } catch (ParseException | NumberFormatException e) {
187                                throw new HttpProtocolException(protocolVersion,
188                                        HttpStatus.BAD_REQUEST.getStatusCode(), e.getMessage());
189                        }
190                } catch (HttpProtocolException e) {
191                        states.clear();
192                        states.push(State.CLOSED);
193                        throw e;
194                }
195        }
196
197        private Decoder.Result<R> uncheckedDecode(
198                ByteBuffer in, Buffer out, boolean endOfInput)
199                        throws HttpProtocolException, ParseException {
200                while(true) {
201                        switch (states.peek()) {
202                        // Waiting for CR (start of end of line)
203                        case RECEIVE_LINE: {
204                                if (!in.hasRemaining()) {
205                                        return resultFactory().newResult(false, true);
206                                }
207                                byte ch = in.get();
208                                if (ch == '\r') {
209                                        states.pop();
210                                        states.push(State.AWAIT_LINE_END);
211                                        break;
212                                }
213                                lineBuilder.append(ch);
214                                // RFC 7230 3.2.5
215                                if (headerLength + lineBuilder.position() > maxHeaderLength) {
216                                        throw new HttpProtocolException(protocolVersion,
217                                                HttpStatus.BAD_REQUEST.getStatusCode(),
218                                                "Maximum header size exceeded");
219                                }
220                                break;
221                        }
222                        // Waiting for LF (confirmation of end of line)
223                        case AWAIT_LINE_END: {
224                                if (!in.hasRemaining()) {
225                                        return resultFactory().newResult(false, true);
226                                }
227                                char ch = (char) in.get();
228                                if (ch == '\n') {
229                                        try {
230                                                // RFC 7230 3.2.4
231                                                receivedLine = new String(lineBuilder.array(), 0,
232                                                        lineBuilder.position(), "iso-8859-1");
233                                        } catch (UnsupportedEncodingException e) {
234                                                // iso-8859-1 is guaranteed to be supported
235                                        }
236                                        lineBuilder.clear();
237                                        states.pop();
238                                        break;
239                                }
240                                throw new HttpProtocolException(protocolVersion,
241                                        HttpStatus.BAD_REQUEST.getStatusCode(),
242                                        "CR not followed by LF");
243                        }
244                        // Waiting for the initial request line
245                        case AWAIT_MESSAGE_START:
246                                if (receivedLine.isEmpty()) {
247                                        // Ignore as recommended by RFC2616/RFC7230
248                                        states.push(State.RECEIVE_LINE);
249                                        break;
250                                }
251                                building = newMessage(receivedLine);
252                                messageHeader = null;
253                                charDecoder = null;
254                                states.pop();
255                                headerLine = null;
256                                states.push(State.HEADER_LINE_RECEIVED);
257                                states.push(State.RECEIVE_LINE);
258                                break;
259
260                        case HEADER_LINE_RECEIVED:
261                                if (headerLine != null) {
262                                        // RFC 7230 3.2.4
263                                        if (!receivedLine.isEmpty()
264                                                && (receivedLine.charAt(0) == ' '
265                                                        || receivedLine.charAt(0) == '\t')) {
266                                                headerLine += (" " + receivedLine.substring(1));
267                                                states.push(State.RECEIVE_LINE);
268                                                break;
269                                        }
270                                        // Header line complete, evaluate
271                                        newHeaderLine();
272                                }
273                                if (receivedLine.isEmpty()) {
274                                        // Body starts
275                                        BodyMode bm = headerReceived(building);
276                                        adjustToBodyMode(bm);
277                                        messageHeader = building;
278                                        building = null;
279                                        if (bm == BodyMode.NO_BODY) {
280                                                adjustToEndOfMessage();
281                                                return resultFactory().newResult(false, false);
282                                        }
283                                        if (out == null) {
284                                                return resultFactory().newResult(true, false);
285                                        }
286                                        break;
287                                }
288                                headerLine = receivedLine;
289                                states.push(State.RECEIVE_LINE);
290                                break;
291
292                        case LENGTH_RECEIVED:
293                                // We "drop" to this state after COPY_SPECIFIED
294                                // if we had a content length field
295                                states.pop();
296                                adjustToEndOfMessage();
297                                return resultFactory().newResult(false, false);
298
299                        case CHUNK_START_RECEIVED:
300                                // We "drop" to this state when a line has been read
301                                String sizeText = receivedLine.split(";")[0];
302                                long chunkSize = Long.parseLong(sizeText, 16);
303                                if (chunkSize == 0) {
304                                        states.pop();
305                                        states.push(State.CHUNK_TRAILER_LINE_RECEIVED);
306                                        states.push(State.RECEIVE_LINE);
307                                        break;
308                                }
309                                leftToRead = chunkSize;
310                                // We expect the chunk data and the trailing CRLF (empty line)
311                                // (which must be skipped). In reverse order:
312                                states.push(State.CHUNK_END_RECEIVED);
313                                states.push(State.RECEIVE_LINE);
314                                states.push(State.COPY_SPECIFIED);
315                                break;
316
317                        case CHUNK_END_RECEIVED:
318                                // We "drop" to this state when the CR/LF after chunk data
319                                // has been read. There's nothing to do except to wait for
320                                // next chunk
321                                if (receivedLine.length() != 0) {
322                                        throw new HttpProtocolException(protocolVersion,
323                                                HttpStatus.BAD_REQUEST.getStatusCode(),
324                                                "No CRLF after chunk data.");
325                                }
326                                states.pop();
327                                states.push(State.CHUNK_START_RECEIVED);
328                                states.push(State.RECEIVE_LINE);
329                                break;
330
331                        case CHUNK_TRAILER_LINE_RECEIVED:
332                                // We "drop" to this state when a line has been read
333                                if (!receivedLine.isEmpty()) {
334                                        headerLine = receivedLine;
335                                        newTrailerLine();
336                                        states.push(State.RECEIVE_LINE);
337                                        break;
338                                }
339                                // All chunked data received
340                                adjustToEndOfMessage();
341                                return resultFactory().newResult(false, false);
342
343                        case COPY_SPECIFIED:
344                                if (out == null) {
345                                        return resultFactory().newResult(true, false);
346                                }
347                                int initiallyRemaining = in.remaining();
348                                CoderResult decRes;
349                                if (in.remaining() <= leftToRead) {
350                                        decRes = copyBodyData(out, in, in.remaining(), endOfInput);
351                                } else {
352                                        decRes = copyBodyData(
353                                                        out, in, (int) leftToRead, endOfInput);
354                                }
355                                leftToRead -= (initiallyRemaining - in.remaining());
356                                if (leftToRead == 0) {
357                                        // Everything written (except, maybe, final bytes 
358                                        // from decoder)
359                                        states.pop();
360                                        if (out instanceof CharBuffer && charDecoder != null) {
361                                                states.push(State.FINISH_CHARDECODER);
362                                        }
363                                        break;
364                                }
365                                return resultFactory().newResult(
366                                                (!out.hasRemaining() && in.hasRemaining())
367                                                || (decRes != null && decRes.isOverflow()),
368                                        !in.hasRemaining() 
369                                        || (decRes != null && decRes.isUnderflow()));
370
371                        case FINISH_CHARDECODER:
372                                if (charDecoder.decode(EMPTY_IN, (CharBuffer) out, true)
373                                        .isOverflow()) {
374                                        return resultFactory().newResult(true, false);
375                                }
376                                states.pop();
377                                states.push(State.FLUSH_CHARDECODER);
378                                break;
379                                
380                        case FLUSH_CHARDECODER:
381                                if (charDecoder.flush((CharBuffer)out).isOverflow()) {
382                                        return resultFactory().newResult(true, false);
383                                }
384                                states.pop();
385                                break;
386
387                        case COPY_UNTIL_CLOSED:
388                                if (out == null) {
389                                        return resultFactory().newResult(true, false);
390                                }
391                                decRes = copyBodyData(out, in, in.remaining(), endOfInput);
392                                boolean overflow = (!out.hasRemaining() && in.hasRemaining())
393                                                || (decRes != null && decRes.isOverflow());
394                                if (overflow) {
395                                        return resultFactory().newResult(true, false);
396                                }
397                                if (!endOfInput) {
398                                        return resultFactory().newResult(false, true);
399                                }
400                                // Final input successfully processed.
401                                states.pop();
402                                states.push(State.CLOSED);
403                                if (out instanceof CharBuffer && charDecoder != null) {
404                                        // Final flush needed
405                                        states.push(State.FINISH_CHARDECODER);
406                                }
407                                break;
408
409                        case CLOSED:
410                                in.position(in.limit());
411                                return resultFactory().newResult(false, false);
412                        }
413                }
414        }

416        private void newHeaderLine() throws HttpProtocolException, ParseException {
417                headerLength += headerLine.length() + 2;
418                // RFC 7230 3.2
419                Matcher hlp = headerLinePatter.matcher(headerLine);
420                if (!hlp.matches()) {
421                        throw new HttpProtocolException(protocolVersion,
422                                HttpStatus.BAD_REQUEST.getStatusCode(), "Invalid header");
423                }
424                String fieldName = hlp.group(1);
425                // RFC 7230 3.2.4
426                String fieldValue = hlp.group(2).trim();
427                HttpField<?> field = HttpField.fromString(fieldName, fieldValue);
428                switch (field.getName()) {
429                case HttpField.CONTENT_LENGTH:
430                        // RFC 7230 3.3.3 (3.)
431                        if (building.fields()
432                                .containsKey(HttpField.TRANSFER_ENCODING)) {
433                                field = null;
434                                break;
435                        }
436                        // RFC 7230 3.3.3 (4.)
437                        HttpContentLengthField existing = building.getField(
438                                HttpContentLengthField.class, HttpField.CONTENT_LENGTH)
439                                        .orElse(null);
440                        if (existing != null && !existing.getValue()
441                                .equals(((HttpContentLengthField) field).getValue())) {
442                                throw new HttpProtocolException(protocolVersion,
443                                        HttpStatus.BAD_REQUEST);
444                        }
445                        break;
446                case HttpField.TRANSFER_ENCODING:
447                        // RFC 7230 3.3.3 (3.)
448                        building.removeField(HttpField.CONTENT_LENGTH);
449                        break;
450                }
451                if (field == null) {
452                        return;
453                }
454                addHeaderField(building, field);
455        }
456
457        private void newTrailerLine() throws HttpProtocolException, ParseException {
458                headerLength += headerLine.length() + 2;
459                // RFC 7230 3.2
460                Matcher hlp = headerLinePatter.matcher(headerLine);
461                if (!hlp.matches()) {
462                        throw new HttpProtocolException(protocolVersion,
463                                HttpStatus.BAD_REQUEST.getStatusCode(), "Invalid header");
464                }
465                String fieldName = hlp.group(1);
466                // RFC 7230 3.2.4
467                String fieldValue = hlp.group(2).trim();
468                HttpField<?> field = HttpField.fromString(fieldName, fieldValue);
469                // RFC 7230 4.4
470                HttpStringListField trailerField = messageHeader
471                        .computeIfAbsent(HttpStringListField.class, HttpField.TRAILER,
472                                        n -> new HttpStringListField(n));
473                if (!trailerField.containsIgnoreCase(field.getName())) {
474                        trailerField.add(field.getName());
475                }
476                addHeaderField(messageHeader, field);
477        }
478
479        private void addHeaderField(T header, HttpField<?> field)
480                throws HttpProtocolException, ParseException {
481                // RFC 7230 3.2.2
482                HttpField<?> existing = header.fields().get(field.getName());
483                if (existing != null) {
484                        if (!(existing instanceof HttpListField<?>)
485                                || !(field instanceof HttpListField<?>)
486                                || !(existing.getClass().equals(field.getClass()))) {
487                                throw new HttpProtocolException(protocolVersion,
488                                        HttpStatus.BAD_REQUEST.getStatusCode(),
489                                        "Multiple occurences of field " + field.getName());
490                        }
491                        ((HttpListField<?>) existing).combine((HttpListField<?>) field);
492                } else {
493                        header.setField(field);
494                }
495        }

497        private void adjustToBodyMode(BodyMode bm) {
498                building.setMessageHasBody(bm != BodyMode.NO_BODY);
499                states.pop();
500                switch (bm) {
501                case UNTIL_CLOSE:
502                        states.push(State.COPY_UNTIL_CLOSED);
503                        break;
504                case CHUNKED:
505                        states.push(State.CHUNK_START_RECEIVED);
506                        states.push(State.RECEIVE_LINE);
507                        break;
508                case LENGTH:
509                        HttpContentLengthField clf = building.getField(
510                                HttpContentLengthField.class,
511                                HttpField.CONTENT_LENGTH).get();
512                        leftToRead = clf.getValue();
513                        if (leftToRead > 0) {
514                                states.push(State.LENGTH_RECEIVED);
515                                states.push(State.COPY_SPECIFIED);
516                                break;
517                        }
518                        // Length == 0 means no body, fall through
519                case NO_BODY:
520                        break;
521                }
522        }

524        private CoderResult copyBodyData(
525                        Buffer out, ByteBuffer in, int limit, boolean endOfInput) {
526                if (out instanceof ByteBuffer) {
527                        ByteBufferUtils.putAsMuchAsPossible((ByteBuffer) out, in, limit);
528                        return null;
529                } else if (out instanceof CharBuffer) {
530                        if (charDecoder == null) {
531                                charDecoder = new OptimizedCharsetDecoder(
532                                        Charset.forName(bodyCharset()).newDecoder());
533                        }
534                        int oldLimit = in.limit();
535                        try {
536                                if (in.remaining() > limit) {
537                                        in.limit(in.position() + limit);
538                                }
539                                return charDecoder.decode(in, (CharBuffer)out, endOfInput);
540                        } finally {
541                                in.limit(oldLimit);
542                        }
543                } else {
544                        throw new IllegalArgumentException(
545                                "Only Byte- or CharBuffer are allowed.");
546                }
547        }
548
549        private void adjustToEndOfMessage() {
550                // RFC 7230 6.3
551                Optional<HttpStringListField> connection = messageHeader
552                        .getField(HttpStringListField.class, HttpField.CONNECTION);
553                if (connection.isPresent() 
554                                && connection.get().containsIgnoreCase("close")) {
555                        states.push(State.CLOSED);
556                        return;
557                }
558                if (messageHeader.getProtocol().compareTo(HttpProtocol.HTTP_1_1) >= 0) {
559                        states.push(State.AWAIT_MESSAGE_START);
560                        states.push(State.RECEIVE_LINE);
561                        return;
562                }
563                states.push(State.CLOSED);
564        }

566        /**
567         * Results from {@link HttpDecoder} add no additional
568         * information to {@link org.jdrupes.httpcodec.Decoder.Result}. This
569         * class provides only a factory for creating 
570         * the results as required by {@link HttpDecoder}.
571         */
572        public static class Result<R extends MessageHeader>
573                extends Decoder.Result<R> {
574
575                public Result(boolean overflow, boolean underflow,
576                        boolean closeConnection, boolean headerCompleted, R response,
577                        boolean responseOnly) {
578                        super(overflow, underflow, closeConnection, headerCompleted, response,
579                                responseOnly);
580                }
581
582                /**
583                 * A factory for creating new Results.
584                 */
585                protected abstract static class Factory<R extends MessageHeader> 
586                        extends Decoder.Result.Factory<R> {
587                        
588                        /**
589                         * Create a new result. Implementing classes can
590                         * obtain the value for 
591                         * {@link org.jdrupes.httpcodec.Codec.Result#getCloseConnection()}
592                         * from {@link HttpDecoder#isClosed()}.
593                         * 
594                         * @param overflow
595                         *            {@code true} if the data didn't fit in the out buffer
596                         * @param underflow
597                         *            {@code true} if more data is expected
598                         * @return the result
599                         */
600                        protected abstract Result<R> newResult(
601                                boolean overflow, boolean underflow);
602                }               
603        }
