018package org.jdrupes.httpcodec.protocols.http.server;
019
020import java.net.URI;
021import java.net.URISyntaxException;
022import java.nio.Buffer;
023import java.nio.ByteBuffer;
024import java.util.Optional;
025import java.util.regex.Matcher;
026import java.util.regex.Pattern;
027
028import org.jdrupes.httpcodec.Decoder;
029import org.jdrupes.httpcodec.protocols.http.HttpDecoder;
030import org.jdrupes.httpcodec.protocols.http.HttpProtocolException;
031import org.jdrupes.httpcodec.protocols.http.HttpRequest;
032import org.jdrupes.httpcodec.protocols.http.HttpResponse;
033import org.jdrupes.httpcodec.protocols.http.fields.HttpField;
034import org.jdrupes.httpcodec.protocols.http.fields.HttpStringField;
035import org.jdrupes.httpcodec.protocols.http.fields.HttpStringListField;
036
037/**
038 * A decoder for HTTP requests that accepts data from a sequence of
039 * {@link ByteBuffer}s.
040 * 
041 * @author Michael N. Lipp
042 */
043public class HttpRequestDecoder 
044        extends HttpDecoder<HttpRequest, HttpResponse> {
045
046        // RFC 7230 3.1.1
047        private final static Pattern requestLinePatter = Pattern
048                .compile("^(" + TOKEN + ")" + SP + "([^ \\t]+)" + SP + "("
049                        + HTTP_VERSION + ")$");
050
051        /**
052         * Creates a new encoder that belongs to the given HTTP engine.
053         */
054        public HttpRequestDecoder() {
055                super();
056        }
057
058        /* (non-Javadoc)
059         * @see org.jdrupes.httpcodec.HttpDecoder#decode(java.nio.ByteBuffer)
060         */
061        @Override
062        public Result decode (ByteBuffer in, Buffer out, boolean endOfInput) {
063                try {
064                        return (Result)super.decode(in, out, endOfInput);
065                } catch (HttpProtocolException e) {
066                        HttpResponse response = new HttpResponse(e.getHttpVersion(), 
067                                        e.getStatusCode(), e.getReasonPhrase(), false);
068                        response.setField(
069                                new HttpStringListField(HttpField.CONNECTION, "close"));
070                        return newResult(false, false, false, response, true);
071                }
072        }
073
074        /**
075         * Checks whether the first line of a message is a valid request.
076         * If so, create a new request message object with basic information, else
077         * throw an exception.
078         * <P>
079         * Called by the base class when a first line is received.
080         * 
081         * @param startLine the first line
082         * @throws HttpProtocolException if the line is not a correct request line
083         */
084        @Override
085        protected HttpRequest newMessage(String startLine)
086                throws HttpProtocolException {
087                Matcher requestMatcher = requestLinePatter.matcher(startLine);
088                if (!requestMatcher.matches()) {
089                        // RFC 7230 3.1.1
090                        throw new HttpProtocolException(protocolVersion,
091                                HttpStatus.BAD_REQUEST.getStatusCode(),
092                                "Illegal request line");
093                }
094                String method = requestMatcher.group(1);
095                String uriGroup = requestMatcher.group(2);
096                String httpVersion = requestMatcher.group(3);
097                boolean found = false;
098                for (HttpProtocol v : HttpProtocol.values()) {
099                        if (v.toString().equals(httpVersion)) {
100                                protocolVersion = v;
101                                found = true;
102                        }
103                }
104                if (!found) {
105                        throw new HttpProtocolException(HttpProtocol.HTTP_1_1,
106                                HttpStatus.HTTP_VERSION_NOT_SUPPORTED);
107                }
108                URI uri = null;
109                if (uriGroup.equals("*")) {
110                        uri = HttpRequest.ASTERISK_REQUEST;
111                } else {
112                        try {
113                                uri = new URI(uriGroup);
114                        } catch (URISyntaxException e) {
115                                throw new HttpProtocolException(protocolVersion,
116                                        HttpStatus.BAD_REQUEST.getStatusCode(), e.getMessage());
117                        }
118                }
119                HttpRequest request = new HttpRequest
120                                (method, uri, protocolVersion, false);
121                HttpResponse response = (new HttpResponse(protocolVersion,
122                                HttpStatus.NOT_IMPLEMENTED, false)).setRequest(request); 
123                return request.setResponse(response);
124        }
125
126        /* (non-Javadoc)
127         * @see org.jdrupes.httpcodec.HttpDecoder#headerReceived(org.jdrupes.httpcodec.HttpMessage)
128         */
129        @Override
130        protected BodyMode headerReceived(HttpRequest message) 
131                        throws HttpProtocolException {
132                // Handle field of special interest
133                Optional<HttpStringField> host = message.getField
134                                (HttpStringField.class, HttpField.HOST);
135                if (host.isPresent()) {
136                        try {
137                                URI parsed = new URI("http://" + host.get().getValue());
138                                message.setHostAndPort(parsed.getHost(), parsed.getPort());
139                        } catch (URISyntaxException e) {
140                                throw new HttpProtocolException(protocolVersion,
141                                        HttpStatus.BAD_REQUEST.getStatusCode(),
142                                        "Invalid Host port.");
143                        }
144                } else {
145                        // RFC 7230 5.4.
146                        if (message.getProtocol().compareTo(HttpProtocol.HTTP_1_1) >= 0) {
147                                throw new HttpProtocolException(protocolVersion,
148                                        HttpStatus.BAD_REQUEST.getStatusCode(),
149                                        "HTTP 1.1 request must have a Host field.");
150                        }
151                }
152                if (message.getField(HttpStringListField.class, HttpField.CONNECTION)
153                                .map(f -> f.containsIgnoreCase("close")).orElse(false)) {
154                        // RFC 7230 6.6.
155                        message.getResponse().get().setField(new HttpStringListField(
156                                HttpField.CONNECTION, "close"));
157                }
158
159                // Find out about body
160                HttpStringListField transEncs = message.getField(
161                        HttpStringListField.class, HttpField.TRANSFER_ENCODING)
162                                .orElse(null);
163                if (transEncs != null) {
164                        // RFC 7230 3.3.1
165                        HttpStringListField tec = transEncs.clone();
166                        tec.removeIgnoreCase(TransferCoding.CHUNKED.toString());
167                        if (tec.size() > 0) {
168                                throw new HttpProtocolException(protocolVersion,
169                                        HttpStatus.NOT_IMPLEMENTED);
170                        }
171                        // RFC 7230 3.3.3 (3.)
172                        if (transEncs != null) {
173                                if (transEncs.get(transEncs.size() - 1)
174                                        .equalsIgnoreCase(TransferCoding.CHUNKED.toString())) {
175                                        return BodyMode.CHUNKED;
176                                } else {
177                                        throw new HttpProtocolException(protocolVersion,
178                                                HttpStatus.BAD_REQUEST);
179                                }
180                        }
181                }
182                // RFC 7230 3.3.3 (5.)
183                if (message.fields().containsKey(HttpField.CONTENT_LENGTH)) {
184                        return BodyMode.LENGTH;
185                }
186                // RFC 7230 3.3.3 (6.)
187                return BodyMode.NO_BODY;
188        }
189
190        /**
191         * Overrides the base interface's factory method in order to make
192         * it return the extended return type. As the {@link HttpRequestDecoder}
193         * does not know about a response, this implementation always
194         * returns a result without one.
195         * 
196         * @param overflow
197         *            {@code true} if the data didn't fit in the out buffer
198         * @param underflow
199         *            {@code true} if more data is expected
200         * @param headerCompleted
201         *            indicates that the message header has been completed and
202         *            the message (without body) is available
203         */
204        @Override
205        public Result newResult(boolean overflow, boolean underflow,
206                boolean headerCompleted) {
207                return newResult(overflow, underflow, headerCompleted, null, false);
208        }
209
210        /**
211         * Factory method for result.
212         * 
213         * @param overflow
214         *            {@code true} if the data didn't fit in the out buffer
215         * @param underflow
216         *            {@code true} if more data is expected
217         * @param headerCompleted {@code true} if the header has completely
218         * been decoded
219         * @param response a response to send due to an error
220         * @param responseOnly if the result includes a response 
221         * this flag indicates that no further processing besides 
222         * sending the response is required
223         * @return the result
224         */
225        public Result newResult (boolean overflow, boolean underflow, 
226                        boolean headerCompleted, HttpResponse response, 
227                        boolean responseOnly) {
228                return new Result(overflow, underflow, 
229                                headerCompleted, response, responseOnly) {
230                };
231        }
232
233        /**
234         * Short for {@code RequestDecoder.Result<HttpResponse>}, provided
235         * for convenience.
236         * <P>
237         * The class is declared abstract to promote the usage of the factory
238         * method.
239         * 
240         * @author Michael N. Lipp
241         */
242        public static abstract class Result 
243                extends Decoder.Result<HttpResponse> {
244
245                /**
246                 * Creates a new result.
247                 * @param overflow
248                 *            {@code true} if the data didn't fit in the out buffer
249                 * @param underflow
250                 *            {@code true} if more data is expected
251                 * @param headerCompleted
252                 *            {@code true} if the header has completely been decoded
253                 * @param response
254                 *            a response to send due to an error
255                 * @param responseOnly
256                 *            if the result includes a response this flag indicates that
257                 *            no further processing besides sending the response is
258                 *            required
259                 */
260                public Result(boolean overflow, boolean underflow,
261                        boolean headerCompleted, HttpResponse response, 
262                        boolean responseOnly) {
263                        super(overflow, underflow, false, headerCompleted, response,
264                                        responseOnly);
265                }
266        }
