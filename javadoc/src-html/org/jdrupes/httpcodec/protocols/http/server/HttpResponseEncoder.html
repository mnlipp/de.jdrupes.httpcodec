041/**
042 * @author Michael N. Lipp
043 */
044public class HttpResponseEncoder extends HttpEncoder<HttpResponse> {
045
046        private static ServiceLoader<ProtocolProvider> pluginLoader 
047                = ServiceLoader.load(ProtocolProvider.class);
048        private static Result.Factory resultFactory = new Result.Factory() {
049        };
050        
051        private Map<String,ProtocolProvider> plugins = new HashMap<>();
052        private String switchingTo;
053        private ProtocolProvider protocolPlugin;
054
055        /**
056         * Creates a new encoder that belongs to the given HTTP engine.
057         */
058        public HttpResponseEncoder() {
059                super();
060        }
061
062        /* (non-Javadoc)
063         * @see org.jdrupes.httpcodec.protocols.http.HttpEncoder#resultFactory()
064         */
065        @Override
066        protected Result.Factory resultFactory() {
067                return resultFactory;
068        }
069
070        /* (non-Javadoc)
071         * @see org.jdrupes.httpcodec.protocols.http.HttpEncoder#encode(org.jdrupes.httpcodec.protocols.http.HttpMessageHeader)
072         */
073        @Override
074        public void encode(HttpResponse messageHeader) {
075                if (messageHeader.getStatusCode()
076                                        == HttpStatus.SWITCHING_PROTOCOLS.getStatusCode()) {
077                        switchingTo = prepareSwitchProtocol(messageHeader);
078                }
079                super.encode(messageHeader);
080        }
081
082        private String prepareSwitchProtocol(HttpResponse response) {
083                Optional<String> protocol = response
084                                .getField(HttpStringListField.class, HttpField.UPGRADE)
085                                .map(l -> l.get(0));
086                if (!protocol.isPresent()) {
087                        response.setStatus(HttpStatus.BAD_REQUEST)
088                                .setMessageHasBody(false).clearHeaders();
089                        return null;
090                }
091                synchronized (pluginLoader) {
092                        if (plugins.containsKey(protocol.get())) {
093                                protocolPlugin = plugins.get(protocol.get());
094                        } else {
095                                protocolPlugin = StreamSupport
096                                                .stream(pluginLoader.spliterator(), false)
097                                                .filter(p -> p.supportsProtocol(protocol.get()))
098                                                .findFirst().get();
099                                plugins.put(protocol.get(), protocolPlugin);
100                        }
101                }
102                if (protocolPlugin == null) {
103                        response.setStatus(HttpStatus.BAD_REQUEST)
104                                .setMessageHasBody(false).clearHeaders();
105                        return null;
106                }
107                protocolPlugin.augmentInitialResponse(response);
108                if (response.getStatusCode() 
109                                != HttpStatus.SWITCHING_PROTOCOLS.getStatusCode()) {
110                        // Not switching after all
111                        return null;
112                }
113                return protocol.get();
114        }
115
116        /* (non-Javadoc)
117         * @see org.jdrupes.httpcodec.internal.Encoder#encode(java.nio.ByteBuffer, java.nio.ByteBuffer)
118         */
119        @Override
120        public Result encode(Buffer in, ByteBuffer out, boolean endOfInput) {
121                Result result = (Result)super.encode(in, out, endOfInput);
122                if (switchingTo != null && endOfInput 
123                                && !result.isUnderflow() && !result.isOverflow()) {
124                        // Last invocation of encode
125                        return resultFactory().newResult(false, false, 
126                                        result.getCloseConnection(), switchingTo, 
127                                        protocolPlugin.createRequestDecoder(switchingTo), 
128                                        protocolPlugin.createResponseEncoder(switchingTo));
129                }
130                return result;
131        }
132
133        /* (non-Javadoc)
134         * @see org.jdrupes.httpcodec.internal.Encoder#startMessage(java.io.Writer)
135         */
136        @Override
137        protected void startMessage(HttpResponse response, Writer writer)
138                throws IOException {
139                writer.write(response.getProtocol().toString());
140                writer.write(" ");
141                writer.write(Integer.toString(response.getStatusCode()));
142                writer.write(" ");
143                writer.write(response.getReasonPhrase());
144                writer.write("\r\n");
145        }
146
147        /**
148         * The result from encoding a response. In addition to the usual
149         * codec result, a response encoder may signal to the invoker that the
150         * connection to the requester must be closed and that the protocol has
151         * been switched.
152         * 
153         * @author Michael N. Lipp
154         */
155        public abstract static class Result extends HttpEncoder.Result
156                implements Codec.ProtocolSwitchResult {
157
158                private String newProtocol;
159                private Decoder<?, ?> newDecoder;
160                private Encoder<?> newEncoder;
161                
162                /**
163                 * Returns a new result.
164                 * 
165                 * @param overflow
166                 *            {@code true} if the data didn't fit in the out buffer
167                 * @param underflow
168                 *            {@code true} if more data is expected
169                 * @param closeConnection
170                 *            {@code true} if the connection should be closed
171                 * @param newProtocol the name of the new protocol if a switch occurred
172                 * @param newDecoder the new decoder if a switch occurred
173                 * @param newEncoder the new decoder if a switch occurred
174                 */
175                protected Result(boolean overflow, boolean underflow,
176                        boolean closeConnection, String newProtocol,
177                        Decoder<?, ?> newDecoder, Encoder<?> newEncoder) {
178                        super(overflow, underflow, closeConnection);
179                        this.newProtocol = newProtocol;
180                        this.newEncoder = newEncoder;
181                        this.newDecoder = newDecoder;
182                }
183
184                @Override
185                public String newProtocol() {
186                        return newProtocol;
187                }
188                
189                @Override
190                public Encoder<?> newEncoder() {
191                        return newEncoder;
192                }
193                
194                @Override
195                public Decoder<?, ?> newDecoder() {
196                        return newDecoder;
197                }
198
199                /* (non-Javadoc)
200                 * @see java.lang.Object#hashCode()
201                 */
202                @Override
203                public int hashCode() {
204                        final int prime = 31;
205                        int result = super.hashCode();
206                        result = prime * result
207                                + ((newDecoder == null) ? 0 : newDecoder.hashCode());
208                        result = prime * result
209                                + ((newEncoder == null) ? 0 : newEncoder.hashCode());
210                        result = prime * result
211                                + ((newProtocol == null) ? 0 : newProtocol.hashCode());
212                        return result;
213                }
214
215                /* (non-Javadoc)
216                 * @see java.lang.Object#equals(java.lang.Object)
217                 */
218                @Override
219                public boolean equals(Object obj) {
220                        if (this == obj)
221                                return true;
222                        if (!super.equals(obj))
223                                return false;
224                        if (getClass() != obj.getClass())
225                                return false;
226                        Result other = (Result) obj;
227                        if (newDecoder == null) {
228                                if (other.newDecoder != null)
229                                        return false;
230                        } else if (!newDecoder.equals(other.newDecoder))
231                                return false;
232                        if (newEncoder == null) {
233                                if (other.newEncoder != null)
234                                        return false;
235                        } else if (!newEncoder.equals(other.newEncoder))
236                                return false;
237                        if (newProtocol == null) {
238                                if (other.newProtocol != null)
239                                        return false;
240                        } else if (!newProtocol.equals(other.newProtocol))
241                                return false;
242                        return true;
243                }
244
245                /* (non-Javadoc)
246                 * @see java.lang.Object#toString()
247                 */
248                @Override
249                public String toString() {
250                        StringBuilder builder = new StringBuilder();
251                        builder.append("HttpResponseEncoder.Result [overflow=");
252                        builder.append(isOverflow());
253                        builder.append(", underflow=");
254                        builder.append(isUnderflow());
255                        builder.append(", closeConnection=");
256                        builder.append(getCloseConnection());
257                        builder.append(", ");
258                        if (newProtocol != null) {
259                                builder.append("newProtocol=");
260                                builder.append(newProtocol);
261                                builder.append(", ");
262                        }
263                        if (newDecoder != null) {
264                                builder.append("newDecoder=");
265                                builder.append(newDecoder);
266                                builder.append(", ");
267                        }
268                        if (newEncoder != null) {
269                                builder.append("newEncoder=");
270                                builder.append(newEncoder);
271                        }
272                        builder.append("]");
273                        return builder.toString();
274                }
275
276                /**
277                 * A factory for creating new Results.
278                 */
279                protected static class Factory extends HttpEncoder.Result.Factory {
280                        
281                        /**
282                         * Create a new result.
283                         * 
284                         * @param overflow
285                         *            {@code true} if the data didn't fit in the out buffer
286                         * @param underflow
287                         *            {@code true} if more data is expected
288                         * @param closeConnection
289                         *            {@code true} if the connection should be closed
290                         * @param newProtocol the name of the new protocol if a switch occurred
291                         * @param newDecoder the new decoder if a switch occurred
292                         * @param newEncoder the new decoder if a switch occurred
293                         * @return the result
294                         */
295                        public Result newResult (boolean overflow, boolean underflow,
296                                boolean closeConnection, String newProtocol,
297                                Decoder<?, ?> newDecoder, Encoder<?> newEncoder) {
298                                return new Result (overflow, underflow, closeConnection,
299                                                newProtocol, newDecoder, newEncoder) {
300                                };
301                        }
302
303                        /**
304                         * Create a new (preliminary) result. This is invoked by the
305                         * base class. We cannot supply the missing information yet.
306                         * If necessary the result will be modified in 
307                         * {@link HttpResponseEncoder#encode(Buffer, ByteBuffer, boolean)}.
308                         * 
309                         * @param overflow
310                         *            {@code true} if the data didn't fit in the out buffer
311                         * @param underflow
312                         *            {@code true} if more data is expected
313                         * @param closeConnection
314                         *            {@code true} if the connection should be closed
315                         * @return the result
316                         */
317                        @Override
318                        public Result newResult(
319                                boolean overflow, boolean underflow,
320                                boolean closeConnection) {
321                                return newResult(overflow, underflow, closeConnection,
322                                                null, null, null);
323                        }
324                        
325                }               
326
327        }
