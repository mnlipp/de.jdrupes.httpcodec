019package org.jdrupes.httpcodec.protocols.http;
020
021/**
022 * Common HTTP constants.
023 */
024public interface HttpConstants {
025
026    public static enum HttpProtocol { 
027        HTTP_1_0("HTTP/1.0"), HTTP_1_1("HTTP/1.1");
028        
029        private String repr;
030        
031        HttpProtocol(String repr) {
032                this.repr = repr;
033        }
034
035        @Override
036                public String toString() {
037                        return repr;
038                }
039    }
040
041    public static enum HttpStatus {
042        CONTINUE(100, "Continue"),
043        SWITCHING_PROTOCOLS(101, "Switching Protocols"),
044        OK(200, "OK"),
045        CREATED(201, "Created"),
046        ACCEPTED(202, "Accepted"),
047        NON_AUTHORITATIVE_INFORMATION(203, "Non-Authoritative Information"),
048        NO_CONTENT(204, "No Content"),
049        RESET_CONTENT(205, "Reset Content"),
050        PARTIAL_CONTENT(206, "Partial Content"),
051        MULTIPLE_CHOICES(300, "Multiple Choices"),
052        MOVED_PERMANENTLY(301, "Moved Permanently"),
053        FOUND(302, "Found"),
054        SEE_OTHER(303, "See Other"),
055        NOT_MODIFIED(304, "Not Modified"),
056        USE_PROXY(305, "Use Proxy"),
057        TEMPORARY_REDIRECT(307, "Temporary Redirect"),
058        BAD_REQUEST(400, "Bad Request"),
059        UNAUTHORIZED(401, "Unauthorized"),
060        PAYMENT_REQUIRED(402, "Payment Required"),
061        FORBIDDEN(403, "Forbidden"),
062        NOT_FOUND(404, "Not Found"),
063        METHOD_NOT_ALLOWED(405, "Method Not Allowed"),
064        NOT_ACCEPTABLE(406, "Not Acceptable"),
065        PROXY_AUTHENTICATION_REQUIRED(407, "Proxy Authentication Required"),
066        REQUEST_TIME_OUT(408, "Request Time-out"),
067        CONFLICT(409, "Conflict"),
068        GONE(410, "Gone"),
069        LENGTH_REQUIRED(411, "Length Required"),
070        PRECONDITION_FAILED(412, "Precondition Failed"),
071        REQUEST_ENTITY_TOO_LARGE(413, "Request Entity Too Large"),
072        REQUEST_URI_TOO_LARGE(414, "Request-URI Too Large"),
073        UNSUPPORTED_MEDIA_TYPE(415, "Unsupported Media Type"),
074        REQUESTED_RANGE_NOT_SATISFIABLE(416, "Requested range not satisfiable"),
075        EXPECTATION_FAILED(417, "Expectation Failed"),
076        INTERNAL_SERVER_ERROR(500, "Internal Server Error"),
077        NOT_IMPLEMENTED(501, "Not Implemented"),
078        BAD_GATEWAY(502, "Bad Gateway"),
079        SERVICE_UNAVAILABLE(503, "Service Unavailable"),
080        GATEWAY_TIME_OUT(504, "Gateway Time-out"),
081        HTTP_VERSION_NOT_SUPPORTED(505, "HTTP Version not supported");
082        
083        private int statusCode;
084        private String reasonPhrase;
085        
086        HttpStatus(int statusCode, String reasonPhrase) {
087                this.statusCode = statusCode;
088                this.reasonPhrase = reasonPhrase;
089        }
090        
091                /**
092                 * @return the status code
093                 */
094                public int getStatusCode() {
095                        return statusCode;
096                }
097
098                /**
099                 * @return the reason phrase
100                 */
101                public String getReasonPhrase() {
102                        return reasonPhrase;
103                }
104    }
105    
106    public enum TransferCoding { CHUNKED("chunked"), COMPRESS("compress"),
107        DEFLATE("deflate"), GZIP("gzip");
108        
109        private String repr;
110        
111        TransferCoding(String repr) {
112                this.repr = repr;
113        }
114
115        @Override
116                public String toString() {
117                        return repr;
118                }
119    }
120
121        public static final String TOKEN_CHARS 
122                = "!#$%&'*+-.0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZ_"
123                        + "^`abcdefghijklmnopqrstuvwxyz|~";
124
125}
