035/**
036 * A base class for all kinds of header field values.
037 * 
038 * @param <T> the type of the header field's value

040 * @see "[MessageHeaders](https://www.iana.org/assignments/message-headers/message-headers.xhtml)"

143        private final String name;
144        private T value;
145        private Converter<T> converter;
146        
147        /**
148         * Creates a new representation of a header field with the 
149         * given value and converter. For fields with a 
150         * constant definition in this class, the name is normalized.
151         * 
152         * @param name the field name
153         * @param value the value
154         * @param converter the converter
155         */
156        public HttpField(String name, T value, Converter<T> converter) {
157                this.name = fieldNameMap.getOrDefault(name, name);
158                this.converter = converter;
159                this.value = value;
160        }
161
162        /**
163         * Creates a new representation of a header field from its textual
164         * representation. For fields with a 
165         * constant definition in this class, the name is normalized.
166         * 
167         * @param headerLine the header line
168         * @param converter the converter
169         * @throws ParseException if an error occurs while parsing the header line
170         */
171        public HttpField(String headerLine, Converter<T> converter) 
172                        throws ParseException {
173                this.converter = converter;
174                Matcher hlp = headerLinePatter.matcher(headerLine);
175                if (!hlp.matches()) {
176                        throw new ParseException("Invalid header: ", 0);
177                }
178                this.name = fieldNameMap.getOrDefault(hlp.group(1), hlp.group(1));
179                // RFC 7230 3.2.4
180                this.value = converter.fromFieldValue(hlp.group(2).trim());
181        }
182
183        /**
184         * Returns the proper converter for the header field with the given
185         * name. Works for all well known
186         * field names, i.e. the field names defined as constants in this class.
187         * If the field name is unknown, a string converter is returned.
188         * 
189         * @param fieldName
190         *            the field name
191         * @return the converter
192         */
193        public static Converter<?> lookupConverter(String fieldName) {
194                String normalizedFieldName = fieldNameMap
195                                .getOrDefault(fieldName, fieldName);
196                switch (normalizedFieldName) {
197                case ACCEPT:
198                        return Converters.MEDIA_RANGE;
199                case ACCEPT_LANGUAGE:
200                        return Converters.LANGUAGE;
201                case ALLOW:
202                        return Converters.STRING_LIST;
203                case COOKIE:
204                        return Converters.COOKIE_LIST;
205                case CONNECTION:
206                        return Converters.STRING_LIST;
207                case CONTENT_LENGTH:
208                        return Converters.LONG;
209                case CONTENT_LOCATION:
210                        return Converters.URI_CONV; 
211                case CONTENT_TYPE:
212                        return Converters.MEDIA_TYPE;
213                case DATE:
214                        return Converters.DATE_TIME;
215                case IF_MATCH:
216                        return Converters.STRING_LIST;
217                case IF_MODIFIED_SINCE:
218                        return Converters.DATE_TIME;
219                case IF_NONE_MATCH:
220                        return Converters.STRING_LIST;
221                case IF_UNMODIFIED_SINCE:
222                        return Converters.DATE_TIME;
223                case LAST_MODIFIED:
224                        return Converters.DATE_TIME;
225                case LOCATION:
226                        return Converters.URI_CONV; 
227                case MAX_FORWARDS:
228                        return Converters.LONG; 
229                case RETRY_AFTER:
230                        return Converters.DATE_TIME;
231                case SERVER:
232                        return Converters.PRODUCT_DESCRIPTIONS;
233                case SET_COOKIE:
234                        return Converters.SET_COOKIE;
235                case TRAILER:
236                        return Converters.STRING_LIST;
237                case TRANSFER_ENCODING:
238                        return Converters.STRING_LIST;
239                case UPGRADE:
240                        return Converters.STRING_LIST;
241                case USER_AGENT:
242                        return Converters.PRODUCT_DESCRIPTIONS;
243                case VIA:
244                        return Converters.STRING_LIST;
245                default:
246                        return Converters.STRING;
247                }
248        }

297        /**
298         * Returns the string representation of this header field as it appears in
299         * an HTTP message. Note that the returned string may span several
300         * lines (may contain CR/LF), but is has no trailing CR/LF.
301         * 
302         * @return the field as it occurs in a header
303         */
304        public String asHeaderField() {
305                if (!(converter instanceof ListConverter)
306                                || !((ListConverter<?,?>)converter).isSeparateItems()) {
307                        return name() + ": " + asFieldValue();
308                }
309                // Convert list of items to seperate fields
310                @SuppressWarnings("unchecked")
311                ListConverter <List<Object>,Object> listConverter
312                        = (ListConverter <List<Object>,Object>)converter;
313                Converter<Object> itemConverter = listConverter.getItemConverter();
314                @SuppressWarnings("unchecked")
315                List<Object> list = (List<Object>)value();
316                return list.stream().map(
317                                item -> name() + ": " + itemConverter.asFieldValue(item))
318                                .collect(Collectors.joining("\r\n"));
319        }

321        /* (non-Javadoc)
322         * @see java.lang.Object#toString()
323         */
324        @Override
325        public String toString() {
326                StringBuilder result = new StringBuilder();
327                result.append(getClass().getSimpleName());
328                result.append(" [");
329                result.append(asHeaderField().replace("\r\n", " CRLF "));
330                result.append("]");
331                return result.toString();
332        }

334        /**
335         * Determines the length of a token.
336         * 
337         * @param text the text to parse
338         * @param startPos the start position
339         * @return the length of the token
340         */
341        public static int tokenLength(String text, int startPos) {
342                int pos = startPos;
343                while (pos < text.length()
344                                && HttpConstants.TOKEN_CHARS.indexOf(text.charAt(pos)) >= 0) {
345                        pos += 1;
346                }
347                return pos - startPos;
348        }
349
350        /**
351         * Determines the length of a white space sequence. 
352         * 
353         * @param text the test to parse 
354         * @param startPos the start position
355         * @return the length of the white space sequence
356         */
357        public static int whiteSpaceLength(String text, int startPos) {
358                int pos = startPos;
359                while (pos < text.length()) {
360                        switch (text.charAt(pos)) {
361                        case ' ':
362                                // fall through
363                        case '\t':
364                                pos += 1;
365                                continue;
366                                
367                        default:
368                                break;
369                        }
370                        break;
371                }
372                return pos - startPos;
373        }
374        
375        /**
376         * Determines the length of a comment.
377         * 
378         * @param text the text to parse
379         * @param startPos the staring position (must be the position of the
380         * opening brace)
381         * @return the length of the comment
382         */
383        public static int commentLength(String text, int startPos) {
384                int pos = startPos + 1;
385                while (pos < text.length()) {
386                        switch(text.charAt(pos)) {
387                        case ')':
388                                return pos - startPos + 1;
389                                
390                        case '(':
391                                pos += commentLength(text, pos);
392                                break;
393                                
394                        case '\\':
395                                pos = Math.min(pos + 2, text.length());
396                                break;
397                                
398                        default:
399                                pos += 1;
400                                break;
401                        }
402                }
403                return pos - startPos;
404        }
