018package org.jdrupes.httpcodec.protocols.http;
019
020import java.text.ParseException;
021import java.util.Optional;
022
023import org.jdrupes.httpcodec.protocols.http.fields.HttpContentLengthField;
024import org.jdrupes.httpcodec.protocols.http.fields.HttpField;
025import org.jdrupes.httpcodec.protocols.http.fields.HttpMediaTypeField;
026
027/**
028 * Represents an HTTP response header.
029 * 
030 * @author Michael N. Lipp
031 */
032public class HttpResponse extends HttpMessageHeader {
033
034        private int statusCode = -1;
035        private String reasonPhrase;
036        private HttpRequest request;
037        
038        public HttpResponse(HttpProtocol protocol, HttpStatus status, 
039                        boolean messageHasBody) {
040                super(protocol, messageHasBody);
041                setStatus(status);
042        }
043        
044        public HttpResponse(HttpProtocol protocol, int statusCode, 
045                        String reasonPhrase, boolean messageHasBody) {
046                super(protocol, messageHasBody);
047                setStatusCode(statusCode);
048                setReasonPhrase(reasonPhrase);
049        }
050        
051        /* (non-Javadoc)
052         * @see org.jdrupes.httpcodec.protocols.http.HttpMessageHeader#setField(org.jdrupes.httpcodec.fields.HttpField)
053         */
054        @Override
055        public HttpResponse setField(HttpField<?> value) {
056                super.setField(value);
057                return this;
058        }
059
060        /* (non-Javadoc)
061         * @see org.jdrupes.httpcodec.protocols.http.HttpMessageHeader#setMessageHasBody(boolean)
062         */
063        @Override
064        public HttpResponse setMessageHasBody(boolean messageHasBody) {
065                super.setMessageHasBody(messageHasBody);
066                return this;
067        }
068
069        /**
070         * @return the responseCode
071         */
072        public int getStatusCode() {
073                return statusCode;
074        }
075
076        /**
077         * @param statusCode the responseCode to set
078         * @return the response for easy chaining
079         */
080        public HttpResponse setStatusCode(int statusCode) {
081                this.statusCode = statusCode;
082                return this;
083        }
084
085        /**
086         * @return the reason phrase
087         */
088        public String getReasonPhrase() {
089                return reasonPhrase;
090        }
091
092        /**
093         * @param reasonPhrase the reason phrase to set
094         * @return the response for easy chaining
095         */
096        public HttpResponse setReasonPhrase(String reasonPhrase) {
097                this.reasonPhrase = reasonPhrase;
098                return this;
099        }
100
101        /**
102         * Sets both status code and reason phrase from the given 
103         * http status value.
104         * 
105         * @param status the status value
106         * @return the response for easy chaining
107         */
108        public HttpResponse setStatus(HttpStatus status) {
109                statusCode = status.getStatusCode();
110                reasonPhrase = status.getReasonPhrase();
111                return this;
112        }
113        
114        /**
115         * A convenience method for setting the "Content-Type" header.
116         * 
117         * @param type the type
118         * @param subtype the subtype
119         * @return the response for easy chaining
120         * @throws ParseException if the values cannot be parsed
121         */
122        public HttpResponse setContentType(String type, String subtype) 
123                        throws ParseException {
124                return setField(new HttpMediaTypeField
125                                (HttpField.CONTENT_TYPE, type, subtype));
126        }
127
128        /**
129         * A convenience method for setting the "Content-Type" header (usually
130         * of type "text") together with its charset parameter.
131         * 
132         * @param type the type
133         * @param subtype the subtype
134         * @param charset the charset
135         * @return the response for easy chaining
136         * @throws ParseException if the values cannot be parsed
137         */
138        public HttpResponse setContentType(String type, String subtype,
139                        String charset) throws ParseException {
140                HttpMediaTypeField mt = new HttpMediaTypeField(HttpField.CONTENT_TYPE,
141                        type, subtype);
142                mt.setParameter("charset", charset);
143                return setField(mt);
144        }
145        
146        /**
147         * A convenience method for setting the "Content-Length" header.
148         * 
149         * @param length the length
150         * @return the response for easy chaining
151         */
152        public HttpResponse setContentLength(long length) {
153                return setField(new HttpContentLengthField(length));
154        }
155        
156        /**
157         * Associates the response with the request that it responds to. This method
158         * is invoked by the request decoder when it creates the prepared
159         * response for a request. The relationship with the request is required
160         * because information from the request headers may be needed when encoding
161         * the response. 
162         * 
163         * @param request
164         *            the request
165         * @return the response for easy chaining
166         * @see HttpRequest#setResponse(HttpResponse)
167         */
168        public HttpResponse setRequest(HttpRequest request) {
169                this.request = request;
170                return this;
171        }
172        
173        /**
174         * Returns the request that this response responds to.
175         * 
176         * @return the request
177         * @see #setRequest(HttpRequest)
178         */
179        public Optional<HttpRequest> getRequest() {
180                return Optional.ofNullable(request);
181        }
182}
