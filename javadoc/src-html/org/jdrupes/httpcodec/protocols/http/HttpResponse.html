019package org.jdrupes.httpcodec.protocols.http;
020
021import java.io.IOException;
022import java.net.URI;
023import java.nio.file.Files;
024import java.nio.file.Paths;
025import java.text.ParseException;
026import java.util.Optional;
027
028import javax.activation.MimetypesFileTypeMap;
029
030import static org.jdrupes.httpcodec.protocols.http.HttpConstants.*;
031
032import org.jdrupes.httpcodec.protocols.http.HttpConstants.HttpStatus;
033import org.jdrupes.httpcodec.types.Converters;
034import org.jdrupes.httpcodec.types.MediaType;
035
036/**
037 * Represents an HTTP response header.
038 */
039public class HttpResponse extends HttpMessageHeader {
040
041        private static MimetypesFileTypeMap typesMap = new MimetypesFileTypeMap();
042        
043        private int statusCode = -1;
044        private String reasonPhrase;
045        private HttpRequest request;
046        
047        public HttpResponse(HttpProtocol protocol, HttpStatus status, 
048                        boolean hasPayload) {
049                super(protocol, hasPayload);
050                setStatus(status);
051        }
052        
053        public HttpResponse(HttpProtocol protocol, int statusCode, 
054                        String reasonPhrase, boolean hasPayload) {
055                super(protocol, hasPayload);
056                setStatusCode(statusCode);
057                setReasonPhrase(reasonPhrase);
058        }
059        
060        /* (non-Javadoc)
061         * @see HttpMessageHeader#setField(org.jdrupes.httpcodec.fields.HttpField)
062         */
063        @Override
064        public HttpResponse setField(HttpField<?> value) {
065                super.setField(value);
066                return this;
067        }
068        
069        /* (non-Javadoc)
070         * @see HttpMessageHeader#setField(java.lang.String, java.lang.Object)
071         */
072        @Override
073        public <T> HttpResponse setField(String name, T value) {
074                super.setField(name, value);
075                return this;
076        }
077
078        /* (non-Javadoc)
079         * @see org.jdrupes.httpcodec.protocols.http.HttpMessageHeader#setHasPayload(boolean)
080         */
081        @Override
082        public HttpResponse setHasPayload(boolean hasPayload) {
083                super.setHasPayload(hasPayload);
084                return this;
085        }
086
087        /**
088         * @return the responseCode
089         */
090        public int statusCode() {
091                return statusCode;
092        }
093
094        /**
095         * @param statusCode the responseCode to set
096         * @return the response for easy chaining
097         */
098        public HttpResponse setStatusCode(int statusCode) {
099                this.statusCode = statusCode;
100                return this;
101        }
102
103        /**
104         * @return the reason phrase
105         */
106        public String reasonPhrase() {
107                return reasonPhrase;
108        }
109
110        /**
111         * @param reasonPhrase the reason phrase to set
112         * @return the response for easy chaining
113         */
114        public HttpResponse setReasonPhrase(String reasonPhrase) {
115                this.reasonPhrase = reasonPhrase;
116                return this;
117        }
118
119        /**
120         * Sets both status code and reason phrase from the given 
121         * http status value.
122         * 
123         * @param status the status value
124         * @return the response for easy chaining
125         */
126        public HttpResponse setStatus(HttpStatus status) {
127                statusCode = status.statusCode();
128                reasonPhrase = status.reasonPhrase();
129                return this;
130        }
131        
132        /**
133         * A convenience method for setting the "Content-Type" header. Also sets 
134         * the "has payload" flag.
135         * 
136         * @param type the type
137         * @param subtype the subtype
138         * @return the response for easy chaining
139         * @throws ParseException if the values cannot be parsed
140         */
141        public HttpResponse setContentType(String type, String subtype) 
142                        throws ParseException {
143                setField(HttpField.CONTENT_TYPE, new MediaType(type, subtype));
144                setHasPayload(true);
145                return this;
146        }
147
148        /**
149         * A convenience method for setting the "Content-Type" header (usually
150         * of type "text") together with its charset parameter. Also sets 
151         * the "has payload" flag.
152         * 
153         * @param type the type
154         * @param subtype the subtype
155         * @param charset the charset
156         * @return the response for easy chaining
157         * @throws ParseException if the values cannot be parsed
158         */
159        public HttpResponse setContentType(String type, String subtype,
160                        String charset) throws ParseException {
161                setField(HttpField.CONTENT_TYPE,
162                                MediaType.builder().setType(type, subtype)
163                                .setParameter("charset", charset).build());
164                setHasPayload(true);
165                return this;
166        }
167        
168        /**
169         * Convenience method for setting the "Content-Type" header using 
170         * the path information of the given request. Also sets 
171         * the "has payload" flag.
172         * 
173         * @param requestUri the requested resource
174         * @return the response for easy chaining
175         */
176        public HttpResponse setContentType(URI requestUri) {
177                // Get content type
178                String mimeTypeName;
179                try {
180                        // probeContentType is most advanced, but may fail if it tries
181                        // to look at the file's content (which doesn't exist).
182                        mimeTypeName = Files.probeContentType(Paths.get(
183                                        requestUri.getPath()));
184                } catch (IOException e) {
185                        mimeTypeName = null;
186                }
187                if (mimeTypeName == null) {
188                        mimeTypeName = typesMap.getContentType(requestUri.getPath());
189                }
190                MediaType mediaType = null;
191                try {
192                        mediaType = Converters.MEDIA_TYPE.fromFieldValue(mimeTypeName);
193                } catch (ParseException e) {
194                        // Cannot happen
195                }
196                
197                // Send response 
198                if ("text".equals(mediaType.topLevelType())) {
199                        mediaType = MediaType.builder().from(mediaType)
200                                        .setParameter("charset", System.getProperty(
201                                                        "file.encoding", "UTF-8")).build();
202                }
203                setField(HttpField.CONTENT_TYPE, mediaType);
204                setHasPayload(true);
205                return this;
206        }
207        
208        /**
209         * A convenience method for setting the "Content-Length" header.
210         * 
211         * @param length the length
212         * @return the response for easy chaining
213         */
214        public HttpResponse setContentLength(long length) {
215                return setField(new HttpField<>(
216                                HttpField.CONTENT_LENGTH, length, Converters.LONG));
217        }
218        
219        /**
220         * Associates the response with the request that it responds to. This method
221         * is invoked by the request decoder when it creates the prepared
222         * response for a request. The relationship with the request is required
223         * because information from the request headers may be needed when encoding
224         * the response. 
225         * 
226         * @param request
227         *            the request
228         * @return the response for easy chaining
229         * @see HttpRequest#setResponse(HttpResponse)
230         */
231        public HttpResponse setRequest(HttpRequest request) {
232                this.request = request;
233                return this;
234        }
235        
236        /**
237         * Returns the request that this response responds to.
238         * 
239         * @return the request
240         * @see #setRequest(HttpRequest)
241         */
242        public Optional<HttpRequest> request() {
243                return Optional.ofNullable(request);
244        }
245
