043/**
044 * Implements an encoder for HTTP. The class can be used as base class for both
045 * a request and a response encoder.
046 * 
047 * @author Michael N. Lipp
048 */
049public abstract class HttpEncoder<T extends HttpMessageHeader> 
050        extends HttpCodec<T> implements Encoder<T> {
051
052        private enum State {
053                // Main states
054                INITIAL, DONE, CLOSED,
055                // Sub states
056                HEADERS, CHUNK_BODY, STREAM_CHUNK, FINISH_CHUNK, FINISH_CHUNKED,
057                START_COLLECT_BODY, COLLECT_BODY, STREAM_COLLECTED, 
058                STREAM_BODY, FLUSH_ENCODER
059        }
060
061        private Stack<State> states = new Stack<>();
062        private boolean closeAfterBody = false;
063        private ByteBufferOutputStream outStream;
064        private Writer writer;
065        private Iterator<HttpField<?>> headerIter = null;
066        private int pendingLimit = 1024 * 1024;
067        private long leftToStream;
068        private ByteBufferOutputStream collectedBodyData;
069        private CharsetEncoder charEncoder = null;
070        private Writer charWriter = null;
071        private ByteBuffer chunkData;
072
073        /**
074         * Creates a new encoder.
075         */
076        public HttpEncoder() {
077                outStream = new ByteBufferOutputStream();
078                try {
079                        writer = new OutputStreamWriter(outStream, "ascii");
080                } catch (UnsupportedEncodingException e) {
081                }
082                states.push(State.INITIAL);
083        }
084
085        /**
086         * Returns the limit for pending body bytes. If the protocol is HTTP/1.0 and
087         * the message has a body but no "Content-Length" header, the only
088         * (unreliable) way to indicate the end of the body is to close the
089         * connection after all body bytes have been sent.
090         * <P>
091         * The encoder tries to calculate the content length by buffering the body
092         * data up to the "pending" limit. If the body is smaller than the limit,
093         * the message is set with the calculated content length header, else the
094         * data is sent without such a header and the connection is closed.
095         * <P>
096         * If the response protocol is HTTP/1.1 and there is no "Content-Length"
097         * header, chunked transfer encoding is used.
098         *
099         * @return the limit
100         */
101        public int getPendingLimit() {
102                return pendingLimit;
103        }
104
105        /**
106         * Sets the limit for the pending body bytes.
107         * 
108         * @param pendingLimit
109         *            the limit to set
110         */
111        public void setPendingLimit(int pendingLimit) {
112                this.pendingLimit = pendingLimit;
113        }
114
115        /**
116         * Returns {@code true} if the encoder does not accept further input because
117         * the processed data indicated that the connection has been or is to be
118         * closed.
119         * 
120         * @return the result
121         */
122        public boolean isClosed() {
123                return states.peek() == State.CLOSED;
124        }

126        /**
127         * Writes the first line of the message (including the terminating CRLF).
128         * Must be provided by the derived class because the first line depends on
129         * whether a request or response is encoded.
130         * 
131         * @param messageHeader
132         *            the message header to encode (see
133         *            {@link #encode(HttpMessageHeader)}
134         * @param writer
135         *            the Writer to use for writing
136         * @throws IOException
137         *             if an I/O error occurs
138         */
139        protected abstract void startMessage(T messageHeader, Writer writer) 
140                        throws IOException;

155        /**
156         * Set a new HTTP message that is to be encoded.
157         * 
158         * @param messageHeader
159         *            the response
160         */
161        public void encode(T messageHeader) {
162                if (states.peek() != State.INITIAL) {
163                        throw new IllegalStateException();
164                }
165                this.messageHeader = messageHeader;
166                charEncoder = null;
167                charWriter = null;
168        }

170        /**
171         * Encodes a HTTP message.
172         * 
173         * @param in
174         *            the body data
175         * @param out
176         *            the buffer to which data is written
177         * @param endOfInput
178         *            {@code true} if there is no input left beyond the data
179         *            currently in the {@code in} buffer (indicates end of body or
180         *            no body at all)
181         * @return the result
182         */
183        public Encoder.Result encode(Buffer in, ByteBuffer out,
184                boolean endOfInput) {
185                outStream.assignBuffer(out);
186                Encoder.Result result = newResult(false, false, false);
187                if (out.remaining() == 0) {
188                        return newResult(true, false, false);
189                }
190                while (true) {
191                        if (result.isOverflow() || result.isUnderflow()) {
192                                return result;
193                        }
194                        switch (states.peek()) {
195                        case INITIAL:
196                                outStream.clear();
197                                outStream.assignBuffer(out);
198                                startMessage();
199                                break;
200
201                        case HEADERS:
202                                // If headers remain (new request or buffer was full) write them
203                                if (writeHeaders()) {
204                                        continue;
205                                }
206                                break;
207
208                        case START_COLLECT_BODY:
209                                // Start collecting
210                                if (in.remaining() == 0 && !endOfInput) {
211                                        // Has probably been invoked with a dummy buffer,
212                                        // cannot be used to create pending body buffer.
213                                        return newResult(false, true, false);
214                                }
215                                collectedBodyData = new ByteBufferOutputStream(
216                                        Math.min(pendingLimit, 
217                                                        Math.max(in.capacity(), 64 * 1024)));
218                                states.pop();
219                                states.push(State.COLLECT_BODY);
220                                // fall through (no write occurred yet)
221                        case COLLECT_BODY:
222                                result = collectBody(in, endOfInput);
223                                continue; // never writes to out
224
225                        case STREAM_COLLECTED:
226                                // Output collected body
227                                if (collectedBodyData.remaining() < 0) {
228                                        collectedBodyData.assignBuffer(out);
229                                        break;
230                                }
231                                states.pop();
232                                continue;
233
234                        case STREAM_BODY:
235                                // Stream, i.e. simply copy from source to destination
236                                int initiallyRemaining = out.remaining();
237                                result = copyBodyData(in, out, endOfInput);
238                                leftToStream -= (initiallyRemaining - out.remaining());
239                                if (!result.isOverflow() && (leftToStream == 0 || endOfInput)) {
240                                        // end of data
241                                        states.pop();
242                                        if (charEncoder != null) {
243                                                states.push(State.FLUSH_ENCODER);
244                                        }
245                                        continue;
246                                }
247                                // Buffer written, waiting for space, more data or end of data
248                                return result;
249
250                        case FLUSH_ENCODER:
251                                CoderResult flushResult = charEncoder.flush(out);
252                                if (flushResult.isOverflow()) {
253                                        return newResult(true, false, false);
254                                }
255                                states.pop();
256                                break;
257                                
258                        case CHUNK_BODY:
259                                // Send in data as chunk
260                                result = startChunk(in, out, endOfInput);
261                                continue; // remaining check already done 
262
263                        case STREAM_CHUNK:
264                                // Stream, i.e. simply copy from source to destination
265                                if (!ByteBufferUtils.putAsMuchAsPossible(out, chunkData)) {
266                                        // Not enough space in out buffer
267                                        return newResult(true, false, false);
268                                }
269                                // everything from in written
270                                states.pop();
271                                continue;
272        
273                        case FINISH_CHUNK:
274                                try {
275                                        outStream.write("\r\n".getBytes("ascii"));
276                                        states.pop();
277                                } catch (IOException e) {
278                                        // Formally thrown by write
279                                }
280                                break;
281                                
282                        case FINISH_CHUNKED:
283                                try {
284                                        outStream.write("0\r\n\r\n".getBytes("ascii"));
285                                        states.pop();
286                                } catch (IOException e) {
287                                        // Formally thrown by write
288                                }
289                                break;
290                                
291                        case DONE:
292                                // Everything is written
293                                if (!endOfInput) {
294                                        if (in.remaining() == 0) {
295                                                return newResult(false, true, false);
296                                        }
297                                        throw new IllegalStateException("Unexpected input.");
298                                }
299                                states.pop();
300                                if (closeAfterBody) {
301                                        states.push(State.CLOSED);
302                                } else {
303                                        states.push(State.INITIAL);
304                                }
305                                return newResult(false, false, false);
306
307                        default:
308                                throw new IllegalStateException();
309                        }
310                        // Using "continue" above avoids this check. Use it only
311                        // if the state has changed and no addition output is expected. 
312                        if (out.remaining() == 0) {
313                                return newResult(true, false, false);
314                        }
315                }
316        }

318        /**
319         * Called during the initial state. Writes the status line, determines the
320         * body-mode and puts the state machine in output-headers mode, unless the
321         * body-mode is "collect body".
322         */
323        private void startMessage() {
324                // Make sure we have a Date, RFC 7231 7.1.1.2
325                if (!messageHeader.fields().containsKey(HttpField.DATE)) {
326                        messageHeader.setField(new HttpDateField());
327                }
328
329                // Complete content type
330                HttpMediaTypeField contentType = (HttpMediaTypeField) messageHeader
331                        .fields().get(HttpField.CONTENT_TYPE);
332                String charset = null;
333                if (contentType != null) {
334                        charset = contentType.getParameter("charset");
335                        if (charset == null) {
336                                charset = "utf-8";
337                                contentType.setParameter("charset", charset);
338                        }
339                }
340                
341                // Prepare encoder
342                headerIter = null;
343
344                // Write request or status line
345                try {
346                        startMessage(messageHeader, writer);
347                } catch (IOException e) {
348                        // Formally thrown by writer, cannot happen.
349                }
350                states.pop();
351                // We'll eventually fall back to this state
352                states.push(State.DONE);
353                // Get a default for closeAfterBody from the header fields
354                closeAfterBody = messageHeader
355                        .getField(HttpStringListField.class, HttpField.CONNECTION)
356                        .map(f -> f.containsIgnoreCase("close")).orElse(false);
357                // If there's no body, start outputting header fields
358                if (!messageHeader.messageHasBody()) {
359                        states.push(State.HEADERS);
360                        return;
361                }
362                // Message has a body, find out how to handle it
363                HttpIntField cl = messageHeader.getField(HttpIntField.class,
364                        HttpField.CONTENT_LENGTH).orElse(null);
365                leftToStream = (cl == null ? -1 : cl.getValue());
366                if (leftToStream >= 0) {
367                        // Easiest: we have a content length, works always
368                        states.push(State.STREAM_BODY);
369                        states.push(State.HEADERS);
370                        return;
371                } 
372                if (messageHeader.getProtocol()
373                        .compareTo(HttpProtocol.HTTP_1_0) > 0) {
374                        // At least 1.1, use chunked
375                        HttpStringListField transEnc = messageHeader.getField(
376                                HttpStringListField.class, HttpField.TRANSFER_ENCODING)
377                                        .orElse(null);
378                        if (transEnc == null) {
379                                messageHeader.setField(new HttpStringListField(
380                                        HttpField.TRANSFER_ENCODING,
381                                        TransferCoding.CHUNKED.toString()));
382                        } else {
383                                transEnc.remove(TransferCoding.CHUNKED.toString());
384                                transEnc.add(TransferCoding.CHUNKED.toString());
385                        }
386                        states.push(State.CHUNK_BODY);
387                        states.push(State.HEADERS);
388                        return;
389                }
390                // Bad: 1.0 and no content length.
391                if (pendingLimit > 0) {
392                        // Try to calculate length by collecting the data
393                        leftToStream = 0;
394                        states.push(State.START_COLLECT_BODY);
395                        return;
396                }
397                // May not buffer, use close
398                states.push(State.STREAM_BODY);
399                states.push(State.HEADERS);
400                closeAfterBody = true;
401        }

403        /**
404         * Outputs as many headers as fit in the current out buffer. If all headers
405         * are output, pops a state (thus proceeding with the selected body mode).
406         * Unless very small out buffers are used (or very large headers occur),
407         * this is invoked only once. Therefore no attempt has been made to avoid
408         * the usage of temporary buffers in the header header stream (there may be
409         * a maximum overflow of one partial header).
410         * 
411         * @return {@code true} if all headers could be written to the out
412         * buffer (nothing pending)
413         */
414        private boolean writeHeaders() {
415                try {
416                        if (headerIter == null) {
417                                headerIter = messageHeader.fields().values().iterator();
418                        }
419                        while (true) {
420                                if (!headerIter.hasNext()) {
421                                        writer.write("\r\n");
422                                        writer.flush();
423                                        states.pop();
424                                        return outStream.remaining() >= 0;
425                                }
426                                HttpField<?> header = headerIter.next();
427                                writer.write(header.asHeaderField());
428                                writer.write("\r\n");
429                                writer.flush();
430                                if (outStream.remaining() <= 0) {
431                                        break;
432                                }
433                        }
434                } catch (IOException e) {
435                        // Formally thrown by writer, cannot happen.
436                }
437                return false;
438        }

440        /**
441         * Copy as much data as possible from in to out.
442         * 
443         * @param in
444         * @param out
445         * @param endOfInput
446         */
447        private Encoder.Result copyBodyData(Buffer in, ByteBuffer out,
448                boolean endOfInput) {
449                if (in instanceof CharBuffer) {
450                        // copy via encoder
451                        if (charEncoder == null) {
452                                charEncoder = Charset.forName(bodyCharset()).newEncoder();
453                        }
454                        CoderResult result = charEncoder.encode((CharBuffer) in, out,
455                                endOfInput);
456                        return newResult(result.isOverflow(), 
457                                        !endOfInput && result.isUnderflow(), false);
458                }
459                if (out.remaining() <= leftToStream) {
460                        ByteBufferUtils.putAsMuchAsPossible(out, (ByteBuffer) in);
461                } else {
462                        ByteBufferUtils.putAsMuchAsPossible(out, (ByteBuffer) in,
463                                (int) leftToStream);
464                }
465                return newResult(out.remaining() == 0, 
466                                !endOfInput && in.remaining() == 0, false);
467        }

469        /**
470         * Handle the input as appropriate for the collect-body mode.
471         * 
472         * @param in
473         * @param endOfInput
474         */
475        private Encoder.Result collectBody(Buffer in, boolean endOfInput) {
476                if (collectedBodyData.remaining() - in.remaining() < -pendingLimit) {
477                        // No space left, output headers, collected and rest (and then
478                        // close)
479                        states.pop();
480                        closeAfterBody = true;
481                        leftToStream = Long.MAX_VALUE;
482                        states.push(State.STREAM_BODY);
483                        states.push(State.STREAM_COLLECTED);
484                        states.push(State.HEADERS);
485                        return newResult(false, false, false);
486                }
487                // Space left, collect
488                if (in instanceof ByteBuffer) {
489                        collectedBodyData.write((ByteBuffer)in);
490                } else {
491                        if (charWriter == null) {
492                                try {
493                                        charWriter = new OutputStreamWriter(collectedBodyData,
494                                                bodyCharset());
495                                } catch (UnsupportedEncodingException e) {
496                                        throw new IllegalArgumentException(e);
497                                }
498                        }
499                        try {
500                                if (in.hasArray()) {
501                                        // more efficient than CharSequence
502                                        charWriter.write(((CharBuffer) in).array(),
503                                                in.arrayOffset() + in.position(), in.remaining());
504                                } else {
505                                        charWriter.append((CharBuffer)in);
506                                }
507                                in.position(in.limit());
508                                charWriter.flush();
509                        } catch (IOException e) {
510                                // Formally thrown, cannot happen
511                        }
512                }
513                if (endOfInput) {
514                        // End of body, found content length!
515                        messageHeader.setField(new HttpContentLengthField(
516                                collectedBodyData.bytesWritten()));
517                        states.pop();
518                        states.push(State.STREAM_COLLECTED);
519                        states.push(State.HEADERS);
520                        return newResult(false, false, false);
521                }
522                // Get more input
523                return newResult(false, true, false);
524        }

526        /**
527         * Handle the input as appropriate for the chunked-body mode.
528         * 
529         * @param in
530         * @return
531         */
532        private Encoder.Result startChunk
533                (Buffer in, ByteBuffer out, boolean endOfInput) {
534                if (endOfInput) {
535                        states.pop();
536                        states.push(State.FINISH_CHUNKED);
537                }
538                try {
539                        // Don't write zero sized chunks
540                        if (in.hasRemaining()) {
541                                if (in instanceof CharBuffer) {
542                                        if (charEncoder == null) {
543                                                charEncoder = Charset.forName(bodyCharset())
544                                                        .newEncoder();
545                                        }
546                                        chunkData = charEncoder.encode((CharBuffer)in);
547                                } else {
548                                        chunkData = (ByteBuffer)in;
549                                }
550                                outStream.write(
551                                        Long.toHexString(chunkData.remaining())
552                                                .getBytes("ascii"));
553                                outStream.write("\r\n".getBytes("ascii"));
554                                states.push(State.FINISH_CHUNK);
555                                states.push(State.STREAM_CHUNK);
556                                return newResult(outStream.remaining() <= 0, false, false);
557                        }
558                } catch (IOException e) {
559                        // Formally thrown by outStream, cannot happen.
560                }
561                return newResult(outStream.remaining() < 0, 
562                                in.remaining() == 0 && !endOfInput, false);
563        }
