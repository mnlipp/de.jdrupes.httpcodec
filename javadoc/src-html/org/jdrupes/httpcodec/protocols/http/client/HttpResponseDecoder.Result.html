041/**
042 * A decoder for HTTP reponses that accepts data from a sequence of
043 * {@link ByteBuffer}s and decodes them into {@link HttpResponse}s
044 * and their (optional) payload.
045 * 
046 * ![HttpResponseDecoder](httpresponsedecoder.svg)
047 * 
048 * Headers
049 * -------
050 * 
051 * The decoder converts a `Retry-After` header with a delay value to
052 * a header with a date by adding the span to the time in the `Date` header.
053 *  
054 * 
055 * @see "[RFC 7231, Section 7.1.3](https://tools.ietf.org/html/rfc7231#section-7.1.3)"
056 * 
057 * @startuml httpresponsedecoder.svg
058 * class HttpResponseDecoder {
059 *      +HttpRequestDecoder()
060 *      +Result decode(ByteBuffer in, Buffer out, boolean endOfInput)
061 * }
062 * 
063 * class HttpDecoder<T extends HttpMessageHeader, R extends HttpMessageHeader> {
064 * }
065 * 
066 * HttpDecoder <|-- HttpResponseDecoder: <<bind>> <T -> HttpResponse, R -> HttpRequest>  
067 * @enduml
068 */
069public class HttpResponseDecoder 
070        extends HttpDecoder<HttpResponse, HttpRequest>
071        implements ResponseDecoder<HttpResponse, HttpRequest> {
072
073        // RFC 7230 3.1.2
074        private static final Pattern responseLinePatter = Pattern
075                .compile("^(" + HTTP_VERSION + ")" + SP + "([1-9][0-9][0-9])"
076                        + SP + "(.*)$");
077
078        private final Result.Factory resultFactory      = new Result.Factory(this);
079        private String requestMethod = "";
080        private boolean reportHeaderReceived = false;
081

083         * @see org.jdrupes.httpcodec.protocols.http.HttpDecoder#resultFactory()
084         */

087                return resultFactory;

095                return HttpResponse.class;
096        }

098        /**
099         * Starts decoding a new response to a given request.
100         * Specifying the request is necessary because the existence of a body
101         * cannot be derived by looking at the header only. It depends on the kind
102         * of request made. Must be called before the response is decoded.
103         * 
104         * @param request
105         *            the request
106         */
107        public void decodeResponseTo(HttpRequest request) {
108                this.requestMethod = request.method();
109        }

112         * @see org.jdrupes.httpcodec.internal.Decoder#decode(java.nio.ByteBuffer, java.nio.ByteBuffer)
113         */
114        @Override
115        public Result decode(ByteBuffer in, Buffer out, boolean endOfInput)
116                throws HttpProtocolException {
117                return (Result)super.decode(in, out, endOfInput);
118        }

120        /**
121         * Checks whether the first line of a message is a valid response.
122         * If so, create a new response message object with basic information, else
123         * throw an exception.
124         * <P>
125         * Called by the base class when a first line is received.
126         * 
127         * @param startLine the first line
128         * @throws HttpProtocolException if the line is not a correct request line
129         */
130        @Override
131        protected HttpResponse newMessage(String startLine)
132                throws HttpProtocolException {
133                Matcher responseMatcher = responseLinePatter.matcher(startLine);
134                if (!responseMatcher.matches()) {
135                        throw new HttpProtocolException(protocolVersion,
136                                HttpStatus.BAD_REQUEST.statusCode(),
137                                "Illegal request line");
138                }
139                String httpVersion = responseMatcher.group(1);
140                int statusCode = Integer.parseInt(responseMatcher.group(2));
141                String reasonPhrase = responseMatcher.group(3);
142                boolean found = false;
143                for (HttpProtocol v : HttpProtocol.values()) {
144                        if (v.toString().equals(httpVersion)) {
145                                protocolVersion = v;
146                                found = true;
147                        }
148                }
149                if (!found) {
150                        throw new HttpProtocolException(HttpProtocol.HTTP_1_1,
151                                HttpStatus.HTTP_VERSION_NOT_SUPPORTED);
152                }
153                return new HttpResponse(protocolVersion, statusCode, reasonPhrase,
154                        false);
155        }
156
157        /*
158         * (non-Javadoc)
159         * 
160         * @see
161         * org.jdrupes.httpcodec.HttpDecoder#headerReceived(org.jdrupes.httpcodec.
162         * HttpMessage)
163         */
164        @Override
165        protected BodyMode headerReceived(HttpResponse message)
166                throws HttpProtocolException {
167                reportHeaderReceived = true;
168                // Adjust Retry-After
169                HttpField<?> hdr = message.fields().get(HttpField.RETRY_AFTER);
170                if (hdr != null && String.class
171                                .isAssignableFrom(hdr.value().getClass())) {
172                        String value = (String)hdr.value();
173                        if (Character.isDigit(value.charAt(0))) {
174                                Instant base = message.findField(
175                                                HttpField.DATE, Converters.DATE_TIME)
176                                                .map(HttpField<Instant>::value).orElse(Instant.now());
177                                message.setField(new HttpField<>(HttpField.RETRY_AFTER,
178                                                base.plusSeconds(Long.parseLong(value)),
179                                                Converters.DATE_TIME));
180                        }
181                }
182                // RFC 7230 3.3.3 (1. & 2.)
183                int statusCode = message.statusCode();
184                if (requestMethod.equalsIgnoreCase("HEAD")
185                        || (statusCode % 100) == 1
186                        || statusCode == 204
187                        || statusCode == 304
188                        || (requestMethod.equalsIgnoreCase("CONNECT")
189                                && (statusCode % 100 == 2))) {
190                        return BodyMode.NO_BODY;
191                }
192                Optional<HttpField<StringList>> transEncs = message.findField(
193                        HttpField.TRANSFER_ENCODING, Converters.STRING_LIST);
194                // RFC 7230 3.3.3 (3.)
195                if (transEncs.isPresent()) {
196                        StringList values = transEncs.get().value(); 
197                        if (values.get(values.size() - 1)
198                                .equalsIgnoreCase(TransferCoding.CHUNKED.toString())) {
199                                return BodyMode.CHUNKED;
200                        } else {
201                                return BodyMode.UNTIL_CLOSE;
202                        }
203                }
204                // RFC 7230 3.3.3 (5.)
205                if (message.fields().containsKey(HttpField.CONTENT_LENGTH)) {
206                        return BodyMode.LENGTH;
207                }
208                // RFC 7230 3.3.3 (7.)
209                return BodyMode.UNTIL_CLOSE;
210        }

212        /**
213         * The result from encoding a response. In addition to the usual
214         * codec result, a result decoder may signal to the invoker that the
215         * connection to the responder must be closed.
216         * 
217         * The class is declared abstract to promote the usage of the factory
218         * method.
219         */
220        public abstract static class Result extends HttpDecoder.Result<HttpRequest>
221                implements Codec.ProtocolSwitchResult {
222
223                private String newProtocol;
224                private ResponseDecoder<?, ?> newDecoder;
225                private Encoder<?> newEncoder;
226                
227                /**
228                 * Returns a new result.
229                 * 
230                 * @param overflow
231                 *            {@code true} if the data didn't fit in the out buffer
232                 * @param underflow
233                 *            {@code true} if more data is expected
234                 * @param closeConnection
235                 *            {@code true} if the connection should be closed
236                 * @param headerCompleted
237                 *            indicates that the message header has been completed and
238                 *            the message (without body) is available
239                 * @param newProtocol the name of the new protocol if a switch occurred
240                 * @param newDecoder the new decoder if a switch occurred
241                 * @param newEncoder the new decoder if a switch occurred
242                 */
243                protected Result(boolean overflow, boolean underflow,
244                        boolean closeConnection, boolean headerCompleted,
245                        String newProtocol, ResponseDecoder<?, ?> newDecoder, 
246                        Encoder<?> newEncoder) {
247                        super(overflow, underflow, closeConnection, headerCompleted,
248                                        null, false);
249                        this.newProtocol = newProtocol;
250                        this.newDecoder = newDecoder;
251                        this.newEncoder = newEncoder;
252                }
253
254                @Override
255                public String newProtocol() {
256                        return newProtocol;
257                }
258                
259                @Override
260                public ResponseDecoder<?, ?> newDecoder() {
261                        return newDecoder;
262                }
263                
264                @Override
265                public Encoder<?> newEncoder() {
266                        return newEncoder;
267                }
268
269                /* (non-Javadoc)
270                 * @see java.lang.Object#hashCode()
271                 */
272                @Override
273                public int hashCode() {
274                        final int prime = 31;
275                        int result = super.hashCode();
276                        result = prime * result
277                                + ((newDecoder == null) ? 0 : newDecoder.hashCode());
278                        result = prime * result
279                                + ((newEncoder == null) ? 0 : newEncoder.hashCode());
280                        result = prime * result
281                                + ((newProtocol == null) ? 0 : newProtocol.hashCode());
282                        return result;
283                }
284
285                /* (non-Javadoc)
286                 * @see java.lang.Object#equals(java.lang.Object)
287                 */
288                @Override
289                public boolean equals(Object obj) {
290                        if (this == obj) {
291                                return true;
292                        }
293                        if (!super.equals(obj)) {
294                                return false;
295                        }
296                        if (!(obj instanceof Result)) {
297                                return false;
298                        }
299                        Result other = (Result) obj;
300                        if (newDecoder == null) {
301                                if (other.newDecoder != null) {
302                                        return false;
303                                }
304                        } else if (!newDecoder.equals(other.newDecoder)) {
305                                return false;
306                        }
307                        if (newEncoder == null) {
308                                if (other.newEncoder != null) {
309                                        return false;
310                                }
311                        } else if (!newEncoder.equals(other.newEncoder)) {
312                                return false;
313                        }
314                        if (newProtocol == null) {
315                                if (other.newProtocol != null) {
316                                        return false;
317                                }
318                        } else if (!newProtocol.equals(other.newProtocol)) {
319                                return false;
320                        }
321                        return true;
322                }
323
324                /* (non-Javadoc)
325                 * @see java.lang.Object#toString()
326                 */
327                @Override
328                public String toString() {
329                        StringBuilder builder = new StringBuilder();
330                        builder.append("HttpResponseDecoder.Result [overflow=");
331                        builder.append(isOverflow());
332                        builder.append(", underflow=");
333                        builder.append(isUnderflow());
334                        builder.append(", closeConnection=");
335                        builder.append(closeConnection());
336                        builder.append(", headerCompleted=");
337                        builder.append(isHeaderCompleted());
338                        builder.append(", ");
339                        if (response() != null) {
340                                builder.append("response=");
341                                builder.append(response());
342                                builder.append(", ");
343                        }
344                        builder.append("responseOnly=");
345                        builder.append(isResponseOnly());
346                        builder.append(", ");
347                        if (newProtocol != null) {
348                                builder.append("newProtocol=");
349                                builder.append(newProtocol);
350                                builder.append(", ");
351                        }
352                        if (newDecoder != null) {
353                                builder.append("newDecoder=");
354                                builder.append(newDecoder);
355                                builder.append(", ");
356                        }
357                        if (newEncoder != null) {
358                                builder.append("newEncoder=");
359                                builder.append(newEncoder);
360                        }
361                        builder.append("]");
362                        return builder.toString();
363                }
364                
365                /**
366                 * The Factory for (extended) results.
367                 */
368                protected static class Factory
369                        extends HttpDecoder.Result.Factory<HttpRequest> {
370
371                        private HttpResponseDecoder decoder;
372
373                        /**
374                         * Creates a new factory for the given decoder. 
375                         * 
376                         * @param decoder the decoder
377                         */
378                        protected Factory(HttpResponseDecoder decoder) {
379                                super();
380                                this.decoder = decoder;
381                        }
382
383                        /* (non-Javadoc)
384                         * @see HttpDecoder.Result.Factory#newResult(boolean, boolean, boolean)
385                         */
386                        @Override
387                        protected Result newResult(
388                                boolean overflow, boolean underflow) {
389                                Result result = new Result(overflow, underflow, decoder.isClosed(),
390                                                decoder.reportHeaderReceived, null, null, null) {
391                                };
392                                decoder.reportHeaderReceived = false;
393                                return result;
394                        }
395                }
396        }
