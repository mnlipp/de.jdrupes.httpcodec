019package org.jdrupes.httpcodec.protocols.http;
020
021import java.net.URI;
022import java.net.URISyntaxException;
023import java.util.Optional;
024
025import static org.jdrupes.httpcodec.protocols.http.HttpConstants.*;
026
027/**
028 * Represents an HTTP request header.
029 */
030public class HttpRequest extends HttpMessageHeader {
031
032        public static final URI ASTERISK_REQUEST 
033                = createUri("http://127.0.0.1/");
034        
035        private String method;
036        private URI requestUri;
037        private String host;
038        private int port;
039        private HttpResponse response;
040
041        /**
042         * Helper for initialization.
043         * 
044         * @param uri as string
045         * @return the URI
046         */
047        private static URI createUri(String uri) {
048                try {
049                        return new URI(uri);
050                } catch (URISyntaxException e) {
051                        throw new IllegalArgumentException(e);
052                }
053        }
054        
055        /**
056         * Creates a new request with basic data. 
057         * 
058         * @param method the method
059         * @param requestUri the requested resource
060         * @param httpProtocol the HTTP protocol version
061         * @param messageHasBody indicates that the message has a body
062         */
063        public HttpRequest(String method, URI requestUri, 
064                        HttpProtocol httpProtocol, boolean messageHasBody) {
065                super(httpProtocol, messageHasBody);
066                this.method = method;
067                this.requestUri = requestUri;
068        }
069
070        /* (non-Javadoc)
071         * @see HttpMessageHeader#setField(org.jdrupes.httpcodec.fields.HttpField)
072         */
073        @Override
074        public HttpRequest setField(HttpField<?> value) {
075                super.setField(value);
076                return this;
077        }
078
079        /* (non-Javadoc)
080         * @see HttpMessageHeader#setField(java.lang.String, java.lang.Object)
081         */
082        @Override
083        public <T> HttpRequest setField(String name, T value) {
084                super.setField(name, value);
085                return this;
086        }
087
088        /* (non-Javadoc)
089         * @see org.jdrupes.httpcodec.protocols.http.HttpMessageHeader#setMessageHasBody(boolean)
090         */
091        @Override
092        public HttpRequest setMessageHasBody(boolean messageHasBody) {
093                super.setMessageHasBody(messageHasBody);
094                return this;
095        }
096
097        /**
098         * Return the method.
099         * 
100         * @return the method
101         */
102        public String getMethod() {
103                return method;
104        }
105
106        /**
107         * Return the URI of the requested resource.
108         * 
109         * @return the requestUri
110         */
111        public URI getRequestUri() {
112                return requestUri;
113        }
114
115        /**
116         * Set the host and port attributes.
117         * 
118         * @param host the host
119         * @param port the port
120         * @return the request for easy chaining
121         */
122        public HttpRequest setHostAndPort(String host, int port) {
123                this.host = host;
124                this.port = port;
125                return this;
126        }
127
128        /**
129         * @return the host
130         */
131        public String getHost() {
132                return host;
133        }
134
135        /**
136         * @return the port
137         */
138        public int getPort() {
139                return port;
140        }
141
142        /**
143         * Associates the request with a response. This method is
144         * invoked by the request decoder that initializes the response with
145         * basic information that can be derived from the request 
146         * (e.g. by default the HTTP version is copied). The status code
147         * of the preliminary response is 501 "Not implemented".
148         * <P>
149         * Although not strictly required, users of the API are encouraged to 
150         * modify this prepared request and use it when building the response.
151         *  
152         * @param response the prepared response
153         * @return the request for easy chaining
154         */
155        public HttpRequest setResponse(HttpResponse response) {
156                this.response = response;
157                return this;
158        }
159        
160        /**
161         * Returns the prepared response.
162         * 
163         * @return the prepared response
164         * @see #setResponse(HttpResponse)
165         */
166        public Optional<HttpResponse> getResponse() {
167                return Optional.ofNullable(response);
168        }
169        
170        /* (non-Javadoc)
171         * @see java.lang.Object#toString()
172         */
173        @Override
174        public String toString() {
175                StringBuilder builder = new StringBuilder();
176                builder.append("HttpRequest [");
177                if (method != null) {
178                        builder.append("method=");
179                        builder.append(method);
180                        builder.append(", ");
181                }
182                if (requestUri != null) {
183                        builder.append("requestUri=");
184                        builder.append(requestUri);
185                        builder.append(", ");
186                }
187                if (getProtocol() != null) {
188                        builder.append("httpVersion=");
189                        builder.append(getProtocol());
190                }
191                builder.append("]");
192                return builder.toString();
193        }
194        
195
196}
