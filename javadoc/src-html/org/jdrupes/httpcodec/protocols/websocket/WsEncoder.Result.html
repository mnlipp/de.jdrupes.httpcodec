036/**
037 * The Websocket encoder.
038 */
039public class WsEncoder implements Encoder<WsFrameHeader> {
040
041        private static enum State { STARTING_FRAME, WRITING_HEADER,  
042                WRITING_LENGTH, WRITING_MASK, WRITING_PAYLOAD }
043        
044        private static float bytesPerCharUtf8           
045                = Charset.forName("utf-8").newEncoder().averageBytesPerChar();
046        private static final Result.Factory resultFactory = new Result.Factory();
047        
048        private SecureRandom randoms = new SecureRandom();
049        private State state = State.STARTING_FRAME;
050        private boolean continuationFrame;
051        private Stack<WsFrameHeader> messageHeaders = new Stack<>();
052        private int headerHead;
053        private long bytesToSend;
054        private long payloadSize;
055        private int payloadBytes;
056        private boolean doMask = false;
057        private byte[] maskingKey = new byte[4];
058        private int maskIndex;
059        private ByteBufferOutputStream convData = new ByteBufferOutputStream();
060
061        /**
062         * Creates new encoder.
063         * 
064         * @param mask set if the data is to be masked (client)
065         */
066        public WsEncoder(boolean mask) {
067                super();
068                this.doMask = mask;
069        }
070
071        /**
072         * Returns the result factory for this codec.
073         * 
074         * @return the factory
075         */
076        protected Result.Factory resultFactory() {
077                return resultFactory;
078        }
079        
080        /* (non-Javadoc)
081         * @see org.jdrupes.httpcodec.Encoder#encoding()
082         */
083        @Override
084        public Class<WsFrameHeader> encoding() {
085                return WsFrameHeader.class;
086        }
087
088        private Result frameFinished(boolean endOfInput) {
089                final boolean close = (messageHeaders.peek() instanceof WsCloseFrame);
090                if (!(messageHeaders.peek() instanceof WsMessageHeader) 
091                                || endOfInput) {
092                        messageHeaders.pop();
093                }
094                state = State.STARTING_FRAME;
095                bytesToSend = 2;
096                return resultFactory().newResult(false, 
097                                !endOfInput || !messageHeaders.isEmpty(), close);
098        }
099        
100        /* (non-Javadoc)
101         * @see org.jdrupes.httpcodec.ResponseEncoder#encode(org.jdrupes.httpcodec.MessageHeader)
102         */
103        @Override
104        public void encode(WsFrameHeader messageHeader) {
105                if (state != State.STARTING_FRAME) {
106                        throw new IllegalStateException(
107                                        "Trying to start new frame while previous "
108                                                + "has not completely been sent");
109                }
110                if (messageHeader instanceof WsMessageHeader) {
111                        messageHeaders.clear();
112                        messageHeaders.push(messageHeader);
113                        if (((WsMessageHeader) messageHeader).isTextMode()) {
114                                headerHead = (1 << 8);
115                        } else {
116                                headerHead = (2 << 8);
117                        }
118                        continuationFrame = false;
119                } else {
120                        messageHeaders.push(messageHeader);
121                        if (messageHeader instanceof WsCloseFrame) {
122                                headerHead = (8 << 8);
123                        } else if (messageHeader instanceof WsPingFrame) {
124                                headerHead = (9 << 8);
125                        } else if (messageHeader instanceof WsPongFrame) {
126                                headerHead = (10 << 8);
127                        } else {
128                                throw new IllegalArgumentException(
129                                        "Invalid hessage header type");
130                        }
131                }
132                state = State.STARTING_FRAME;
133                bytesToSend = 2;
134        }
135
136        @Override
137        public Result encode(Buffer in, ByteBuffer out, boolean endOfInput) {
138                Result result = null;
139                while (out.remaining() > 0) {
140                        switch(state) {
141                        case STARTING_FRAME:
142                                prepareHeaderHead(in, endOfInput);
143                                // If called again without new message header...
144                                continuationFrame = true;
145                                state = State.WRITING_HEADER;
146                                // fall through
147                        case WRITING_HEADER:
148                                out.put((byte)(headerHead >> 8 * --bytesToSend));
149                                if (bytesToSend > 0) {
150                                        continue;
151                                }
152                                if (payloadBytes > 0) {
153                                        state = State.WRITING_LENGTH;
154                                        bytesToSend = payloadBytes;
155                                        continue;
156                                }
157                                // Length written
158                                result = nextAfterLength(endOfInput);
159                                break;
160                        case WRITING_LENGTH:
161                                out.put((byte)(payloadSize >> 8 * --bytesToSend));
162                                if (bytesToSend > 0) {
163                                        continue;
164                                }
165                                result = nextAfterLength(endOfInput);
166                                break;
167                        case WRITING_MASK:
168                                out.put(maskingKey[4 - (int)bytesToSend]);
169                                if (--bytesToSend > 0) {
170                                        continue;
171                                }
172                                result = nextAfterMask(endOfInput);
173                                break;
174                        case WRITING_PAYLOAD:
175                                int posBefore = out.position();
176                                outputPayload(in, out);
177                                bytesToSend -= (out.position() - posBefore);
178                                if (bytesToSend == 0) {
179                                        convData.clear();
180                                        return frameFinished(endOfInput);
181                                }
182                                return resultFactory().newResult(!out.hasRemaining(),
183                                                (messageHeaders.peek() instanceof WsMessageHeader) 
184                                                        && !in.hasRemaining(), false);
185                        }
186                        if (result != null) {
187                                return result;
188                        }
189                }
190                return resultFactory().newResult(true, false, false);
191        }
192
193        private void prepareHeaderHead(Buffer in, boolean endOfInput) {
194                WsFrameHeader hdr = messageHeaders.peek();
195                if (hdr instanceof WsMessageHeader) {
196                        if (continuationFrame) {
197                                headerHead = 0;
198                        }                       
199                        if (endOfInput) {
200                                headerHead |= 0x8000;
201                        }
202                        // Prepare payload
203                        if (in instanceof CharBuffer) {
204                                convData.clear();
205                                convTextData(in);
206                        } else {
207                                payloadSize = in.remaining();
208                        }
209                } else {
210                        // Control frame
211                        headerHead |= 0x8000;
212                        // Prepare payload
213                        if (hdr instanceof WsCloseFrame) {
214                                payloadSize = 0;
215                                ((WsCloseFrame)hdr).statusCode().ifPresent(code -> {
216                                        convData.clear();
217                                        try {
218                                                convData.write(code >> 8);
219                                                convData.write(code & 0xff);
220                                                payloadSize = 2;
221                                        } catch (IOException e) {
222                                                // Formally thrown, cannot happen
223                                        }
224                                });
225                                ((WsCloseFrame)hdr).reason().ifPresent(reason -> {
226                                                convTextData(CharBuffer.wrap(reason));
227                                });
228                        } else if (hdr instanceof WsDefaultControlFrame) {
229                                payloadSize = ((WsDefaultControlFrame)hdr)
230                                                .applicationData().map(ByteBuffer::remaining).orElse(0);
231                        }
232                }
233                
234                // Finally add mask bit
235                if (doMask) {
236                        headerHead |= 0x80;
237                        randoms.nextBytes(maskingKey);
238                }
239
240                // Code payload size
241                if (payloadSize <= 125) {
242                        headerHead |= payloadSize;
243                        payloadBytes = 0;
244                } else if (payloadSize < 0x10000) {
245                        headerHead |= 126;
246                        payloadBytes = 2;
247                } else {
248                        headerHead |= 127;
249                        payloadBytes = 8;
250                }
251        }
252
253        private void convTextData(Buffer in) {
254                convData.setOverflowBufferSize(
255                                (int) (in.remaining() * bytesPerCharUtf8));
256                try {
257                        OutputStreamWriter charWriter = new OutputStreamWriter(
258                                convData, "utf-8");
259                        if (in.hasArray()) {
260                                // more efficient than CharSequence
261                                charWriter.write(((CharBuffer) in).array(),
262                                        in.arrayOffset() + in.position(),
263                                        in.remaining());
264                        } else {
265                                charWriter.append((CharBuffer) in);
266                        }
267                        in.position(in.limit());
268                        charWriter.flush();
269                        payloadSize = convData.bytesWritten();
270                } catch (IOException e) {
271                        // Formally thrown, cannot happen
272                }
273        }
274        
275        private Result nextAfterLength(boolean endOfInput) {
276                if (doMask) {
277                        bytesToSend = 4;
278                        state = State.WRITING_MASK;
279                        return null;
280                }
281                return nextAfterMask(endOfInput);
282        }
283        
284        private Result nextAfterMask(boolean endOfInput) {
285                if (payloadSize == 0) {
286                        return frameFinished(endOfInput);
287                }
288                maskIndex = 0;
289                bytesToSend = payloadSize;
290                state = State.WRITING_PAYLOAD;
291                return null;
292        }
293        
294        private void outputPayload(Buffer in, ByteBuffer out) {
295                WsFrameHeader hdr = messageHeaders.peek();
296                if ((hdr instanceof WsMessageHeader) 
297                                && ((WsMessageHeader)hdr).isTextMode()
298                                || (hdr instanceof WsCloseFrame)) {
299                        // Data is in convData
300                        if (!doMask) {
301                                convData.assignBuffer(out);
302                                return;
303                        }
304                        // Retrieving as much as fit in out buffer for conversion
305                        in = ByteBuffer.allocate(out.remaining());
306                        convData.assignBuffer((ByteBuffer)in);
307                        in.flip();
308                } else {
309                        // Take data from in
310                        if (hdr instanceof WsDefaultControlFrame) {
311                                in = ((WsDefaultControlFrame)hdr)
312                                                .applicationData().orElse(Codec.EMPTY_IN);
313                        }
314                        if (!doMask) {
315                                ByteBufferUtils.putAsMuchAsPossible(out, (ByteBuffer) in);
316                                return;
317                        }
318                }
319                // Mask while writing
320                while (bytesToSend > 0
321                        && in.hasRemaining() && out.hasRemaining()) {
322                        out.put((byte) (((ByteBuffer) in)
323                                .get() ^ maskingKey[maskIndex]));
324                        maskIndex = (maskIndex + 1) % 4;
325                }
326        }

328        /**
329         * Results from {@link HttpEncoder} provide no additional
330         * information compared to {@link org.jdrupes.httpcodec.Codec.Result}. This
331         * class only provides a factory for creating concrete results.
332         */
333        public static class Result extends Codec.Result {
334        
335                protected Result(boolean overflow, boolean underflow,
336                        boolean closeConnection) {
337                        super(overflow, underflow, closeConnection);
338                }
339
340                /**
341                 * A factory for creating new Results.
342                 */
343                protected static class Factory extends Codec.Result.Factory {
344
345                        /**
346                         * Create new result.
347                         * 
348                         * @param overflow
349                         *            {@code true} if the data didn't fit in the out buffer
350                         * @param underflow
351                         *            {@code true} if more data is expected
352                         * @param closeConnection
353                         *            {@code true} if the connection should be closed
354                         * @return the result
355                         */
356                        public Result newResult(boolean overflow, boolean underflow,
357                                boolean closeConnection) {
358                                return new Result(overflow, underflow, closeConnection) {
359                                };
360                        }
361                }
362        }
