018package org.jdrupes.httpcodec;
019
020import java.nio.Buffer;
021import java.nio.ByteBuffer;
022import java.util.Optional;
023
024/**
025 * The general interface of a decoder.
026 * 
027 * @param <T> the type of message decoded by this decoder
028 * @param <R> the type of message that may be generated as response
029 * (see {@link Decoder.Result#getResponse()})
030 * 
031 * @author Michael N. Lipp
032 *
033 */
034public interface Decoder<T extends MessageHeader,
035        R extends MessageHeader> extends Codec {
036
037        /**
038         * Decodes the next chunk of data. This method will never leave
039         * remaining data in the `in` buffer unless a header has been
040         * decoded completely and/or the `out` buffer is full. In either case, 
041         * decoding will therefore continue when the method is invoked again
042         * with the same `in` buffer and another (or emptied) `out` buffer.
043         * It is never necessary (though possible) to add data to an existing 
044         * `in` buffer.
045         * 
046         * @param in
047         *            holds the data to be decoded
048         * @param out
049         *            gets the body data (if any) written to it
050         * @param endOfInput
051         *            {@code true} if there is no input left beyond the data
052         *            currently in the {@code in} buffer (indicates end of body or
053         *            no body at all)
054         * @return the result
055         * @throws ProtocolException
056         *             if the message violates the HTTP
057         */
058        public Result<R> decode(ByteBuffer in, Buffer out, boolean endOfInput)
059                throws ProtocolException;
060        
061        /**
062         * Returns the last message (header) received completely 
063         * if it exists.
064         * 
065         * @return the result
066         */
067        public Optional<T> getHeader();
068        
069        /**
070         * Factory method for result.
071         * 
072         * @param overflow
073         *            {@code true} if the data didn't fit in the out buffer
074         * @param underflow
075         *            {@code true} if more data is expected
076         * @param closeConnection
077         *            {@code true} if the connection should be closed
078         * @param headerCompleted {@code true} if the header has completely
079         * been decoded
080         * @param response a response to send due to an error
081         * @param responseOnly if the result includes a response 
082         * this flag indicates that no further processing besides 
083         * sending the response is required
084         * @return the result
085         */
086        default Result<R> newResult (boolean overflow, boolean underflow, 
087                        boolean closeConnection, boolean headerCompleted, 
088                        R response, boolean responseOnly) {
089                return new Result<R>(overflow, underflow, closeConnection,
090                                headerCompleted, response, responseOnly) {
091                };
092        }
093
094        /**
095         * Overrides the base interface's factory method in order to make
096         * it return the extended return type.
097         * 
098         * @param overflow
099         *            {@code true} if the data didn't fit in the out buffer
100         * @param underflow
101         *            {@code true} if more data is expected
102         * @param headerCompleted
103         *            indicates that the message header has been completed and
104         *            the message (without body) is available
105         */
106        Result<R> newResult (boolean overflow, boolean underflow, 
107                        boolean headerCompleted);
108        
109        /**
110         * The result from decoding. In addition to the common codec result, this
111         * includes the information wheteher a complete message header has been
112         * received and it can include a response that is to be sent back to the
113         * sender in order to fulfill the requirements of the protocol. As the
114         * decoder can (obviously) not sent back this response by itself, it is
115         * included in the result.
116         * <P>
117         * The class is declared abstract to promote the usage of the factory
118         * method.
119         *
120         * @param <R>
121         *            the type of the optionally generated response message
122         * @author Michael N. Lipp
123         */
124        public static abstract class Result<R extends MessageHeader> 
125                extends Codec.Result {
126
127                private boolean headerCompleted;
128                private Optional<R> response;
129                private boolean responseOnly;
130
131                /**
132                 * Creates a new result.
133                 * @param overflow {@code true} if the data didn't fit in the out buffer
134                 * @param underflow {@code true} if more data is expected
135                 * @param closeConnection
136                 *       {@code true} if the connection should be closed
137                 * @param headerCompleted {@code true} if the header has completely
138                 * been decoded
139                 * @param response a response to send due to an error
140                 * @param responseOnly if the result includes a response 
141                 * this flag indicates that no further processing besides 
142                 * sending the response is required
143                 */
144                protected Result(boolean overflow, boolean underflow, 
145                                boolean closeConnection, boolean headerCompleted, 
146                                R response, boolean responseOnly) {
147                        super(overflow, underflow, closeConnection);
148                        this.headerCompleted =  headerCompleted;
149                        this.response = Optional.ofNullable(response);
150                        this.responseOnly = responseOnly;
151                }
152
153                /**
154                 * Returns {@code true} if the message header has been decoded 
155                 * completely during the decoder invocation that returned this 
156                 * result and is now available. Note that not only the header may
157                 * heve been decoded, but that the output buffer may also
158                 * already contain decoded data.
159                 * 
160                 * @return the result
161                 */
162                public boolean isHeaderCompleted() {
163                        return headerCompleted;
164                }
165
166                /**
167                 * Returns the response if a response exists. A response in
168                 * the decoder result indicates that some information
169                 * must be signaled back to the sender.
170                 * 
171                 * @return the response
172                 */
173                public Optional<R> getResponse() {
174                        return response;
175                }
176
177                /**
178                 * If the result includes a response (see {@link #getResponse()})
179                 * and this method returns {@code true} then no
180                 * further processing of the received data is required. After sending
181                 * the response data, the decode method should be invoked 
182                 * again with the same parameters. 
183                 * 
184                 * @return the result
185                 */
186                public boolean isResponseOnly() {
187                        return responseOnly;
188                }
189
190                /* (non-Javadoc)
191                 * @see java.lang.Object#hashCode()
192                 */
193                @Override
194                public int hashCode() {
195                        final int prime = 31;
196                        int result = super.hashCode();
197                        result = prime * result + (headerCompleted ? 1231 : 1237);
198                        result = prime * result + (responseOnly ? 1231 : 1237);
199                        result = prime * result
200                                + ((response == null) ? 0 : response.hashCode());
201                        return result;
202                }
203
204                /* (non-Javadoc)
205                 * @see java.lang.Object#equals(java.lang.Object)
206                 */
207                @Override
208                public boolean equals(Object obj) {
209                        if (this == obj)
210                                return true;
211                        if (!super.equals(obj))
212                                return false;
213                        if (getClass() != obj.getClass())
214                                return false;
215                        Result<?> other = (Result<?>) obj;
216                        if (headerCompleted != other.headerCompleted)
217                                return false;
218                        if (responseOnly != other.responseOnly)
219                                return false;
220                        if (response == null) {
221                                if (other.response != null)
222                                        return false;
223                        } else if (!response.equals(other.response))
224                                return false;
225                        return true;
226                }
227
228                /* (non-Javadoc)
229                 * @see java.lang.Object#toString()
230                 */
231                @Override
232                public String toString() {
233                        StringBuilder builder = new StringBuilder();
234                        builder.append("Decoder.Result [overflow=");
235                        builder.append(isOverflow());
236                        builder.append(", underflow=");
237                        builder.append(isUnderflow());
238                        builder.append(", closeConnection=");
239                        builder.append(getCloseConnection());
240                        builder.append(", headerCompleted=");
241                        builder.append(headerCompleted);
242                        builder.append(", ");
243                        if (response != null) {
244                                builder.append("response=");
245                                builder.append(response);
246                                builder.append(", ");
247                        }
248                        builder.append("requestComleted=");
249                        builder.append(responseOnly);
250                        builder.append("]");
251                        return builder.toString();
252                }
253
254        }
255        
256}
