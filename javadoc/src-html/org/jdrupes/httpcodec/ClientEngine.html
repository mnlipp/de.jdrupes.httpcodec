019package org.jdrupes.httpcodec;
020
021import java.nio.Buffer;
022import java.nio.ByteBuffer;
023import java.util.Optional;
024
025import org.jdrupes.httpcodec.Codec.ProtocolSwitchResult;
026import org.jdrupes.httpcodec.protocols.http.client.HttpResponseDecoder;
027
028/**
029 * An engine that can be used as a client. It has an associated
030 * request encoder and a response decoder. Using a {@link ClientEngine}
031 * has two main advantages over using an encoder and decoder
032 * directly. It invokes the 
033 * {@link ResponseDecoder#decodeResponseTo(MessageHeader)} when
034 * appropriate and it replaces the encoder and decoder if the
035 * decoded result indicates a switch. The change takes place upon
036 * the next `encode` or `decode` invocation. The "old" encoders
037 * and decoders are therefore still available when the result of
038 * a decode invocation indicates a switch.
039 * 
040 * @param <Q> the message header type handled be the encoder (the request)
041 * @param <R> the message header type handled by the decoder (the response)
042 */
043public class ClientEngine<Q extends MessageHeader, 
044        R extends MessageHeader> extends Engine {
045
046        private Encoder<?> requestEncoder;
047        private ResponseDecoder<?, ?> responseDecoder;
048        private Encoder<?> newRequestEncoder;
049        private ResponseDecoder<?, ?> newResponseDecoder;
050        
051        /**
052         * Creates a new instance.
053         * 
054         * @param requestEncoder the encoder for the request
055         * @param responseDecoder the decoder for the response
056         */
057        public ClientEngine(Encoder<Q> requestEncoder, 
058                        ResponseDecoder<R, Q> responseDecoder) {
059                this.requestEncoder = requestEncoder;
060                this.responseDecoder = responseDecoder;
061        }
062        
063        /**
064         * @return the requestEncoder
065         */
066        @SuppressWarnings("unchecked")
067        public Encoder<Q> requestEncoder() {
068                return (Encoder<Q>)requestEncoder;
069        }
070        
071        /**
072         * @return the responseDecoder
073         */
074        @SuppressWarnings("unchecked")
075        public ResponseDecoder<R,Q> responseDecoder() {
076                return (ResponseDecoder<R,Q>)responseDecoder;
077        }
078
079        /**
080         * Convenience method to invoke the encoder's encode method.
081         * 
082         * @param out the buffer to use for the result
083         * @return the result
084         */
085        public Codec.Result encode(ByteBuffer out) {
086                return requestEncoder.encode(out);
087        }
088
089        /**
090         * Convenience method to invoke the encoder's encode method.
091         * 
092         * @param in the buffer with the data to encode
093         * @param out the buffer to use for the result
094         * @param endOfInput {@code true} if end of input
095         * @return the result
096         */
097        public Codec.Result encode(Buffer in, ByteBuffer out, boolean endOfInput) {
098                return requestEncoder.encode(in, out, endOfInput);
099        }
100
101        /**
102         * Convenience method to invoke the encoder's encode method.
103         * 
104         * @param messageHeader the message header
105         */
106        @SuppressWarnings("unchecked")
107        public void encode(Q messageHeader) {
108                if (newRequestEncoder != null) {
109                        requestEncoder = newRequestEncoder;
110                        newRequestEncoder = null;
111                }
112                ((Encoder<Q>)requestEncoder).encode(messageHeader);
113        }
114
115        /**
116         * Convenience method to invoke the decoder's decode method.
117         * 
118         * @param request the request
119         */
120        @SuppressWarnings("unchecked")
121        public void decodeResponseTo(Q request) {
122                ((ResponseDecoder<R,Q>)responseDecoder).decodeResponseTo(request);
123        }
124
125        /**
126         * Convenience method to invoke the decoder's decode method.
127         * 
128         * @param in the buffer with the data to decode
129         * @param out the buffer to use for the result
130         * @param endOfInput {@code true} if end of input
131         * @return the result
132         * @throws ProtocolException if the input violates the protocol
133         * @see HttpResponseDecoder#decode(java.nio.ByteBuffer, java.nio.Buffer, boolean)
134         */
135        @SuppressWarnings("unchecked")
136        public Decoder.Result<Q> decode(
137                ByteBuffer in, Buffer out, boolean endOfInput)
138                throws ProtocolException {
139                if (newResponseDecoder != null) {
140                        responseDecoder = newResponseDecoder;
141                        newResponseDecoder = null;
142                }
143                if (responseDecoder.isAwaitingMessage()) {
144                        ((Encoder<Q>)requestEncoder).header().ifPresent(request ->  
145                                ((ResponseDecoder<R,Q>)responseDecoder).decodeResponseTo(request));
146                }
147                Decoder.Result<Q> result 
148                        = (Decoder.Result<Q>)responseDecoder.decode(in, out, endOfInput);
149                if (result instanceof ProtocolSwitchResult) {
150                        ProtocolSwitchResult res = (ProtocolSwitchResult)result;
151                        if (res.newProtocol() != null) {
152                                setSwitchedTo(res.newProtocol());
153                                newResponseDecoder = (ResponseDecoder<?,?>)res.newDecoder();
154                                newRequestEncoder = res.newEncoder();
155                        }
156                }
157                return result;
158        }
159        
160        /**
161         * Returns the last encoded request.
162         * 
163         * @return the request
164         */
165        @SuppressWarnings("unchecked")
166        public Optional<Q> currentRequest() {
167                return (Optional<Q>)requestEncoder.header();
168        }
169        
170}
