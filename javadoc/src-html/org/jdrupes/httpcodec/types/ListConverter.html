019package org.jdrupes.httpcodec.types;
020
021import java.text.ParseException;
022import java.util.List;
023import java.util.function.Supplier;
024
025import org.jdrupes.httpcodec.util.ListItemizer;
026
027/**
028 * Used by by classes that convert field values which consist
029 * of lists of values.
030 * 
031 * This class uses an extra type parameter `L` instead of
032 * simply using `List<I>` to allow the convertion to special
033 * lists, i.e. types derived from `List<T>` with additional
034 * methods.
035 * 
036 * @param <L> the type of the list
037 * @param <I> the type of the items
038 */
039public class ListConverter<L extends List<I>, I> 
040        implements Converter<L> {
041
042        private Supplier<L> listCreator;
043        private Converter<I> itemConverter;
044        // Used by default in RFC 7230, see section 7.
045        private String delimiters = ",";
046        private boolean separateItems = false;
047        
048        /**
049         * Creates a new list converter with the given converter for the items
050         * and comma as item delimiter.
051         * 
052         * @param listCreator a function that creates a new empty list
053         * @param itemConverter the converter for the items
054         * @see "[ABNF List Extension](https://tools.ietf.org/html/rfc7230#section-7)"
055         */
056        public ListConverter(Supplier<L> listCreator, Converter<I> itemConverter) {
057                this(listCreator, itemConverter, ",");
058        }
059
060        /**
061         * Creates a new list converter with the given converter for the items
062         * and the given delimiters for the items. Any character in `delimiters`
063         * is considered a delimiter when parsing the string. The first
064         * character will be used when items are joined in a
065         * textual representation.
066         * 
067         * @param listCreator a function that creates a new empty list
068         * @param itemConverter the converter for the items
069         * @param delimiters the delimiters
070         */
071        public ListConverter(Supplier<L> listCreator, Converter<I> itemConverter, 
072                        String delimiters) {
073                this.listCreator = listCreator;
074                this.itemConverter = itemConverter;
075                this.delimiters = delimiters;
076        }
077
078        /**
079         * Create a new converter 
080         * like {@link #ListConverter(Supplier, Converter, String)}. In addition,
081         * the parameter `splitItems` can be set to indicate that the
082         * values from the list should be converted to individual
083         * header fields.
084         * 
085         * @param listCreator a function that creates a new empty list
086         * @param itemConverter the converter for the items
087         * @param delimiters the delimiters
088         * @param separateItems indicates that each value should be represented
089         * by a header field of its own in a message header
090         */
091        public ListConverter(Supplier<L> listCreator, Converter<I> itemConverter, 
092                        String delimiters, boolean separateItems) {
093                this.listCreator = listCreator;
094                this.itemConverter = itemConverter;
095                this.delimiters = delimiters;
096                this.separateItems = separateItems;
097        }
098
099        /**
100         * @return the separateItems
101         */
102        public boolean isSeparateItems() {
103                return separateItems;
104        }
105
106        /**
107         * Returns the item converter.
108         * 
109         * @return the itemConverter
110         */
111        public Converter<I> getItemConverter() {
112                return itemConverter;
113        }
114
115        /* (non-Javadoc)
116         * @see org.jdrupes.httpcodec.protocols.http.fields.Converter#asFieldValue(java.lang.Object)
117         */
118        @Override
119        public String asFieldValue(L value) {
120                if (value.size() == 0) {
121                        throw new IllegalStateException(
122                                "Field with list value may not be empty.");
123                }
124                boolean first = true;
125                StringBuilder result = new StringBuilder();
126                for (I e: value) {
127                        if (first) {
128                                first = false;
129                        } else {
130                                result.append(delimiters.charAt(0));
131                                result.append(' ');
132                        }
133                        result.append(itemConverter.asFieldValue(e));
134                }
135                return result.toString();
136        }
137
138        
139        /* (non-Javadoc)
140         * @see Converter#fromFieldValue(java.lang.String)
141         */
142        @Override
143        public L fromFieldValue(String text) throws ParseException {
144                L result = listCreator.get();
145                ListItemizer itemizer = new ListItemizer(text, delimiters);
146                while (true) {
147                        String nextRepr = itemizer.nextItem();
148                        if (nextRepr == null) {
149                                break;
150                        }
151                        result.add(itemConverter.fromFieldValue(nextRepr));
152                }
153                return result;
154        }
155        
156}
