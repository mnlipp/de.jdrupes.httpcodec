019package org.jdrupes.httpcodec.types;
020
021import java.text.ParseException;
022import java.util.Collections;
023import java.util.HashMap;
024import java.util.Map;
025
026public class MediaType extends ParameterizedValue<MediaType.MediaMainType> {
027
028        public static final MediaType ALL_MEDIA = new MediaType("*", "*");
029        
030        /**
031         * Create a new object with the given type and subtype.
032         * 
033         * @param type the top-level type
034         * @param subtype the subtype
035         */
036        public MediaType(String type, String subtype) {
037                super(new MediaMainType(type, subtype), Collections.emptyMap());
038        }
039        
040        /**
041         * Create a new object with the given type, subtype and parameters.
042         * 
043         * @param type the top-level type
044         * @param subtype the subtype
045         * @param parameters the parameters
046         */
047        public MediaType(String type, String subtype, 
048                        Map<String, String> parameters) {
049                super(new MediaMainType(type, subtype), parameters);
050        }
051
052        /**
053         * Create a new object with the given type and parameters.
054         * 
055         * @param type the type
056         * @param parameters the parameters
057         */
058        public MediaType(MediaMainType type, Map<String, String> parameters) {
059                super(type, parameters);
060        }
061
062        /**
063         * For builder.
064         */
065        private MediaType() {
066                super(ALL_MEDIA.getValue(), Collections.emptyMap());
067        }
068        
069        /**
070         * Creates a new instance with values obtained from parsing
071         * the given text.
072         * 
073         * @param text the text
074         * @return the mime type
075         * @throws ParseException if the text is not well-formed
076         */
077        public static MediaType fromString(String text) 
078                        throws ParseException {
079                return Converters.MEDIA_TYPE_CONVERTER.fromFieldValue(text);
080        }
081        
082        /**
083         * Returns the top-level type.
084         * 
085         * @return the type
086         */
087        public String getTopLevelType() {
088                return getValue().topLevelType;
089        }
090        
091        /**
092         * Returns the subtype
093         * 
094         * @return the subtype
095         */
096        public String getSubtype() {
097                return getValue().subtype;
098        }
099
100        /**
101         * Creates a new builder for a media type.
102         * 
103         * @return the builder
104         */
105        @SuppressWarnings("unchecked")
106        public static Builder builder() {
107                return new Builder();
108        }
109        
110        /* (non-Javadoc)
111         * @see java.lang.Object#toString()
112         */
113        @Override
114        public String toString() {
115                return Converters.MEDIA_TYPE_CONVERTER.asFieldValue(this);
116        }
117
118
119        /**
120         * A builder for the (immutable) parameterized type.
121         */
122        public static class Builder 
123                extends ParameterizedValue.Builder<MediaMainType> {
124
125                private Builder() {
126                        super(new MediaType("*", "*", new HashMap<>()));
127                }
128                
129                @Override
130                public MediaType build() {
131                        return (MediaType)super.build();
132                }
133
134                public Builder from(MediaType existing) {
135                        return (Builder)super.from(existing);
136                }
137
138                @Override
139                public Builder setValue(MediaMainType value) {
140                        return (Builder)super.setValue(value);
141                }
142
143                @Override
144                public Builder setParameter(String name, String value) {
145                        return (Builder)super.setParameter(name, value);
146                }
147
148                @Override
149                public Builder remove(String name) {
150                        return (Builder)super.remove(name);
151                }
152
153                /**
154                 * Sets the media type.
155                 * 
156                 * @param topLevelType the top level type
157                 * @param subtype the subtype
158                 * @return the builder for easy chaining
159                 */
160                public Builder setType(String topLevelType, String subtype) {
161                        setValue(new MediaMainType(topLevelType, subtype));
162                        return this;
163                }
164        }
165        
166        /**
167         * Represents a media type, including a media range (type or subtype
168         * equals "`*`").
169         */
170        public static class MediaMainType {
171                
172                private String topLevelType;
173                private String subtype;
174                
175                /**
176                 * Create a new object with the given type and subtype.
177                 * 
178                 * @param type the type
179                 * @param subtype the subtype
180                 */
181                public MediaMainType(String type, String subtype) {
182                        super();
183                        this.topLevelType = type.toLowerCase();
184                        this.subtype = subtype.toLowerCase();
185                }
186                
187                /**
188                 * Create a main type from its textual representation. 
189                 * 
190                 * @param text the textual representation
191                 * @return the result
192                 * @throws ParseException if the text is ill-formed
193                 */
194                public static MediaMainType fromString(String text) 
195                                throws ParseException {
196                        int sepPos = text.indexOf('/');
197                        if (sepPos <= 0 || sepPos > text.length() - 2) {
198                                throw new ParseException(
199                                                "Format not \"type/subtype\": " + text, 0);
200                        }
201                        return new MediaMainType(text.substring(0, sepPos),
202                                        text.substring(sepPos + 1));
203                        
204                }
205                
206                /**
207                 * Returns the top-level type.
208                 * 
209                 * @return the type
210                 */
211                public String getTopLevelType() {
212                        return topLevelType;
213                }
214                
215                /**
216                 * Returns the subtype
217                 * 
218                 * @return the subtype
219                 */
220                public String getSubtype() {
221                        return subtype;
222                }
223
224                /* (non-Javadoc)
225                 * @see java.lang.Object#toString()
226                 */
227                @Override
228                public String toString() {
229                        return topLevelType + "/" + subtype;
230                }
231
232                /* (non-Javadoc)
233                 * @see java.lang.Object#hashCode()
234                 */
235                @Override
236                public int hashCode() {
237                        final int prime = 31;
238                        int result = 1;
239                        result = prime * result
240                                + ((subtype == null) ? 0 : subtype.hashCode());
241                        result = prime * result + ((topLevelType == null) ? 0 : topLevelType.hashCode());
242                        return result;
243                }
244
245                /* (non-Javadoc)
246                 * @see java.lang.Object#equals(java.lang.Object)
247                 */
248                @Override
249                public boolean equals(Object obj) {
250                        if (this == obj) {
251                                return true;
252                        }
253                        if (obj == null) {
254                                return false;
255                        }
256                        if (getClass() != obj.getClass()) {
257                                return false;
258                        }
259                        MediaType.MediaMainType other = (MediaType.MediaMainType) obj;
260                        if (subtype == null) {
261                                if (other.subtype != null) {
262                                        return false;
263                                }
264                        } else if (!subtype.equals(other.subtype)) {
265                                return false;
266                        }
267                        if (topLevelType == null) {
268                                if (other.topLevelType != null) {
269                                        return false;
270                                }
271                        } else if (!topLevelType.equals(other.topLevelType)) {
272                                return false;
273                        }
274                        return true;
275                }
276        }
