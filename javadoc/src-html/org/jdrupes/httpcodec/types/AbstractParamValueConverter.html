019package org.jdrupes.httpcodec.types;
020
021import java.text.ParseException;
022import java.util.HashMap;
023import java.util.Map;
024import java.util.Map.Entry;
025import java.util.function.BiFunction;
026
027import org.jdrupes.httpcodec.util.ListItemizer;
028
029/**
030 * A base class for converters for values with parameters. 
031 * Converts field values such as `value; param1=value1; param2=value2`.
032 * 
033 * This class is actually a {@link Converter}, but does not implement
034 * `Converter<ParameterizedValue<T>>` to avoid conflicts with
035 * derived types that want to implement `Converter` with a different
036 * type paramter `T`. {@link ParameterizedValueConverter} provides
037 * a concrete converter for `Converter<ParameterizedValue<T>>`.
038 * 
039 * @param <T> the type of the value
040 */
041public class AbstractParamValueConverter<T> {
042
043        private Converter<T> valueConverter;
044        private Converter<String> paramValueConverter;
045        boolean quoteIfNecessary = false;
046
047        /**
048         * Creates a new converter by extending the given value converter
049         * with functionality for handling the parameters. Parameter
050         * values are used literally (no quoting).
051         * 
052         * @param valueConverter the converter for a value (without parameters)
053         */
054        public AbstractParamValueConverter(Converter<T> valueConverter) {
055                this(valueConverter, Converters.UNQUOTED_STRING_CONVERTER);
056        }
057
058        /**
059         * Creates a new converter by extending the given value converter
060         * with functionality for handling the parameters.
061         * 
062         * @param valueConverter the converter for a value (without parameters)
063         * @param paramValueConverter the converter for parameterValues
064         */
065        public AbstractParamValueConverter(     Converter<T> valueConverter, 
066                        Converter<String> paramValueConverter) {
067                this.valueConverter = valueConverter;
068                this.paramValueConverter = paramValueConverter;
069        }
070
071        public String asFieldValue(ParameterizedValue<T> value) {
072                StringBuilder result = new StringBuilder();
073                result.append(valueConverter.asFieldValue(value.getValue()));
074                for (Entry<String, String> e: value.getParameters().entrySet()) {
075                        result.append("; ");
076                        result.append(e.getKey());
077                        result.append('=');
078                        result.append(paramValueConverter.asFieldValue(e.getValue()));
079                }
080                return result.toString();
081        }
082
083        /**
084         * Default implementation of {@link #fromFieldValue(String, BiFunction)}.
085         * 
086         * @param text the textual representation
087         * @return an instance created with 
088         * {@link ParameterizedValue#ParameterizedValue(Object, Map)}
089         * @throws ParseException
090         */
091        public ParameterizedValue<T> fromFieldValue(String text)
092                        throws ParseException {
093                return fromFieldValue(text, ParameterizedValue<T>::new);
094        }
095        
096        /**
097         * Used by classes that inherit from {@link ParameterizedValue}
098         * to return a instance of the derived type as result of the
099         * conversion.
100         * 
101         * @param text the textual representation
102         * @param resultConstructor a method that creates the result
103         * from an instance of the type and a map of parameters.
104         * @return the result
105         * @throws ParseException if the text is ill-formed
106         */
107        public <R extends ParameterizedValue<T>> R fromFieldValue(String text, 
108                        BiFunction<T, Map<String,String>, R> resultConstructor)
109                        throws ParseException {
110                ListItemizer li = new ListItemizer(text, ";");
111                String valueRepr = li.nextItem();
112                if (valueRepr == null) {
113                        throw new ParseException("Value may not be empty", 0);
114                }
115                T value = valueConverter.fromFieldValue(valueRepr);
116                Map<String,String> params = new HashMap<>();
117                while (true) {
118                        String param = li.nextItem();
119                        if (param == null) {
120                                break;
121                        }
122                        ListItemizer pi = new ListItemizer(param, "=");
123                        String paramKey = pi.nextItem().trim().toLowerCase();
124                        if (paramKey == null) {
125                                throw new ParseException("parameter may not be empty", 0);
126                        }
127                        String paramValue = pi.nextItem();
128                        if (paramValue != null) {
129                                paramValue = paramValueConverter.fromFieldValue(paramValue);
130                        }
131                        params.put(paramKey, paramValue);
132                }
133                return resultConstructor.apply(value, params);
134        }
135}
