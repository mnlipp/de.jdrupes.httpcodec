206        /**
207         * If the value is double quoted, remove the quotes and escape
208         * characters.
209         * 
210         * @param value the value to unquote
211         * @return the unquoted value
212         * @throws ParseException if the input violates the field format
213         * @see "[Field value components](https://tools.ietf.org/html/rfc7230#section-3.2.6)"
214         */
215        public static String unquote(String value) throws ParseException {
216                if (value.length() == 0 || value.charAt(0) != '\"') {
217                        return value;
218                }
219                int startPosition = 1;
220                int position = 1;
221                try {
222                        StringBuilder result = new StringBuilder();
223                        while (true) {
224                                char ch = value.charAt(position);
225                                switch (ch) {
226                                case '\\':
227                                        result.append(value.substring(startPosition, position));
228                                        position += 1;
229                                        result.append(value.charAt(position));
230                                        position += 1;
231                                        startPosition = position;
232                                        continue;
233                                case '\"':
234                                        if (position != value.length() - 1) {
235                                                throw new ParseException(value, position);
236                                        }
237                                        result.append(value.substring(startPosition, position));
238                                        return result.toString();
239                                default:
240                                        position += 1;
241                                        continue;
242                                }
243                        }
244                } catch (IndexOutOfBoundsException e) {
245                        throw new ParseException(value, position);
246                }
247        }
248
249        /**
250         * Returns the given string as double quoted string if necessary.
251         * 
252         * @param value the value to quote if necessary
253         * @return the result
254         * @see "[Field value components](https://tools.ietf.org/html/rfc7230#section-3.2.6)"
255         */
256        public static String quoteIfNecessary(String value) {
257                StringBuilder result = new StringBuilder();
258                int position = 0;
259                boolean needsQuoting = false;
260                result.append('"');
261                while (position < value.length()) {
262                        char ch = value.charAt(position++);
263                        if (!needsQuoting && HttpConstants.TOKEN_CHARS.indexOf(ch) < 0) {
264                                needsQuoting = true;
265                        }
266                        switch(ch) {
267                        case '"':
268                                // fall through
269                        case '\\':
270                                result.append('\\');
271                                // fall through
272                        default:
273                                result.append(ch);
274                                break;
275                        }
276                }
277                result.append('\"');
278                if (needsQuoting) {
279                        return result.toString();
280                }
281                return value;
282        }
