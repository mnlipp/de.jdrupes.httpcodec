018package org.jdrupes.httpcodec.util;
019
020import java.io.IOException;
021import java.io.OutputStream;
022import java.nio.ByteBuffer;
023import java.util.ArrayDeque;
024import java.util.Queue;
025
026/**
027 * An {@link OutputStream} that is backed by a {@link ByteBuffer} assigned to
028 * the stream. If the buffer becomes full, one or more buffers are allocated as
029 * intermediate storage. Their content is copied to the next assigned buffer(s).
030 * <p>
031 * While writing to this stream, {@link #remaining()} should be checked
032 * regularly and the production of data should be suspended if possible when no
033 * more space is left to avoid the usage of intermediate storage.
034 * 
035 * @author Michael N. Lipp
036 *
037 */
038public class ByteBufferOutputStream extends OutputStream {
039
040        private ByteBuffer assignedBuffer = null;
041        private Queue<ByteBuffer> overflows = new ArrayDeque<>();
042        private ByteBuffer current = null;
043        private int overflowBufferSize = 0;
044        private long bytesWritten = 0;
045
046        /**
047         * Creates a new instance with an unset overflow buffer size.
048         */
049        public ByteBufferOutputStream() {
050                super();
051        }
052
053        /**
054         * Creates a new instance with the given overflow buffer size.
055         * 
056         * @param overflowBufferSize
057         *            the overflow buffer size to use
058         */
059        public ByteBufferOutputStream(int overflowBufferSize) {
060                super();
061                this.overflowBufferSize = overflowBufferSize;
062        }
063
064        /**
065         * Returns the size of the buffers that will be allocated as overflow
066         * buffers.
067         *
068         * @return the allocation size for the overflow buffers
069         */
070        public int getOverflowBufferSize() {
071                return overflowBufferSize;
072        }
073
074        /**
075         * The size of the buffers that are allocated if the assigned buffer
076         * overflows. If not set, buffers are allocated with one fourth of the size
077         * of the assigned buffer or 4096 if no buffer has been assigned yet.
078         * 
079         * @param overflowBufferSize
080         *            the size
081         */
082        public void setOverflowBufferSize(int overflowBufferSize) {
083                this.overflowBufferSize = overflowBufferSize;
084        }
085
086        /**
087         * Clear any buffered data and prepares the buffer for reuse.
088         */
089        public void clear() {
090                assignedBuffer = null;
091                current = null;
092                bytesWritten = 0;
093                overflows.clear();
094        }
095
096        /**
097         * Assign a new buffer to this output stream. If the previously used buffer
098         * had become full and intermediate storage was allocated, the data from the
099         * intermediate storage is copied to the new buffer first. Then, the new
100         * buffer is used for all subsequent write operations.
101         * 
102         * @param buffer
103         *            the buffer to use
104         */
105        public void assignBuffer(ByteBuffer buffer) {
106                assignedBuffer = buffer;
107                // Move any overflow to the new buffer
108                while (!overflows.isEmpty()) {
109                        ByteBuffer head = overflows.peek();
110                        // Do a "flip with position to mark"
111                        int writePos = head.position(); // Save position
112                        head.reset();
113                        head.limit(writePos);
114                        if (!ByteBufferUtils.putAsMuchAsPossible(assignedBuffer, head)) {
115                                // Cannot transfer everything, done what's possible
116                                head.mark(); // new position for next put
117                                head.limit(head.capacity());
118                                head.position(writePos);
119                                return;
120                        }
121                        overflows.remove();
122                }
123                current = assignedBuffer;
124        }
125
126        private void allocateOverflowBuffer() {
127                current = ByteBuffer.allocate(
128                        overflowBufferSize != 0
129                        ? overflowBufferSize
130                        : Math.max(4096, (assignedBuffer == null
131                                          ? 0 : assignedBuffer.capacity() / 4)));
132                current.mark();
133                overflows.add(current);
134        }
135
136        /*
137         * (non-Javadoc)
138         * 
139         * @see java.io.OutputStream#write(int)
140         */
141        @Override
142        public void write(int b) throws IOException {
143                if (current == null || current.remaining() == 0) {
144                        allocateOverflowBuffer();
145                }
146                current.put((byte) b);
147                bytesWritten += 1;
148        }
149
150        /*
151         * (non-Javadoc)
152         * 
153         * @see java.io.OutputStream#write(byte[], int, int)
154         */
155        @Override
156        public void write(byte[] b, int offset, int length) throws IOException {
157                if (current == null) {
158                        allocateOverflowBuffer();
159                }
160                bytesWritten += length;
161                while (true) {
162                        if (current.remaining() >= length) {
163                                current.put(b, offset, length);
164                                return;
165                        }
166                        if (current.remaining() > 0) {
167                                int processed = current.remaining();
168                                current.put(b, offset, processed);
169                                offset += processed;
170                                length -= processed;
171                        }
172                        allocateOverflowBuffer();
173                }
174        }
175
176        /**
177         * Copies the data from the given buffer to this output stream.
178         * 
179         * @param b the buffer
180         */
181        public void write(ByteBuffer b) {
182                if (current == null) {
183                        allocateOverflowBuffer();
184                }
185                bytesWritten += b.remaining();
186                while (true) {
187                        if (ByteBufferUtils.putAsMuchAsPossible(current, b)) {
188                                return;
189                        }
190                        allocateOverflowBuffer();
191                }
192        }
193
194        /**
195         * Copies length bytes from the given buffer to this output stream.
196         * 
197         * @param b the buffer
198         * @param length the number of bytes to copy
199         */
200        public void write(ByteBuffer b, int length) {
201                if (b.remaining() <= length) {
202                        write(b);
203                        return;
204                }
205                int savedLimit = b.limit();
206                b.limit(b.position() + length);
207                write(b);
208                b.limit(savedLimit);
209        }
210
211        /**
212         * Returns the number of bytes remaining in the assigned buffer. A negative
213         * value indicates that the assigned buffer is full and one or more overflow
214         * buffers are being used. The absolute value of the negative number is
215         * the number of bytes in the overflow buffer(s).
216         * 
217         * @return the bytes remaining or buffered (negative value)
218         */
219        public long remaining() {
220                if (overflows.isEmpty()) {
221                        if (assignedBuffer == null) {
222                                return 0;
223                        }
224                        return assignedBuffer.remaining();
225                }
226                long sum = 0;
227                for (ByteBuffer b : overflows) {
228                        int curPos = b.position(); // Save position
229                        b.reset();
230                        sum += curPos - b.position();
231                        b.position(curPos);
232                }
233                return -sum;
234        }
235
236        /**
237         * Does not have any effect. May be called for consistent usage of the
238         * output stream.
239         * 
240         * @throws IOException
241         *             if there is still data in intermediate storage
242         * @see java.io.OutputStream#close()
243         */
244        @Override
245        public void close() throws IOException {
246                // TODO Auto-generated method stub
247                super.close();
248        }
249
250        /**
251         *      The number of bytes written to this output stream.
252         * 
253         * @return the number of bytes written
254         */
255        public long bytesWritten() {
256                return bytesWritten;
257        }
258}
