018package org.jdrupes.httpcodec.util;
019
020import java.nio.ByteBuffer;
021import java.nio.CharBuffer;
022import java.nio.charset.CharacterCodingException;
023import java.nio.charset.Charset;
024import java.nio.charset.CharsetDecoder;
025import java.nio.charset.CoderResult;
026import java.nio.charset.CodingErrorAction;
027
028/**
029 * Provides a wrapper that makes a {@link CharsetDecoder} behave like an
030 * optimized {@code CharsetDecoder}.
031 * <P>
032 * The default implementations of the {@link CharsetDecoder}s in the JRE leave
033 * data in the input buffer if there isn't sufficient data to decode the next
034 * character. The invoker is expected to add more data to the input buffer in
035 * response to the underflow result. This processing model can be used if you
036 * have a single buffer that you fill using some source and drain using the
037 * decoder.
038 * <P>
039 * If you use a model where one buffer is drained by the decoder while in
040 * parallel another buffer is being filled by the source, this processing model
041 * induces some difficulties, because an unknown number of bytes would have to
042 * be copied over from the newly filled buffer to the previously decoded buffer.
043 * This class provides what the documentation of
044 * {@link CharsetDecoder#decodeLoop} calls an "optimized implementation". It
045 * always drains the input buffer completely, caching any bytes that cannot be
046 * decoded (yet) because they don't encode a complete character. When the
047 * {@link #decode(ByteBuffer, CharBuffer, boolean)} method is invoked for the
048 * next time, as many bytes as needed to decode a complete character will be
049 * added to the cached bytes and the now available character is appended to the
050 * output. Then, the rest of the input is bulk processed as usual.
051 * 
052 * @author Michael N. Lipp
053 */
054public class OptimizedCharsetDecoder {
055
056        private CharsetDecoder backing;
057        private ByteBuffer pending;
058
059        /**
060         * Creates a new instance.
061         * 
062         * @param backing the backing charset decoder
063         */
064        public OptimizedCharsetDecoder(CharsetDecoder backing) {
065                super();
066                this.backing = backing;
067                pending = ByteBuffer.allocate(32);
068        }
069
070        /**
071         * @return the result
072         * @see java.nio.charset.CharsetDecoder#averageCharsPerByte()
073         */
074        public final float averageCharsPerByte() {
075                return backing.averageCharsPerByte();
076        }
077
078        /**
079         * @return the result
080         * @see java.nio.charset.CharsetDecoder#charset()
081         */
082        public final Charset charset() {
083                return backing.charset();
084        }
085
086        /**
087         * @param in the data to decode
088         * @param out the decoded data
089         * @param endOfInput {@code true} if the completes the input
090         * @return the result
091         * @see java.nio.charset.CharsetDecoder#decode(java.nio.ByteBuffer,
092         *      java.nio.CharBuffer, boolean)
093         */
094        public final CoderResult decode(ByteBuffer in, CharBuffer out,
095                boolean endOfInput) {
096                CoderResult result;
097                if (pending.position() > 0) {
098                        if (!out.hasRemaining()) {
099                                // There must be space for "the one" decoded character
100                                return CoderResult.OVERFLOW;
101                        }
102                        while (true) {
103                                if (!in.hasRemaining()) {
104                                        if (!endOfInput) {
105                                                return CoderResult.UNDERFLOW;
106                                        }
107                                        return backing.decode(pending, out, endOfInput);
108                                }
109                                pending.put(in.get());
110                                pending.flip();
111                                result = backing.decode(pending, out, endOfInput);
112                                pending.compact();
113                                if (result.isOverflow() || result.isMalformed()) {
114                                        return result;
115                                }
116                                if (pending.position() > 0) {
117                                        continue;
118                                }
119                                break;
120                        }
121                }
122                // Handle rest of input
123                result = backing.decode(in, out, endOfInput);
124                if (result.isUnderflow() && in.hasRemaining()) {
125                        byte[] remaining = new byte[in.remaining()];
126                        in.get(remaining);
127                        pending.put(remaining);
128                }
129                return result;
130        }
131
132        /**
133         * @param in the data
134         * @return the result
135         * @throws CharacterCodingException if an error occurred
136         * @see java.nio.charset.CharsetDecoder#decode(java.nio.ByteBuffer)
137         */
138        public final CharBuffer decode(ByteBuffer in)
139                throws CharacterCodingException {
140                return backing.decode(in);
141        }
142
143        /**
144         * @return the result
145         * @see java.nio.charset.CharsetDecoder#detectedCharset()
146         */
147        public Charset detectedCharset() {
148                return backing.detectedCharset();
149        }
150
151        /**
152         * @param out the decoded data
153         * @return the result
154         * @see java.nio.charset.CharsetDecoder#flush(java.nio.CharBuffer)
155         */
156        public final CoderResult flush(CharBuffer out) {
157                if (pending.position() > 0) {
158                        // Shouldn't happen, hard to know what to do...
159                        pending.clear();
160                }
161                return backing.flush(out);
162        }
163
164        /**
165         * @return the result
166         * @see java.nio.charset.CharsetDecoder#isAutoDetecting()
167         */
168        public boolean isAutoDetecting() {
169                return backing.isAutoDetecting();
170        }
171
172        /**
173         * @return the result
174         * @see java.nio.charset.CharsetDecoder#isCharsetDetected()
175         */
176        public boolean isCharsetDetected() {
177                return backing.isCharsetDetected();
178        }
179
180        /**
181         * @return the result
182         * @see java.nio.charset.CharsetDecoder#malformedInputAction()
183         */
184        public CodingErrorAction malformedInputAction() {
185                return backing.malformedInputAction();
186        }
187
188        /**
189         * @return the result
190         * @see java.nio.charset.CharsetDecoder#maxCharsPerByte()
191         */
192        public final float maxCharsPerByte() {
193                return backing.maxCharsPerByte();
194        }
195
196        /**
197         * @param newAction the action
198         * @return the result
199         * @see java.nio.charset.CharsetDecoder#onMalformedInput
200         *      (java.nio.charset.CodingErrorAction)
201         */
202        public final CharsetDecoder onMalformedInput(CodingErrorAction newAction) {
203                return backing.onMalformedInput(newAction);
204        }
205
206        /**
207         * @param newAction the action
208         * @return the result
209         * @see java.nio.charset.CharsetDecoder#onUnmappableCharacter
210         *      (java.nio.charset.CodingErrorAction)
211         */
212        public final CharsetDecoder onUnmappableCharacter(
213                CodingErrorAction newAction) {
214                return backing.onUnmappableCharacter(newAction);
215        }
216
217        /**
218         * @param newReplacement the replacement
219         * @return the result
220         * @see java.nio.charset.CharsetDecoder#replaceWith(java.lang.String)
221         */
222        public final CharsetDecoder replaceWith(String newReplacement) {
223                return backing.replaceWith(newReplacement);
224        }
225
226        /**
227         * @return the result
228         * @see java.nio.charset.CharsetDecoder#replacement()
229         */
230        public final String replacement() {
231                return backing.replacement();
232        }
233
234        /**
235         * @return the result
236         * @see java.nio.charset.CharsetDecoder#reset()
237         */
238        public final CharsetDecoder reset() {
239                pending.clear();
240                return backing.reset();
241        }
242
243        /**
244         * @return the result
245         * @see java.lang.Object#toString()
246         */
247        public String toString() {
248                return backing.toString();
249        }
250
251        /**
252         * @return the result
253         * @see java.nio.charset.CharsetDecoder#unmappableCharacterAction()
254         */
255        public CodingErrorAction unmappableCharacterAction() {
256                return backing.unmappableCharacterAction();
257        }
258
259        /* (non-Javadoc)
260         * @see java.lang.Object#hashCode()
261         */
262        @Override
263        public int hashCode() {
264                return backing.hashCode();
265        }
266
267        /* (non-Javadoc)
268         * @see java.lang.Object#equals(java.lang.Object)
269         */
270        @Override
271        public boolean equals(Object obj) {
272                if (this == obj)
273                        return true;
274                if (obj == null)
275                        return false;
276                if (getClass() != obj.getClass())
277                        return false;
278                OptimizedCharsetDecoder other = (OptimizedCharsetDecoder) obj;
279                if (backing == null) {
280                        if (other.backing != null)
281                                return false;
282                } else if (!backing.equals(other.backing))
283                        return false;
284                return true;
285        }
286}
