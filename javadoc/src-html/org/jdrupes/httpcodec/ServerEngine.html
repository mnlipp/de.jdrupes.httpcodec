019package org.jdrupes.httpcodec;
020
021import java.nio.Buffer;
022import java.nio.ByteBuffer;
023import java.util.Optional;
024
025import org.jdrupes.httpcodec.Codec.ProtocolSwitchResult;
026
027/**
028 * An engine that can be used as a server. It has an associated
029 * request decoder and response encoder.
030 * 
031 * @param <Q> the message header type handled by the decoder (the request)
032 * @param <R> the message header type handled be the encoder (the response)
033 */
034public class ServerEngine<Q extends MessageHeader, R extends MessageHeader>
035        extends Engine {
036
037        private Decoder<?, ?> requestDecoder;
038        private Encoder<?> responseEncoder;
039        
040        /**
041         * Creates a new instance.
042         * 
043         * @param requestDecoder the decoder for the request
044         * @param responseEncoder the encoder for the response
045         */
046        public ServerEngine(
047                        Decoder<Q, R> requestDecoder, Encoder<R> responseEncoder) {
048                this.requestDecoder = requestDecoder;
049                this.responseEncoder = responseEncoder;
050        }
051
052        /**
053         * Returns the request decoder.
054         * 
055         * @return the request decoder
056         */
057        @SuppressWarnings("unchecked")
058        public Decoder<Q, R> requestDecoder() {
059                return (Decoder<Q, R>)requestDecoder;
060        }
061
062        /**
063         * Returns the response encoder.
064         * 
065         * @return the response encoder
066         */
067        @SuppressWarnings("unchecked")
068        public Encoder<R> responseEncoder() {
069                return (Encoder<R>)responseEncoder;
070        }
071
072        /**
073         * Returns the type of the messages decoded by this server.
074         * 
075         * @return the value
076         */
077        public Class<? extends MessageHeader> decoding() {
078                return requestDecoder.decoding();
079        }
080        
081        /**
082         * Returns the type of the messages encoded by this server.
083         * 
084         * @return the value
085         */
086        public Class<? extends MessageHeader> encoding() {
087                return responseEncoder.encoding();
088        }
089        
090        /**
091         * Decodes a request sent to the server.
092         * 
093         * @param in the data to decode
094         * @param out the decoded data
095         * @param endOfInput {@code true} if this invocation finishes the message
096         * @return the result
097         * @throws ProtocolException if the input violates the protocol
098         */
099        @SuppressWarnings("unchecked")
100        public Decoder.Result<R> decode(
101                ByteBuffer in, Buffer out, boolean endOfInput)
102                        throws ProtocolException {
103                return (Decoder.Result<R>)requestDecoder.decode(in, out, endOfInput);
104        }
105
106        /**
107         * Encodes a response generated by the server. This method must be used
108         * instead of the encoder's method if the encoder and decoder should adapt
109         * to a protocol switch automatically.
110         * 
111         * @param messageHeader
112         *            the message header
113         * @see Encoder#encode(MessageHeader)
114         */
115        @SuppressWarnings("unchecked")
116        public void encode(R messageHeader) {
117                ((Encoder<R>)responseEncoder).encode(messageHeader);
118        }
119
120        /**
121         * Invokes the encoder's encode method. This method must be used instead of
122         * encoder's method if the encoder and decoder should adapt to a protocol
123         * switch automatically.
124         * 
125         * @param out
126         *            the decoded data
127         * @return the result
128         * @see Encoder#encode(ByteBuffer)
129         */
130        public Encoder.Result encode(
131                ByteBuffer out) {
132                return encode(Codec.EMPTY_IN, out, true);
133        }
134
135        /**
136         * Invokes the encoder's encode method. This method must be used
137         * instead of decoding the encoder's method directly to allow derived
138         * server classes to adapt to any information contained in the message.
139         * 
140         * @param in the data to encode
141         * @param out the encoded data
142         * @param endOfInput {@code true} if this invocation finishes the message
143         * @return the result
144         * @see Encoder#encode(Buffer, ByteBuffer, boolean)
145         */
146        public Encoder.Result encode(
147                Buffer in, ByteBuffer out, boolean endOfInput) {
148                Encoder.Result result = responseEncoder.encode(in, out, endOfInput);
149                if (result instanceof ProtocolSwitchResult) {
150                        ProtocolSwitchResult res = (ProtocolSwitchResult)result;
151                        if (res.newProtocol() != null) {
152                                requestDecoder = res.newDecoder();
153                                responseEncoder = res.newEncoder();
154                        }
155                }
156                return result;
157        }
158
159        /**
160         * Returns the last fully decoded request if it exists.
161         * 
162         * @return the request
163         */
164        @SuppressWarnings("unchecked")
165        public Optional<Q> currentRequest() {
166                return (Optional<Q>)requestDecoder.header();
167        }
168        
169}
